# 智能体什么时候应该触发检索？

## 面试问题

智能体什么时候应该触发检索？

## 考察重点

- 是否理解检索触发的多维度判断逻辑
- 是否能设计自适应的检索策略
- 是否考虑检索的成本和延迟影响
- 是否有基于任务类型的差异化策略

## 核心结论

检索触发决策不是简单的"有知识需求就检索"，而是一个需要权衡检索收益、检索成本、任务紧急度、上下文容量的综合决策。好的检索策略应该根据任务类型、问题特征、上下文状态动态调整触发时机。

## 参考回答

一个客服机器人的场景能说明问题复杂性：

**场景1**：用户问"你们公司地址在哪？"
- 这是事实性问题，答案相对稳定
- 如果已有检索结果，直接返回；如果没有，应该检索

**场景2**：用户问"这个bug怎么修？"
- 需要结合具体代码上下文
- 可能需要检索类似问题的解决方案

**场景3**：用户说"帮我重构这段代码"
- 任务导向，不是知识查询
- 检索可能帮助理解代码模式，但核心是执行任务

**场景4**：用户说"继续"
- 需要结合历史上下文判断用户意图
- 可能是在等待上一个操作的反馈

这个例子说明：不同类型的请求，检索的必要性差异很大。

### 检索触发决策框架

```python
class RetrievalTriggerDecision:
    def __init__(self):
        self.query_classifier = QueryClassifier()
        self.context_analyzer = ContextAnalyzer()
        self.retrieval_decider = RetrievalDecider()
    
    def should_retrieve(
        self,
        user_input: UserInput,
        current_context: AgentContext,
        task_state: TaskState
    ) -> RetrievalDecision:
        """
        综合判断是否触发检索
        """
        # 1. 任务类型判断
        task_type = self.query_classifier.classify(user_input.text)
        
        # 2. 上下文相关性评估
        context_relevance = self.context_analyzer.evaluate_relevance(
            existing_context=current_context,
            query=user_input.text
        )
        
        # 3. 检索收益预估
        retrieval_benefit = self._estimate_retrieval_benefit(
            task_type=task_type,
            query=user_input.text,
            context_relevance=context_relevance
        )
        
        # 4. 检索成本评估
        retrieval_cost = self._estimate_retrieval_cost(
            query=user_input.text,
            current_context=current_context
        )
        
        # 5. 综合决策
        decision = self.retrieval_decider.decide(
            benefit=retrieval_benefit,
            cost=retrieval_cost,
            task_type=task_type,
            task_state=task_state
        )
        
        return decision

@dataclass
class RetrievalDecision:
    should_retrieve: bool
    confidence: float
    strategy: str  # "eager", "lazy", "conditional", "skip"
    retrieval_type: str  # "knowledge", "context", "example"
    query_modification: str = None
    priority: int = 0
    reasoning: str = ""
```

### 查询分类与检索必要性

```python
class QueryClassifier:
    KNOWLEDGE_QUERY_TYPES = {
        "factual",      # 事实性问题 - "什么是微服务"
        "definitional", # 定义性问题 - "解释一下REST API"
        "procedural",   # 过程性问题 - "如何配置Docker"
        "explanatory",  # 解释性问题 - "为什么会出现内存泄漏"
    }
    
    TASK_EXECUTION_TYPES = {
        "code_generation",    # 代码生成
        "code_modification",  # 代码修改
        "refactoring",       # 重构
        "testing",           # 测试
        "deployment",        # 部署
    }
    
    CONVERSATIONAL_TYPES = {
        "clarification",      # 澄清请求
        "follow_up",          # 跟进问题
        "acknowledgment",     # 确认/反馈
        "abandonment",        # 放弃/终止
    }
    
    def classify(self, query: str) -> QueryType:
        """
        分类用户查询类型
        """
        # 1. 规则匹配快速分类
        quick_match = self._rule_based_match(query)
        if quick_match:
            return quick_match
        
        # 2. 小模型分类（低成本）
        embedding = self._encode(query)
        predicted_type = self._classifier.predict(embedding)
        
        return QueryType(
            category=predicted_type.category,
            subcategory=predicted_type.subcategory,
            retrieval_helpful=predicted_type.retrieval_helpful,
            confidence=predicted_type.confidence
        )
    
    def _rule_based_match(self, query: str) -> Optional[QueryType]:
        """基于规则的快速匹配"""
        query_lower = query.lower().strip()
        
        # 事实性查询模式
        factual_patterns = [
            r"^(what|who|where|when|why|how)\s+",
            r"^(什么是|谁是|在哪|怎么|为什么|如何)\s+",
            r"^(define|explain|describe|告诉我)\s+",
        ]
        for pattern in factual_patterns:
            if re.match(pattern, query_lower):
                return QueryType(
                    category="knowledge",
                    subcategory="factual",
                    retrieval_helpful=True,
                    confidence=0.9
                )
        
        # 任务执行模式
        task_patterns = [
            r"^(帮我|write|create|generate|build|make)\s+.*(code|function|class|api)",
            r"^(fix|repair|debug)\s+",
            r"^(refactor|optimize|improve)\s+",
        ]
        for pattern in task_patterns:
            if re.match(pattern, query_lower):
                return QueryType(
                    category="task",
                    subcategory="code_generation",
                    retrieval_helpful=False,  # 任务执行不依赖检索
                    confidence=0.85
                )
        
        # 继续/跟随模式
        continuation_patterns = [
            r"^(继续|continue|go ahead|proceed)",
            r"^(next|then|接下来)",
            r"^(keep|保持|维持)",
        ]
        for pattern in continuation_patterns:
            if re.match(pattern, query_lower):
                return QueryType(
                    category="conversational",
                    subcategory="continuation",
                    retrieval_helpful=False,
                    confidence=0.8
                )
        
        return None
```

### 上下文相关性评估

```python
class ContextAnalyzer:
    def __init__(self):
        self.relevance_threshold = 0.3
        self.completeness_threshold = 0.7
    
    def evaluate_relevance(
        self,
        existing_context: AgentContext,
        query: str
    ) -> RelevanceAssessment:
        """
        评估现有上下文对当前查询的相关性
        """
        # 1. 主题相关性
        topic_overlap = self._calculate_topic_overlap(
            existing_context.topics,
            self._extract_topics(query)
        )
        
        # 2. 实体一致性
        entity_consistency = self._check_entity_consistency(
            existing_context.entities,
            self._extract_entities(query)
        )
        
        # 3. 时间相关性
        temporal_relevance = self._evaluate_temporal_relevance(
            existing_context.time_window,
            query
        )
        
        # 4. 上下文完整性
        context_completeness = self._assess_completeness(
            existing_context,
            query
        )
        
        overall_relevance = (
            topic_overlap * 0.4 +
            entity_consistency * 0.3 +
            temporal_relevance * 0.1 +
            context_completeness * 0.2
        )
        
        return RelevanceAssessment(
            overall=overall_relevance,
            topic=topic_overlap,
            entity=entity_consistency,
            temporal=temporal_relevance,
            completeness=context_completeness,
            sufficient_for_answer=overall_relevance >= self.relevance_threshold
        )
    
    def _calculate_topic_overlap(
        self,
        context_topics: Set[str],
        query_topics: Set[str]
    ) -> float:
        """计算主题重叠度"""
        if not context_topics or not query_topics:
            return 0.0
        return len(context_topics & query_topics) / len(query_topics)
    
    def _check_entity_consistency(
        self,
        context_entities: Dict[str, Entity],
        query_entities: Dict[str, Entity]
    ) -> float:
        """检查实体一致性"""
        if not query_entities:
            return 1.0  # 无实体，无冲突
        
        if not context_entities:
            return 0.0  # 无上下文，必然不相关
        
        consistent_count = 0
        for entity_name, entity in query_entities.items():
            if entity_name in context_entities:
                if context_entities[entity_name].type == entity.type:
                    consistent_count += 1
            else:
                consistent_count += 1  # 新实体不算冲突
        
        return consistent_count / len(query_entities)
    
    def _assess_completeness(
        self,
        context: AgentContext,
        query: str
    ) -> float:
        """评估上下文的完整性是否满足回答需求"""
        required_aspects = self._identify_required_aspects(query)
        
        if not required_aspects:
            return 1.0
        
        covered_aspects = 0
        for aspect in required_aspects:
            if self._is_aspect_covered(context, aspect):
                covered_aspects += 1
        
        return covered_aspects / len(required_aspects)
```

### 检索收益与成本权衡

```python
class RetrievalCostBenefitAnalyzer:
    def __init__(self):
        self.base_latency_cost = 0.3
        self.token_cost_factor = 0.0001
        self.quality_bonus = 0.5
    
    def estimate_benefit(
        self,
        task_type: QueryType,
        query: str,
        context_relevance: RelevanceAssessment
    ) -> BenefitEstimate:
        """
        预估检索的收益
        """
        # 基础收益取决于查询类型
        base_benefit = {
            "knowledge": 0.9,
            "task": 0.3,
            "conversational": 0.2
        }.get(task_type.category, 0.5)
        
        # 上下文不足时收益更高
        context_deficiency = 1.0 - context_relevance.overall
        deficiency_multiplier = 1.0 + context_deficiency
        
        # 答案稳定性收益（知识类问题检索更稳定）
        stability_bonus = 0.2 if task_type.category == "knowledge" else 0.0
        
        # 幻觉风险降低（检索增强可减少幻觉）
        hallucination_reduction = 0.15 if task_type.category == "knowledge" else 0.0
        
        total_benefit = (
            base_benefit * deficiency_multiplier +
            stability_bonus +
            hallucination_reduction
        )
        
        return BenefitEstimate(
            total=min(total_benefit, 1.0),
            base=base_benefit,
            context_deficiency=context_deficiency,
            stability=stability_bonus,
            hallucination_reduction=hallucination_reduction
        )
    
    def estimate_cost(
        self,
        query: str,
        current_context: AgentContext,
        retrieval_system: RetrievalSystem
    ) -> CostEstimate:
        """
        预估检索的成本
        """
        # 1. 延迟成本（基于历史数据预估）
        estimated_latency = retrieval_system.estimate_latency(query)
        latency_cost = estimated_latency * self.base_latency_cost
        
        # 2. Token成本
        estimated_tokens = retrieval_system.estimate_result_tokens(query)
        token_cost = estimated_tokens * self.token_cost_factor
        
        # 3. 上下文膨胀成本
        context_overhead = self._estimate_context_overhead(
            current_context,
            estimated_tokens
        )
        
        # 4. 用户等待耐心（基于任务类型）
        patience_threshold = {
            "knowledge": 2.0,   # 用户愿意等知识查询
            "task": 5.0,        # 任务执行用户愿意多等
            "conversational": 1.0  # 对话中断太长时间体验差
        }.get(current_context.current_task_type, 2.0)
        
        return CostEstimate(
            latency_seconds=estimated_latency,
            latency_cost=latency_cost,
            token_cost=token_cost,
            context_overhead=context_overhead,
            within_patience=estimated_latency <= patience_threshold,
            patience_threshold=patience_threshold,
            total_cost=latency_cost + token_cost + context_overhead
        )
    
    def should_retrieve(
        self,
        benefit: BenefitEstimate,
        cost: CostEstimate,
        confidence_threshold: float = 0.6
    ) -> bool:
        """
        基于收益成本决定是否检索
        """
        # 收益大于成本一定检索
        if benefit.total > cost.total_cost * 1.5:
            return True
        
        # 成本大于收益一定不检索
        if cost.total_cost > benefit.total * 1.5:
            return False
        
        # 接近边界时看置信度
        net_value = benefit.total - cost.total_cost
        confidence_adjusted = net_value * (1.0 / confidence_threshold)
        
        return confidence_adjusted > 0.1
```

### Claude Code的检索触发策略

Claude Code根据代码补全和任务执行的实际需求，采用了混合触发策略：

```python
class ClaudeCodeRetrievalTrigger:
    def __init__(self):
        self.triggers = [
            ImportCompletionTrigger(),      # import语句补全
            DocumentationTrigger(),          # 文档查询
            ErrorResolutionTrigger(),        # 错误解决
            TestGenerationTrigger(),          # 测试生成
            RefactoringTrigger(),            # 重构参考
        ]
    
    def check_triggers(
        self,
        cursor_context: CursorContext,
        buffer_state: BufferState
    ) -> List[RetrievalTrigger]:
        """
        检查所有触发条件
        """
        active_triggers = []
        
        for trigger in self.triggers:
            if trigger.should_activate(cursor_context, buffer_state):
                active_triggers.append(RetrievalTrigger(
                    type=trigger.type,
                    priority=trigger.priority,
                    query=trigger.generate_query(cursor_context),
                    urgency=trigger.estimate_urgency(cursor_context)
                ))
        
        # 按优先级排序，合并相似查询
        return self._merge_and_sort_triggers(active_triggers)
    
    def _merge_and_sort_triggers(
        self,
        triggers: List[RetrievalTrigger]
    ) -> List[RetrievalTrigger]:
        """合并相似触发器并排序"""
        # 按类型分组
        by_type = defaultdict(list)
        for t in triggers:
            by_type[t.type].append(t)
        
        # 每类只保留最高优先级的
        merged = []
        for t_type, type_triggers in by_type.items():
            best = max(type_triggers, key=lambda t: t.priority * t.urgency)
            merged.append(best)
        
        return sorted(merged, key=lambda t: t.priority * t.urgency, reverse=True)


class ImportCompletionTrigger:
    type = "import_completion"
    priority = 0.9
    
    def should_activate(
        self,
        cursor: CursorContext,
        buffer: BufferState
    ) -> bool:
        """检测是否在写import语句"""
        line = buffer.get_line(cursor.line_number)
        return (
            cursor.in_comment_or_string or
            "import " in line or
            "from " in line
        ) and not buffer.has_recent_import_completion(cursor.line_number)
    
    def generate_query(self, cursor: CursorContext) -> str:
        """生成检索query"""
        return f"import {cursor.current_word} python"


class ErrorResolutionTrigger:
    type = "error_resolution"
    priority = 0.95  # 错误解决高优先级
    
    def should_activate(
        self,
        cursor: CursorContext,
        buffer: BufferState
    ) -> bool:
        """检测是否在错误上下文中"""
        errors = buffer.get_nearby_errors(cursor.line_number)
        return len(errors) > 0
    
    def generate_query(self, cursor: CursorContext) -> str:
        """生成错误解决查询"""
        error = buffer.get_nearby_errors(cursor.line_number)[0]
        return f"{error.error_type} python solution"
```

### 动态检索策略

```python
class AdaptiveRetrievalStrategy:
    def __init__(self):
        self.strategy_profiles = {
            "eager": EagerRetrievalStrategy(),   # 预检索
            "lazy": LazyRetrievalStrategy(),      # 延迟检索
            "conditional": ConditionalStrategy(), # 条件检索
            "opportunistic": OpportunisticStrategy()  # 机会检索
        }
    
    def select_strategy(
        self,
        task_type: str,
        system_load: SystemLoad,
        user_patience: UserPatience
    ) -> RetrievalStrategy:
        """
        根据当前状态选择检索策略
        """
        # 高负载时优先选择延迟检索
        if system_load.is_high():
            return self.strategy_profiles["lazy"]
        
        # 高耐心用户可以用预检索
        if user_patience.is_high() and task_type == "knowledge":
            return self.strategy_profiles["eager"]
        
        # 默认条件检索
        return self.strategy_profiles["conditional"]


class EagerRetrievalStrategy(RetrievalStrategy):
    """
    预检索策略：在用户表达意图早期就触发检索
    适用于：知识密集型任务、复杂分析任务
    """
    def should_retrieve(
        self,
        context: AgentContext,
        decision: RetrievalDecision
    ) -> bool:
        # 只要有知识需求就检索
        return decision.task_type.category == "knowledge"


class LazyRetrievalStrategy(RetrievalStrategy):
    """
    延迟检索策略：尽可能复用上下文，只在必要时检索
    适用于：高负载系统、低耐心用户、快速任务
    """
    def should_retrieve(
        self,
        context: AgentContext,
        decision: RetrievalDecision
    ) -> bool:
        # 只有上下文明显不足且检索收益高时才检索
        return (
            decision.context_relevance.overall < 0.3 and
            decision.retrieval_benefit.total > 0.7
        )


class OpportunisticStrategy(RetrievalStrategy):
    """
    机会检索策略：利用空闲时间预取可能需要的内容
    适用于：流式交互、代码补全场景
    """
    def __init__(self):
        self.prefetch_queue = []
    
    def should_retrieve(
        self,
        context: AgentContext,
        decision: RetrievalDecision
    ) -> bool:
        # 非阻塞式预取
        if context.system_load < 0.5 and len(self.prefetch_queue) < 3:
            self.prefetch_queue.append(decision)
            return False
        return True
```

```
用户输入
    │
    ▼
查询分类
    │
    ├── 知识类查询 ──→ 高检索收益 ──→ 检查上下文相关性
    │                            │
    │                            ├── 上下文不足 ──→ 触发检索
    │                            └── 上下文充足 ──→ 可选检索

    ├── 任务类查询 ──→ 低检索收益 ──→ 通常不检索

    │
    └── 对话类查询 ──→ 检查是否是继续/跟随
                        │
                        ├── 是 ──→ 检查历史上下文
                        │       │
                        │       ├── 有未完成任务 ──→ 可选检索
                        │       └── 无任务 ──→ 不检索
                        │
                        └── 否 ──→ 按普通查询处理
```

### 检索触发监控系统

```python
class RetrievalTriggerMonitor:
    def __init__(self):
        self.decision_log = deque(maxlen=10000)
    
    def record_decision(
        self,
        query: str,
        decision: RetrievalDecision,
        actual_outcome: RetrievalOutcome = None
    ):
        """记录检索决策及结果"""
        self.decision_log.append({
            "timestamp": time.time(),
            "query_hash": hash(query),
            "decision": decision,
            "outcome": actual_outcome
        })
    
    def analyze_trigger_patterns(self) -> Dict:
        """分析触发模式"""
        trigger_counts = defaultdict(int)
        skip_counts = defaultdict(int)
        
        for entry in self.decision_log:
            if entry["decision"].should_retrieve:
                trigger_counts[entry["decision"].strategy] += 1
            else:
                skip_counts[entry["decision"].reasoning[:50]] += 1
        
        return {
            "triggered_strategies": dict(trigger_counts),
            "skip_reasons": dict(skip_counts),
            "trigger_rate": len(trigger_counts) / len(self.decision_log)
        }
    
    def detect_trigger_anomalies(self) -> List[AnomalyReport]:
        """检测触发异常"""
        anomalies = []
        
        # 频繁触发但效果不好
        recent = list(self.decision_log)[-1000:]
        high_trigger_low_success = self._find_high_trigger_low_success(recent)
        if high_trigger_low_success:
            anomalies.append(AnomalyReport(
                type="inefficient_retrieval",
                details=high_trigger_low_success
            ))
        
        return anomalies
```

## 常见误区

- **一刀切策略**：所有查询都用同一种检索策略
- **忽视上下文**：只管当前查询，不管历史上下文
- **忽视成本**：不考虑检索延迟和token消耗
- **过度检索**：以为检索越多越好，导致上下文膨胀

## 追问方向

- 如何在检索收益和成本之间找到平衡点？
- 多个检索触发条件冲突时如何决策？
- 如何处理用户意图的模糊性（不确定是知识查询还是任务执行）？
- 如何评估检索策略的有效性？

## 学习建议

建议动手实现一个检索触发模拟器：用真实查询数据测试不同策略下的触发率和延迟，选择适合自己场景的策略。重点理解"条件检索"的概念：根据任务类型、上下文状态、系统负载动态调整触发条件。
