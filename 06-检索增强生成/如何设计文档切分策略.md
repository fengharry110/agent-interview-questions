# 如何设计文档切分策略？

## 面试问题

如何设计文档切分策略？

## 考察重点

- 是否理解文档切分对检索效果的影响
- 能否说清楚不同切分策略的适用场景
- 是否能设计保持语义完整的切分方案
- 是否理解chunk size对检索和生成的影响

## 核心结论

文档切分的核心挑战是在"保留完整语义"和"控制chunk大小"之间取得平衡。切分策略取决于：文档结构、内容类型、检索场景和下游任务。

## 切分粒度的影响

```
Chunk大小对比：

过小（<100 tokens）：
- 优点：检索精准、噪声少
- 缺点：上下文不完整、需要更多chunk、生成时需要更多引用

过大（>1000 tokens）：
- 优点：语义完整、上下文丰富
- 缺点：检索精度下降、可能引入噪声、模型处理负担大

适中（200-500 tokens）：
- 平衡精度和完整性的合理区间
```

## 基础切分策略

### 1. 固定长度切分

```python
class FixedSizeChunker:
    def __init__(self, chunk_size: int = 500, overlap: int = 50):
        self.chunk_size = chunk_size
        self.overlap = overlap
    
    def chunk(self, text: str) -> List[Chunk]:
        tokens = self.tokenize(text)
        chunks = []
        
        for i in range(0, len(tokens), self.chunk_size - self.overlap):
            chunk_tokens = tokens[i:i + self.chunk_size]
            chunk_text = self.detokenize(chunk_tokens)
            
            chunks.append(Chunk(
                content=chunk_text,
                start_token=i,
                end_token=i + len(chunk_tokens),
                token_count=len(chunk_tokens)
            ))
        
        return chunks
```

### 2. 句子级别切分

```python
class SentenceChunker:
    def __init__(self, max_tokens: int = 500, min_sentences: int = 1):
        self.max_tokens = max_tokens
        self.min_sentences = min_sentences
    
    def chunk(self, text: str) -> List[Chunk]:
        sentences = self.split_sentences(text)
        chunks = []
        current_chunk = []
        current_tokens = 0
        
        for sentence in sentences:
            sentence_tokens = self.count_tokens(sentence)
            
            if current_tokens + sentence_tokens > self.max_tokens and current_chunk:
                chunks.append(self.create_chunk(current_chunk))
                current_chunk = [sentence]
                current_tokens = sentence_tokens
            else:
                current_chunk.append(sentence)
                current_tokens += sentence_tokens
        
        if current_chunk:
            chunks.append(self.create_chunk(current_chunk))
        
        return chunks
    
    def split_sentences(self, text: str) -> List[str]:
        # 使用NLP库分句
        return nltk.sent_tokenize(text)
```

## 语义感知切分

### 3. 段落感知切分

```python
class ParagraphAwareChunker:
    def __init__(self, max_tokens: int = 500, max_paragraphs: int = 5):
        self.max_tokens = max_tokens
        self.max_paragraphs = max_paragraphs
    
    def chunk(self, text: str) -> List[Chunk]:
        paragraphs = self.split_paragraphs(text)
        chunks = []
        current_paragraphs = []
        current_tokens = 0
        
        for para in paragraphs:
            para_tokens = self.count_tokens(para)
            
            if current_tokens + para_tokens > self.max_tokens:
                if current_paragraphs:
                    chunks.append(self.create_chunk(current_paragraphs))
                    current_paragraphs = []
                    current_tokens = 0
                
                # 如果单个段落就超过限制，需要切分段落
                if para_tokens > self.max_tokens:
                    sub_chunks = self.chunk_large_paragraph(para)
                    chunks.extend(sub_chunks)
                    continue
            
            current_paragraphs.append(para)
            current_tokens += para_tokens
        
        if current_paragraphs:
            chunks.append(self.create_chunk(current_paragraphs))
        
        return chunks
    
    def split_paragraphs(self, text: str) -> List[str]:
        return [p.strip() for p in text.split('\n\n') if p.strip()]
```

### 4. 文档结构感知切分

```python
class StructureAwareChunker:
    def __init__(self, max_tokens: int = 500):
        self.max_tokens = max_tokens
    
    def chunk(self, document: Document) -> List[Chunk]:
        chunks = []
        
        if document.doc_type == "markdown":
            chunks = self.chunk_markdown(document)
        elif document.doc_type == "html":
            chunks = self.chunk_html(document)
        elif document.doc_type == "code":
            chunks = self.chunk_code(document)
        else:
            chunks = self.chunk_plain_text(document)
        
        return chunks
    
    def chunk_markdown(self, document: Document) -> List[Chunk]:
        # 按标题层级结构切分
        sections = self.parse_markdown_structure(document.content)
        chunks = []
        
        for section in sections:
            if self.count_tokens(section.content) <= self.max_tokens:
                chunks.append(Chunk(
                    content=section.content,
                    heading=section.heading,
                    level=section.level
                ))
            else:
                # 大章节需要进一步切分
                sub_chunks = self.chunk_large_section(section)
                chunks.extend(sub_chunks)
        
        return chunks
    
    def parse_markdown_structure(self, text: str) -> List[Section]:
        sections = []
        lines = text.split('\n')
        current_section = None
        
        for line in lines:
            if self.is_heading(line):
                if current_section:
                    sections.append(current_section)
                level, heading = self.parse_heading(line)
                current_section = Section(level=level, heading=heading, content=line + '\n')
            elif current_section:
                current_section.content += line + '\n'
        
        if current_section:
            sections.append(current_section)
        
        return sections
```

## 代码文档切分

### 特殊处理

```python
class CodeDocumentChunker:
    def chunk(self, document: Document) -> List[Chunk]:
        if document.doc_type == "python":
            return self.chunk_python(document)
        elif document.doc_type == "javascript":
            return self.chunk_javascript(document)
        else:
            return self.chunk_generic_code(document)
    
    def chunk_python(self, document: Document) -> List[Chunk]:
        chunks = []
        
        # 使用AST解析Python代码
        tree = ast.parse(document.content)
        
        for node in ast.walk(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)):
                func_code = ast.get_source_segment(document.content, node)
                if func_code and self.count_tokens(func_code) <= 1000:
                    chunks.append(Chunk(
                        content=func_code,
                        entity_name=node.name,
                        entity_type=type(node).__name__
                    ))
        
        return chunks
```

## 自适应切分策略

### 内容感知的动态chunk大小

```python
class AdaptiveChunker:
    def __init__(self):
        self.base_chunker = SentenceChunker()
        self.content_analyzer = ContentAnalyzer()
    
    def chunk(self, text: str, doc_type: str = "general") -> List[Chunk]:
        # 分析内容特征
        analysis = self.content_analyzer.analyze(text)
        
        if analysis["has_code"]:
            return self.chunk_with_code_strategy(text)
        elif analysis["is_highly_structured"]:
            return self.chunk_with_structure_strategy(text)
        elif analysis["is_narrative"]:
            return self.chunk_with_narrative_strategy(text)
        else:
            return self.base_chunker.chunk(text)
    
    def chunk_with_code_strategy(self, text: str) -> List[Chunk]:
        # 代码密集型文档：按函数/类切分
        code_chunker = CodeDocumentChunker()
        return code_chunker.chunk(text)
    
    def chunk_with_narrative_strategy(self, text: str) -> List[Chunk]:
        # 叙述型文档：较大chunk保持上下文
        return SentenceChunker(max_tokens=800).chunk(text)
```

## Chunk元信息

### 保留上下文信息

```python
@dataclass
class ChunkMetadata:
    chunk_id: str
    doc_id: str
    chunk_index: int  # 在文档中的顺序
    total_chunks: int  # 总chunk数
    
    # 上下文信息
    prev_chunk_summary: str = None  # 前一个chunk的摘要
    next_chunk_summary: str = None  # 后一个chunk的摘要
    
    # 来源信息
    source_heading: str = None  # 来源标题
    source_section: str = None  # 来源章节
    
    # 质量信息
    coherence_score: float = None  # 连贯性评分
    completeness_score: float = None  # 完整性评分

class ChunkWithContextEnhancer:
    def add_context(self, chunks: List[Chunk], document: Document) -> List[Chunk]:
        for i, chunk in enumerate(chunks):
            chunk.metadata.chunk_index = i
            chunk.metadata.total_chunks = len(chunks)
            chunk.metadata.doc_id = document.doc_id
            
            if i > 0:
                chunk.metadata.prev_chunk_summary = self.summarize(chunks[i-1].content)
            
            if i < len(chunks) - 1:
                chunk.metadata.next_chunk_summary = self.summarize(chunks[i+1].content)
        
        return chunks
```

## 切分质量评估

### 评估指标

```python
class ChunkQualityEvaluator:
    def evaluate(self, chunks: List[Chunk]) -> QualityReport:
        return QualityReport(
            coverage=self.calculate_coverage(chunks),
            coherence=self.calculate_coherence(chunks),
            completeness=self.calculate_completeness(chunks),
            efficiency=self.calculate_efficiency(chunks)
        )
    
    def calculate_coherence(self, chunks: List[Chunk]) -> float:
        # chunk间重叠度评估
        # 理想情况：相邻chunk有一定重叠保持连贯
        overlaps = []
        for i in range(len(chunks) - 1):
            overlap = self.compute_overlap(chunks[i], chunks[i+1])
            overlaps.append(overlap)
        
        # 最佳重叠率在10-20%之间
        ideal_overlap = 0.15
        coherence = 1.0 - abs(mean(overlaps) - ideal_overlap)
        return max(0.0, coherence)
    
    def calculate_completeness(self, chunks: List[Chunk]) -> float:
        # 语义完整性评估
        # 检查每个chunk是否包含完整的句子
        incomplete_count = 0
        for chunk in chunks:
            if not self.ends_with_complete_sentence(chunk.content):
                incomplete_count += 1
        
        return 1.0 - (incomplete_count / len(chunks))
```

## 切分策略选择指南

| 文档类型 | 推荐策略 | Chunk大小 | 理由 |
|---------|---------|----------|------|
| 论文/文章 | 段落感知 | 300-500 | 保持段落完整性 |
| 文档/手册 | 结构感知 | 200-400 | 尊重标题层级 |
| 代码库 | 代码感知 | 函数级别 | 保持函数完整性 |
| 对话记录 | 固定长度 | 200-300 | 均匀切分便于检索 |
| 法律文档 | 结构+句子 | 500-800 | 保持条款完整 |

## 常见问题

| 问题 | 原因 | 解决方案 |
|------|------|---------|
| chunk截断了完整句子 | 固定长度边界 | 使用句子边界切分 |
| 检索结果不完整 | chunk太小 | 增加overlap或增大chunk |
| 向量检索效果差 | chunk太大 | 减小chunk大小 |
| 相邻chunk语义重复 | overlap过大 | 调整overlap比例 |
