# 如何设计安全的文件工具？

## 面试问题

如何设计安全的文件工具？

## 考察重点

- 是否理解文件操作的安全风险：路径遍历、符号链接攻击、敏感文件访问
- 是否能设计安全的文件工具，包含路径验证、权限控制、操作审计
- 是否理解沙箱化文件访问的实现方式
- 是否能处理文件操作的各种边界情况

## 核心结论

文件工具是智能体系统中最常用的工具之一，也是安全风险最高的工具。设计不安全的文件工具可能导致：
1. 敏感数据泄露（读取 /etc/passwd、SSH 密钥等）
2. 系统文件破坏（删除系统关键文件）
3. 路径遍历攻击（访问工作目录之外的文件）

安全的文件工具设计核心是：**最小权限原则**（只允许访问必要的文件）、**路径验证**（防止 ../ 遍历）、**沙箱隔离**（隔离危险操作）和**操作审计**（记录所有文件操作）。

## 安全风险分析

### 主要安全风险

| 风险类型 | 攻击方式 | 危害程度 | 防御措施 |
|---------|---------|---------|---------|
| 路径遍历 | `../../etc/passwd` | 高 | 路径规范化 + 边界检查 |
| 符号链接 | 链接指向敏感文件 | 高 | 检测并拒绝符号链接 |
| 通配符滥用 | `rm -rf *` | 高 | 禁止高风险操作的通配符 |
| 软链接循环 | 循环符号链接 | 中 | 检测循环引用 |
| 大文件攻击 | 读取超大文件 | 中 | 文件大小限制 |
| 敏感文件访问 | 读取 SSH 密钥、配置 | 高 | 白名单/黑名单机制 |

### 攻击场景示例

```python
# 路径遍历攻击
malicious_input = "../../../etc/passwd"
# 如果没有验证，可能读取到系统密码文件

# 符号链接攻击
os.symlink("/etc/passwd", "sensitive_data.txt")
# 模型以为在读普通文本，实际读取了密码文件

# 大文件攻击
file_to_read = "very_large_file.log"  # 10GB 文件，可能导致系统资源耗尽
```

## 安全防护实现

### 路径安全验证

```python
class SecurePathValidator:
    def __init__(self, allowed_base_dirs: list[str]):
        self.allowed_base_dirs = [os.path.abspath(d) for d in allowed_base_dirs]

    def validate(self, path: str) -> ValidationResult:
        # 1. 路径规范化（解析 .. 和符号链接）
        try:
            normalized = self.normalize_path(path)
        except ValueError as e:
            return ValidationResult(valid=False, reason=str(e))

        # 2. 检查是否为绝对路径
        if os.path.isabs(normalized) and not self.is_allowed_absolute(normalized):
            return ValidationResult(valid=False, reason="Absolute paths not allowed")

        # 3. 检查是否在允许目录内
        if not self.is_within_allowed_dirs(normalized):
            return ValidationResult(valid=False, reason="Path outside allowed directories")

        # 4. 检查是否为符号链接
        if os.path.islink(normalized):
            return ValidationResult(valid=False, reason="Symbolic links not allowed")

        return ValidationResult(valid=True, normalized_path=normalized)

    def normalize_path(self, path: str) -> str:
        # 解析 ..、. 等路径元素
        normalized = os.path.normpath(path)
        # 解析符号链接（仅解析一次，避免循环）
        real_path = os.path.realpath(normalized)
        return real_path

    def is_within_allowed_dirs(self, path: str) -> bool:
        path = os.path.abspath(path)
        for allowed_dir in self.allowed_base_dirs:
            if path.startswith(allowed_dir + os.sep) or path == allowed_dir:
                return True
        return False

    def is_allowed_absolute(self, path: str) -> bool:
        # 只有特定绝对路径才允许
        allowed_abs = ["/tmp", "/var/tmp"]
        return any(path.startswith(allowed) for allowed in allowed_abs)
```

### 文件操作权限控制

```python
class FilePermissionController:
    PERMISSION_MATRIX = {
        "read": {
            "allowed_extensions": {".txt", ".md", ".py", ".js", ".json", ".yaml", ".yml", ".toml", ".cfg", ".conf"},
            "denied_paths": {
                "/etc", "/var/log", "/root", "/home/*/.ssh",
                "*.key", "*.pem", "*.pkcs8", "id_rsa*", "id_ed25519*"
            },
            "max_file_size": 10 * 1024 * 1024,  # 10MB
        },
        "write": {
            "allowed_extensions": {".txt", ".md", ".py", ".js", ".json", ".yaml", ".yml", ".toml"},
            "denied_paths": {"/etc", "/var", "/usr", "/bin", "/sbin", "/root"},
            "max_file_size": 5 * 1024 * 1024,  # 5MB
            "require_confirmation": True
        },
        "delete": {
            "allowed_extensions": set(),
            "denied_paths": {"*"},  # 禁止删除任何文件
            "require_confirmation": True,
            "require_explicit_path": True  # 不允许通配符
        }
    }

    def check_permission(self, operation: str, path: str, file_size: int = 0) -> PermissionResult:
        if operation not in self.PERMISSION_MATRIX:
            return PermissionResult(allowed=False, reason=f"Unknown operation: {operation}")

        rules = self.PERMISSION_MATRIX[operation]

        # 检查扩展名
        ext = os.path.splitext(path)[1].lower()
        if rules["allowed_extensions"] and ext not in rules["allowed_extensions"]:
            return PermissionResult(allowed=False, reason=f"Extension {ext} not allowed for {operation}")

        # 检查敏感路径
        for denied in rules["denied_paths"]:
            if self.path_matches_pattern(path, denied):
                return PermissionResult(allowed=False, reason=f"Path matches denied pattern: {denied}")

        # 检查文件大小
        if "max_file_size" in rules and file_size > rules["max_file_size"]:
            return PermissionResult(allowed=False, reason=f"File size {file_size} exceeds limit")

        return PermissionResult(
            allowed=True,
            requires_confirmation=rules.get("require_confirmation", False)
        )

    def path_matches_pattern(self, path: str, pattern: str) -> bool:
        # 简化实现，实际使用 fnmatch 或正则
        path = os.path.abspath(path)
        if pattern.startswith("*."):
            ext = pattern[1:]
            return path.endswith(ext)
        if "*" in pattern:
            import fnmatch
            return fnmatch.fnmatch(path, pattern)
        return pattern in path
```

### 审计日志

```python
class FileOperationAuditor:
    def __init__(self, storage: AuditStorage):
        self.storage = storage

    async def log_operation(self, operation: FileOperation):
        log = FileOperationLog(
            timestamp=datetime.now(),
            operation=operation.type,  # read, write, delete
            path=operation.path,
            normalized_path=operation.normalized_path,
            user_id=operation.context.user_id,
            task_id=operation.context.task_id,
            session_id=operation.context.session_id,
            success=operation.success,
            error=operation.error,
            bytes_processed=operation.bytes_processed
        )
        await self.storage.insert(log)

    def should_block(self, operation: FileOperation) -> bool:
        # 高风险操作需要额外确认
        if operation.type == "delete":
            return True
        if operation.type == "write" and operation.path.startswith("/etc"):
            return True
        return False
```

## Claude Code 的文件安全实现

### Read 工具的安全检查

Claude Code 的 `Read` 工具实现了多层安全检查：

```typescript
// file_read.ts
async function safeRead(path: string, context: ExecutionContext): Promise<ToolResult> {
  // 1. 路径验证
  const normalized = path.normalize();
  if (!isWithinWorkspace(normalized)) {
    return { error: "Path outside workspace", tool: "Read" };
  }

  // 2. 敏感路径检查
  if (isSensitivePath(normalized)) {
    return { error: "Access to sensitive path denied", tool: "Read" };
  }

  // 3. 文件大小检查
  const stats = await fs.stat(normalized);
  if (stats.size > MAX_FILE_SIZE) {
    return { error: "File too large", tool: "Read", metadata: { maxSize: MAX_FILE_SIZE } };
  }

  // 4. 符号链接检查
  if (await fs.lstat(normalized).then(s => s.isSymbolicLink())) {
    return { error: "Symbolic links not allowed", tool: "Read" };
  }

  // 5. 执行读取
  const content = await fs.readFile(normalized, "utf-8");
  return { result: content, tool: "Read", metadata: { size: stats.size, lines: content.split('\n').length } };
}
```

## 工程实践 Checklist

设计安全文件工具时的 Checklist：

- [ ] **路径验证**：是否实现了路径规范化和边界检查
- [ ] **符号链接处理**：是否拒绝了符号链接或实现了安全处理
- [ ] **敏感路径保护**：是否拦截了对 /etc、/root、.ssh 等敏感路径的访问
- [ ] **文件大小限制**：是否限制了单文件大小和总处理大小
- [ ] **扩展名过滤**：是否禁止了危险扩展名（.exe、.sh 等）
- [ ] **通配符限制**：是否禁止了危险的通配符操作（如 `rm -rf *`）
- [ ] **操作审计**：是否记录了所有文件操作的审计日志
- [ ] **确认机制**：高风险操作是否需要用户确认

## 真实踩坑案例

### 案例：路径遍历导致敏感文件泄露

某团队的文件读取工具没有验证路径，导致攻击者通过 `../../etc/passwd` 读取了系统密码文件。

**修复方案**：
1. 实施路径规范化，解析所有 .. 元素
2. 检查规范化后的路径是否仍在工作目录内
3. 拒绝符号链接
4. 添加敏感路径黑名单

### 案例：软链接循环导致服务崩溃

某工具在遍历目录时遇到循环符号链接，导致无限循环，服务崩溃。

**修复方案**：
1. 实现符号链接深度限制
2. 检测并跳过循环引用
3. 使用 `realpath` 解析后验证
