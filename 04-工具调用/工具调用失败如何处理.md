## 面试问题

工具调用失败如何处理？

## 考察重点

- 是否能系统性地分析工具调用失败的原因分类
- 是否能设计合理的重试策略和降级机制
- 是否理解错误信息如何反馈给模型以支持恢复
- 是否能结合Claude Code/Codex实现说明工程方法

## 核心结论

工具调用失败处理的核心是：**分类**（区分可恢复/不可恢复）、**策略**（重试/降级/回退）、**反馈**（结构化错误信息帮助模型决策）。

## 参考回答

### 失败分类体系

工具调用失败可分为四类：

| 类别 | 特征 | 处理策略 |
|-----|------|---------|
| **瞬时故障** | 网络抖动、超时 | 指数退避重试 |
| **资源故障** | 限流、熔断打开 | 延迟重试、限流降级 |
| **永久故障** | 参数错误、权限不足 | 快速失败、返回修正建议 |
| **未知故障** | 异常未捕获 | 日志记录、返回通用错误 |

### Claude Code的失败处理

Claude Code的MCP客户端实现了完整的失败处理链：

```typescript
interface RetryConfig {
  maxRetries: number;
  baseDelay: number;
  maxDelay: number;
  jitter: boolean;
}

async function executeWithRetry(
  toolCall: ToolCall,
  config: RetryConfig
): Promise<ToolResult> {
  let lastError: Error;

  for (let attempt = 0; attempt <= config.maxRetries; attempt++) {
    try {
      return await executeTool(toolCall);
    } catch (error) {
      lastError = error;

      if (!isRetryable(error)) {
        throw new NonRetryableError(error);
      }

      if (attempt < config.maxRetries) {
        const delay = calculateBackoff(attempt, config);
        await sleep(delay);
      }
    }
  }

  throw new MaxRetriesExceeded(lastError);
}

function isRetryable(error: Error): boolean {
  if (error instanceof TimeoutError) return true;
  if (error instanceof RateLimitError) return true;
  if (error instanceof NetworkError) return true;
  if (error instanceof ServerError) return true;
  return false;
}
```

### Codex的失败处理

Codex采用状态机管理工具执行状态：

```python
class ToolExecutionMachine:
    states = ["idle", "executing", "waiting_retry", "failed", "success"]

    def __init__(self):
        self.state = "idle"
        self.retry_count = 0

    def transition(self, event: ToolEvent) -> "ToolExecutionMachine":
        if self.state == "executing":
            if event.is_success():
                return self._to_success()
            elif event.is_retryable():
                return self._to_waiting_retry(event.error)
            else:
                return self._to_failed(event.error)

        elif self.state == "waiting_retry":
            if event.is_retry_time():
                return self._to_executing()
            else:
                return self._to_failed(event.error)

        return self

    def _to_waiting_retry(self, error: Error):
        self.state = "waiting_retry"
        self.retry_count += 1
        self.current_error = error
        return self
```

## 深度展开

### 第一层：重试策略设计

**指数退避算法**：

```python
class ExponentialBackoff:
    def __init__(
        self,
        base: float = 1.0,
        max_delay: float = 60.0,
        multiplier: float = 2.0
    ):
        self.base = base
        self.max_delay = max_delay
        self.multiplier = multiplier

    def get_delay(self, attempt: int) -> float:
        delay = self.base * (self.multiplier ** attempt)
        jitter = random.uniform(0.5, 1.5)
        return min(delay * jitter, self.max_delay)

class DecorrelatedJitterBackoff:
    def __init__(self, base_delay: float = 1.0, max_delay: float = 60.0):
        self.base_delay = base_delay
        self.max_delay = max_delay
        self.previous_delay = base_delay

    def get_delay(self, attempt: int) -> float:
        delay = self.previous_delay * 3 * random.uniform(0.5, 1.5)
        delay = min(max(delay, self.base_delay), self.max_delay)
        self.previous_delay = delay
        return delay
```

**重试预算**：

```python
class RetryBudget:
    def __init__(self, max_retries_per_minute: int = 60):
        self.max_retries = max_retries_per_minute
        self.retry_times: deque = deque(maxlen=max_retries_per_minute)

    def can_retry(self) -> bool:
        now = time.time()
        self.retry_times = deque(
            (t for t in self.retry_times if now - t < 60),
            maxlen=self.max_retries
        )
        return len(self.retry_times) < self.max_retries

    def record_retry(self):
        self.retry_times.append(time.time())
```

### 第二层：降级策略

**服务降级层次**：

```python
class DegradationStrategy:
    def __init__(self):
        self.fallback_handlers: Dict[str, FallbackHandler] = {}

    def register_fallback(self, tool_name: str, handler: FallbackHandler):
        self.fallback_handlers[tool_name] = handler

    async def execute_with_fallback(
        self,
        tool_call: ToolCall
    ) -> ToolResult:
        try:
            return await self.execute_primary(tool_call)
        except Exception as e:
            if tool_call.name in self.fallback_handlers:
                return await self.fallback_handlers[tool_call.name].execute(
                    tool_call, e
                )
            raise

class FallbackHandler(ABC):
    @abstractmethod
    async def execute(
        self,
        tool_call: ToolCall,
        primary_error: Exception
    ) -> ToolResult:
        pass

class CacheFallback(FallbackHandler):
    async def execute(
        self,
        tool_call: ToolCall,
        primary_error: Exception
    ) -> ToolResult:
        cached = await self.cache.get(self._cache_key(tool_call))
        if cached:
            return ToolResult(
                status="success",
                data=cached,
                metadata={"source": "cache", "original_error": str(primary_error)}
            )
        raise primary_error

class MockFallback(FallbackHandler):
    async def execute(
        self,
        tool_call: ToolCall,
        primary_error: Exception
    ) -> ToolResult:
        return ToolResult(
            status="partial",
            data=self._generate_mock(tool_call),
            metadata={"source": "mock", "warning": "Primary service unavailable"}
        )
```

### 第三层：错误信息结构化

**错误分类码体系**：

```python
class ErrorCode:
    TIMEOUT = "E001"
    RATE_LIMIT = "E002"
    AUTH_FAILED = "E003"
    PERMISSION_DENIED = "E004"
    INVALID_PARAMS = "E005"
    RESOURCE_NOT_FOUND = "E006"
    INTERNAL_ERROR = "E007"
    NETWORK_ERROR = "E008"

@dataclass
class StructuredError:
    code: str
    message: str
    details: Optional[dict] = None
    recoverable: bool = True
    retry_after: Optional[float] = None
    suggested_action: Optional[str] = None

    def to_model_message(self) -> str:
        lines = [f"[Error {self.code}] {self.message}"]
        if self.recoverable:
            lines.append(f" recoverable: true")
            if self.retry_after:
                lines.append(f" retry_after: {self.retry_after}s")
        else:
            lines.append(f" recoverable: false")
        if self.suggested_action:
            lines.append(f" suggestion: {self.suggested_action}")
        return "\n".join(lines)
```

**错误上下文累积**：

```python
class ErrorContext:
    def __init__(self):
        self.errors: List[StructuredError] = []
        self.tool_call_chain: List[str] = []

    def add_error(self, error: StructuredError, tool_name: str):
        self.errors.append(error)
        self.tool_call_chain.append(tool_name)

    def get_summary(self) -> str:
        if not self.errors:
            return "No errors"
        lines = ["Error history:"]
        for i, err in enumerate(self.errors):
            lines.append(f"  {i+1}. [{err.code}] in {self.tool_call_chain[i]}: {err.message}")
        return "\n".join(lines)

    def should_abort(self) -> bool:
        permanent_failures = [e for e in self.errors if not e.recoverable]
        return len(permanent_failures) >= 3
```

### 第四层：与模型协同的失败处理

**自适应重试**：

```python
class AdaptiveRetryStrategy:
    def __init__(self):
        self.success_rates: Dict[str, float] = {}
        self.base_delay = 1.0

    def should_retry(
        self,
        tool_name: str,
        attempt: int,
        error: StructuredError
    ) -> bool:
        if not error.recoverable:
            return False

        success_rate = self.success_rates.get(tool_name, 1.0)
        if success_rate < 0.5:
            return attempt < 2

        return attempt < 5

    def record_outcome(self, tool_name: str, success: bool):
        if tool_name not in self.success_rates:
            self.success_rates[tool_name] = 0.9

        alpha = 0.3
        if success:
            self.success_rates[tool_name] = (
                alpha * 1.0 + (1 - alpha) * self.success_rates[tool_name]
            )
        else:
            self.success_rates[tool_name] = (
                (1 - alpha) * self.success_rates[tool_name]
            )
```

**模型友好的错误提示**：

```python
class ModelFriendlyErrorFormatter:
    def format_for_model(
        self,
        error: StructuredError,
        tool_schema: dict,
        previous_attempts: int
    ) -> str:
        if error.code == ErrorCode.INVALID_PARAMS:
            return self._format_param_error(error, tool_schema)
        elif error.code == ErrorCode.PERMISSION_DENIED:
            return f"[Permission Denied] {error.message}. The selected tool may not have the required permissions."
        elif error.code == ErrorCode.TIMEOUT:
            return f"[Timeout] {error.message}. Consider checking if the target service is responding."
        else:
            return error.to_model_message()

    def _format_param_error(
        self,
        error: StructuredError,
        tool_schema: dict
    ) -> str:
        lines = [f"[Invalid Parameters] {error.message}"]
        if error.details and "param" in error.details:
            param_name = error.details["param"]
            param_schema = tool_schema.get("parameters", {}).get("properties", {}).get(param_name, {})
            lines.append(f"Parameter '{param_name}' schema: {json.dumps(param_schema, indent=2)}")
            lines.append(f"Expected type: {param_schema.get('type', 'any')}")
            if "enum" in param_schema:
                lines.append(f"Allowed values: {param_schema['enum']}")
        return "\n".join(lines)
```

## 工程实践

```python
class ToolFailureHandler:
    def __init__(self, config: FailureHandlerConfig):
        self.backoff = ExponentialBackoff(
            base=config.base_delay,
            max_delay=config.max_delay
        )
        self.circuit_breaker = CircuitBreaker(
            failure_threshold=config.circuit_threshold,
            timeout=config.circuit_timeout
        )
        self.error_context = ErrorContext()
        self.adaptive_retry = AdaptiveRetryStrategy()
        self.error_formatter = ModelFriendlyErrorFormatter()

    async def handle(
        self,
        tool_call: ToolCall,
        error: Exception,
        tool_schema: dict
    ) -> ToolResult:
        structured_error = self.classify_error(error)
        self.error_context.add_error(structured_error, tool_call.name)

        if self.error_context.should_abort():
            return ToolResult(
                status="failure",
                error=structured_error,
                message="Maximum failure threshold reached. Aborting."
            )

        if self.adaptive_retry.should_retry(
            tool_call.name,
            self.error_context.error_count(tool_call.name),
            structured_error
        ):
            delay = self.backoff.get_delay(
                self.error_context.error_count(tool_call.name)
            )
            return ToolResult(
                status="retry_scheduled",
                error=structured_error,
                retry_after=delay,
                message=self.error_formatter.format_for_model(
                    structured_error, tool_schema, self.error_context.error_count(tool_call.name)
                )
            )

        return ToolResult(
            status="failure",
            error=structured_error,
            message=self.error_formatter.format_for_model(
                structured_error, tool_schema, self.error_context.error_count(tool_call.name)
            )
        )
```

## 常见误区

- **对所有错误都重试**：永久性错误（如参数错误）重试无意义，浪费资源
- **重试间隔固定**：无法应对限流等需要等待的场景
- **错误信息过于技术化**：模型无法理解如何修正
- **缺少熔断机制**：持续重试会导致系统雪崩
- **不记录失败历史**：无法分析问题根因和优化策略
