# 如何做工具权限分级？

## 面试问题

如何做工具权限分级？

## 考察重点

- 是否理解工具权限分级的核心目的：防止智能体执行危险操作
- 是否能设计基于风险评估的权限分级体系（read-only、read-write、admin）
- 是否理解 Codex 和 Claude Code 的权限控制实现差异
- 是否能处理权限升级、权限上下文切换等复杂场景

## 核心结论

工具权限分级是智能体安全体系的核心组件。设计良好的权限系统能在提供足够能力的同时防止危险操作。核心要素包括：风险评估（高/中/低风险操作）、分级策略（只读、读写、管理员）、升级机制（临时权限提升）和审计日志。权限分级需要与工具设计紧密配合，不是简单的开/关。

## 风险评估与分级标准

### 一、风险分类

| 风险等级 | 操作类型 | 典型工具 | 默认策略 |
|---------|---------|---------|---------|
| **高风险** | 数据修改、系统配置、凭证访问 | `Write`, `Bash`, `APIRequest` | 需要确认或禁用 |
| **中风险** | 外部数据访问、长时间运行 | `WebSearch`, `LongRunningTask` | 限制范围 |
| **低风险** | 只读操作、信息获取 | `Read`, `Grep`, `List` | 默认允许 |

### 二、分级体系设计

```python
class PermissionLevel(Enum):
    READ_ONLY = "read_only"       # 只读操作，不修改任何数据
    READ_WRITE = "read_write"    # 读写操作，可修改数据
    EXECUTE = "execute"          # 执行权限，可运行命令
    ADMIN = "admin"              # 管理权限，可修改配置

@dataclass
class PermissionRule:
    level: PermissionLevel
    allowed_tools: set[str]
    denied_tools: set[str]
    requires_confirmation: bool
    max_calls_per_hour: int
    rate_limit_burst: int

class PermissionConfig:
    def __init__(self):
        self.rules = {
            PermissionLevel.READ_ONLY: PermissionRule(
                level=PermissionLevel.READ_ONLY,
                allowed_tools={"Read", "Grep", "List", "Glob", "WebSearch"},
                denied_tools=set(),
                requires_confirmation=False,
                max_calls_per_hour=1000,
                rate_limit_burst=50
            ),
            PermissionLevel.READ_WRITE: PermissionRule(
                level=PermissionLevel.READ_WRITE,
                allowed_tools={"Read", "Write", "Grep", "List", "Glob", "WebSearch", "Edit"},
                denied_tools={"Bash", "Shell", "Delete"},
                requires_confirmation=True,
                max_calls_per_hour=100,
                rate_limit_burst=10
            ),
            PermissionLevel.EXECUTE: PermissionRule(
                level=PermissionLevel.EXECUTE,
                allowed_tools={"Read", "Write", "Grep", "Bash", "Shell", "Edit", "Execute"},
                denied_tools=set(),
                requires_confirmation=True,
                max_calls_per_hour=50,
                rate_limit_burst=5
            ),
            PermissionLevel.ADMIN: PermissionRule(
                level=PermissionLevel.ADMIN,
                allowed_tools={"*"},  # 所有工具
                denied_tools=set(),
                requires_confirmation=False,
                max_calls_per_hour=10000,
                rate_limit_burst=100
            )
        }
```

## Codex 的权限控制实现

### 安全工具分类

Codex 对工具进行了严格的安全分类：

```json
{
  "name": "bash",
  "description": "Runs shell commands...",
  "restricted": true,
  "dangerous": true,
  "requires_approval": true,
  "approval_prompt": "确认要执行这个命令吗？"
}
```

关键字段：
- `restricted`: 是否受限（受限工具需要额外检查）
- `dangerous`: 是否危险（危险工具永远需要确认）
- `requires_approval`: 是否需要审批

### 权限检查流程

```python
async def check_permission(tool_call: ToolCall, context: ExecutionContext) -> PermissionResult:
    tool_def = tool_registry.get(tool_call.name)
    user_level = context.user_permission_level

    # 1. 检查工具是否在允许列表中
    rule = permission_config.rules[user_level]
    if tool_call.name in rule.denied_tools:
        return PermissionResult(allowed=False, reason=f"Tool {tool_call.name} is denied for level {user_level}")

    if tool_call.name not in rule.allowed_tools and "*" not in rule.allowed_tools:
        return PermissionResult(allowed=False, reason=f"Tool {tool_call.name} not in allowed list")

    # 2. 检查频率限制
    if await rate_limiter.is_exceeded(user_level, tool_call.name):
        return PermissionResult(allowed=False, reason="Rate limit exceeded")

    # 3. 检查是否需要确认
    if tool_def.requires_approval and not context.approved_tools.get(tool_call.name):
        return PermissionResult(allowed=False, reason="Requires approval", needs_approval=True)

    return PermissionResult(allowed=True)
```

### 权限升级机制

```typescript
async function requestPermissionUpgrade(
  tool: string,
  reason: string,
  context: ExecutionContext
): Promise<PermissionResult> {
  // 检查是否已经请求过
  if (context.cachedApprovals.has(tool)) {
    return context.cachedApprovals.get(tool);
  }

  // 发送确认请求给用户
  const userResponse = await sendConfirmationRequest({
    tool,
    reason,
    context: summarizeContext(context),
    options: ["Allow once", "Allow always", "Deny"]
  });

  const result = {
    allowed: userResponse.choice !== "Deny",
    cached: userResponse.choice === "Allow always"
  };

  if (result.cached) {
    context.cachedApprovals.set(tool, result);
  }

  return result;
}
```

## 动态权限调整

### 基于上下文的权限调整

```python
class DynamicPermissionManager:
    def __init__(self):
        self.base_config: PermissionConfig
        self.context_modifiers: list[ContextModifier] = []

    def get_effective_permission(
        self,
        user_level: PermissionLevel,
        task: Task,
        context: ExecutionContext
    ) -> EffectivePermission:
        base_rule = self.base_config.rules[user_level]

        effective = EffectivePermission(
            allowed_tools=set(base_rule.allowed_tools),
            denied_tools=set(base_rule.denied_tools),
            requires_confirmation=base_rule.requires_confirmation
        )

        for modifier in self.context_modifiers:
            modifier.apply(effective, user_level, task, context)

        return effective

class SensitiveDataModifier(ContextModifier):
    def apply(self, effective: EffectivePermission, level, task, context):
        if context.contains_sensitive_data():
            # 发现敏感数据时降级权限
            effective.allowed_tools.discard("Write")
            effective.allowed_tools.discard("Bash")
            effective.requires_confirmation = True

class ProductionEnvironmentModifier(ContextModifier):
    def apply(self, effective: EffectivePermission, level, task, context):
        if context.environment == "production":
            # 生产环境禁止高风险操作
            effective.denied_tools.update({"Delete", "DropTable", "Truncate"})
            effective.requires_confirmation = True
```

### 临时权限提升

```python
class TemporaryPermissionEscalation:
    def __init__(self, max_duration_seconds: int = 300):
        self.max_duration = max_duration_seconds
        self.active_escalations: dict[str, Escalation] = {}

    async def request_escalation(
        self,
        user_id: str,
        tool: str,
        reason: str,
        duration_seconds: int
    ) -> EscalationResult:
        if duration_seconds > self.max_duration:
            return EscalationResult(allowed=False, reason=f"Duration exceeds max {self.max_duration}s")

        escalation = Escalation(
            user_id=user_id,
            tool=tool,
            reason=reason,
            expires_at=datetime.now() + timedelta(seconds=duration_seconds),
            approved=True  # 实际应该发送审批请求
        )

        self.active_escalations[f"{user_id}:{tool}"] = escalation
        return EscalationResult(allowed=True, expires_at=escalation.expires_at)

    def check_escalation(self, user_id: str, tool: str) -> bool:
        key = f"{user_id}:{tool}"
        if key not in self.active_escalations:
            return False

        escalation = self.active_escalations[key]
        if datetime.now() > escalation.expires_at:
            del self.active_escalations[key]
            return False

        return True
```

## 权限审计与合规

### 审计日志设计

```python
@dataclass
class PermissionAuditLog:
    timestamp: datetime
    user_id: str
    tool_name: str
    permission_level: PermissionLevel
    decision: str  # "allowed", "denied", "escalated"
    reason: str
    context_hash: str  # 用于溯源但不存储完整上下文
    tools_in_context: list[str]  # 调用时上下文中可用的工具列表

class PermissionAuditor:
    def __init__(self, storage: AuditStorage):
        self.storage = storage

    async def log_decision(self, log: PermissionAuditLog):
        await self.storage.insert(log)

    async def get_user_permissions_history(
        self,
        user_id: str,
        start_time: datetime,
        end_time: datetime
    ) -> list[PermissionAuditLog]:
        return await self.storage.query(
            user_id=user_id,
            start=start_time,
            end=end_time
        )

    async def detect_anomalies(self, user_id: str) -> list[Anomaly]:
        recent = await self.get_user_permissions_history(
            user_id,
            datetime.now() - timedelta(hours=24)
        )

        anomalies = []

        # 检测权限升级模式
        escalations = [l for l in recent if l.decision == "escalated"]
        if len(escalations) > 10:
            anomalies.append(Anomaly(
                type="excessive_escalation",
                detail=f"User escalated permissions {len(escalations)} times in 24h"
            ))

        # 检测异常工具使用
        tool_usage = Counter(l.tool_name for l in recent)
        if tool_usage.get("Bash", 0) > 100:
            anomalies.append(Anomaly(
                type="excessive_bash_usage",
                detail=f"User executed Bash {tool_usage['Bash']} times in 24h"
            ))

        return anomalies
```

## 工程实践 Checklist

设计工具权限分级时的 Checklist：

- [ ] **风险评估**：是否对每个工具进行了风险等级划分
- [ ] **分级合理**：权限级别设置是否符合最小权限原则
- [ ] **覆盖完整**：权限规则是否覆盖了所有工具
- [ ] **升级机制**：是否有安全的临时权限提升机制
- [ ] **频率限制**：是否对高风险工具有频率限制
- [ ] **审计日志**：是否记录了所有权限相关操作
- [ ] **动态调整**：是否支持基于上下文的动态权限调整
- [ ] **用户确认**：高风险操作是否需要用户确认

## 真实踩坑案例

### 案例：权限验证绕过

某团队实现权限检查时，只在 API 层检查权限，但工具执行层没有二次检查。攻击者通过构造特殊请求绕过了权限检查。

**修复方案**：
1. 在工具执行层增加权限验证
2. 使用权限中间件统一拦截
3. 增加安全测试用例覆盖权限绕过场景

### 案例：权限升级后未回收

某用户申请了临时管理员权限执行特定任务，任务完成后权限未自动回收，导致后续操作仍在高权限下执行。

**修复方案**：
1. 权限升级设置明确的过期时间
2. 任务完成后立即回收权限
3. 增加权限审计告警，监控异常权限使用时间

### 案例：上下文污染导致权限判断错误

某智能体在处理多用户请求时，上下文未正确隔离，导致用户 A 的权限设置影响了用户 B。

**修复方案**：
1. 权限上下文与用户上下文严格绑定
2. 增加上下文隔离测试用例
3. 使用权限 Context 的 immutable copy
