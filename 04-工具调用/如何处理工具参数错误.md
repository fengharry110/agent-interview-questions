# 如何处理工具参数错误？

## 面试问题

如何处理工具参数错误？

## 考察重点

- 是否能设计完整的参数校验体系（类型、范围、格式）
- 是否能实现参数修复和提示机制帮助模型修正
- 是否理解参数校验对工具调用准确性的影响
- 是否能结合Claude Code/Codex实现说明工程方法

## 核心结论

参数错误处理的核心是：**校验**（多层次验证）、**修复**（自动纠错）、**反馈**（清晰提示）。好的参数处理能显著减少模型选错工具后的修正成本。

## 参考回答

### 参数错误的常见类型

| 错误类型 | 示例 | 处理策略 |
|---------|------|---------|
| **类型错误** | expected string, got number | 类型转换或明确提示 |
| **缺失必填** | required param 'path' not provided | 快速失败+提示 |
| **范围超限** | page=1000 exceeds max 100 | 截断或提示 |
| **格式错误** | date="2024/01/01" not ISO format | 自动转换或提示 |
| **枚举错误** | status="pending" not in [active,completed] | 提示可选值 |
| **语义错误** | path="/tmp/../../../etc/passwd" | 安全校验失败 |

### Claude Code的参数校验

Claude Code在MCP协议层实现了严格的参数校验：

```typescript
interface ParameterSchema {
  type: "string" | "number" | "boolean" | "object" | "array";
  description?: string;
  required?: boolean;
  default?: any;
  enum?: any[];
  minimum?: number;
  maximum?: number;
  pattern?: string;
  items?: ParameterSchema;
}

class ParameterValidator {
  validate(params: Record<string, any>, schema: ParameterSchema): ValidationResult {
    const errors: ValidationError[] = [];

    for (const [key, fieldSchema] of Object.entries(schema.properties)) {
      const value = params[key];

      if (fieldSchema.required && value === undefined) {
        errors.push({
          param: key,
          error: "required",
          message: `Parameter '${key}' is required`
        });
        continue;
      }

      if (value !== undefined) {
        const typeResult = this.validateType(value, fieldSchema.type);
        if (!typeResult.valid) {
          errors.push({
            param: key,
            error: "type_mismatch",
            message: typeResult.message
          });
        }

        if (fieldSchema.enum && !fieldSchema.enum.includes(value)) {
          errors.push({
            param: key,
            error: "invalid_enum",
            message: `Parameter '${key}' must be one of: ${fieldSchema.enum.join(", ")}`,
            suggestion: fieldSchema.enum
          });
        }

        if (fieldSchema.minimum !== undefined && value < fieldSchema.minimum) {
          errors.push({
            param: key,
            error: "below_minimum",
            message: `Parameter '${key}' must be >= ${fieldSchema.minimum}`
          });
        }
      }
    }

    return { valid: errors.length === 0, errors };
  }

  validateType(value: any, expectedType: string): { valid: boolean; message?: string } {
    const actualType = Array.isArray(value) ? "array" : typeof value;
    if (actualType !== expectedType) {
      return {
        valid: false,
        message: `Expected ${expectedType}, got ${actualType}`
      };
    }
    return { valid: true };
  }
}
```

### Codex的参数校验

Codex采用JSON Schema进行参数校验：

```python
class SchemaBasedValidator:
    def __init__(self, schema: dict):
        self.schema = schema
        self.jsonschema_validator = Draft7Validator(schema)

    def validate(self, params: dict) -> ValidationResult:
        errors = []
        for error in self.jsonschema_validator.iter_errors(params):
            path = ".".join(str(p) for p in error.path) if error.path else "root"
            errors.append(ParamError(
                param=path,
                error_type=error.validator,
                message=error.message,
                value=error.instance
            ))
        return ValidationResult(
            valid=len(errors) == 0,
            errors=errors
        )

    def coerce_types(self, params: dict) -> dict:
        coerced = params.copy()
        for name, prop in self.schema.get("properties", {}).items():
            if name in coerced and prop.get("type") == "integer":
                if isinstance(coerced[name], float) and coerced[name].is_integer():
                    coerced[name] = int(coerced[name])
        return coerced
```

## 深度展开

### 第一层：多层次校验策略

**校验层次设计**：

```python
class ValidationPipeline:
    def __init__(self):
        self.stages = [
            TypeCheckStage(),
            RangeCheckStage(),
            FormatCheckStage(),
            SecurityCheckStage(),
            SemanticCheckStage()
        ]

    async def validate(
        self,
        params: dict,
        schema: ParameterSchema,
        context: ValidationContext
    ) -> ValidationResult:
        errors = []
        corrected_params = params.copy()

        for stage in self.stages:
            result = await stage.validate(corrected_params, schema, context)
            errors.extend(result.errors)
            corrected_params.update(result.corrections)

        if errors:
            return ValidationResult(valid=False, errors=errors)

        return ValidationResult(valid=True, params=corrected_params)

class SecurityCheckStage(ValidationStage):
    DANGEROUS_PATTERNS = [
        r"\.\./",  # Path traversal
        r"^\/etc",
        r"^\/root",
        r"\|\s*\w+",  # Command injection
        r"&\s*;\s*\w+",  # Command chaining
        r"\$\(",  # Command substitution
    ]

    async def validate(self, params, schema, context):
        errors = []
        for key, value in params.items():
            if isinstance(value, str) and self._contains_dangerous_pattern(value):
                errors.append(ParamError(
                    param=key,
                    error_type="security",
                    message="Parameter contains potentially dangerous content"
                ))
        return ValidationResult(errors=errors)

    def _contains_dangerous_pattern(self, value: str) -> bool:
        return any(re.search(p, value) for p in self.DANGEROUS_PATTERNS)
```

**校验结果分类**：

@dataclass
class ValidationResult:
    valid: bool
    params: dict = None
    errors: List[ParamError] = field(default_factory=list)
    corrections: dict = field(default_factory=dict)
    warnings: List[str] = field(default_factory=list)

@dataclass
class ParamError:
    param: str
    error_type: str
    message: str
    value: Any = None
    suggestion: Any = None

    def to_model_message(self) -> str:
        msg = f"[Parameter Error] {self.message}"
        if self.suggestion:
            msg += f"\n  Suggestion: {self.suggestion}"
        return msg

### 第二层：参数自动修复

**类型自动转换**：

```python
class TypeCoercer:
    COERCION_RULES = {
        ("string", "number"): lambda v: float(v) if v.replace(".", "").isdigit() else v,
        ("string", "boolean"): lambda v: v.lower() in ("true", "1", "yes"),
        ("number", "string"): lambda v: str(v),
        ("string", "array"): lambda v: v.split(",") if isinstance(v, str) else v,
    }

    def coerce(self, value: Any, target_type: str) -> tuple[Any, bool]:
        current_type = self._get_type(value)
        if current_type == target_type:
            return value, False

        rule_key = (current_type, target_type)
        if rule_key in self.COERCION_RULES:
            try:
                return self.COERCION_RULES[rule_key](value), True
            except (ValueError, TypeError):
                return value, False
        return value, False

    def _get_type(self, value: Any) -> str:
        if isinstance(value, bool):
            return "boolean"
        if isinstance(value, (int, float)):
            return "number"
        if isinstance(value, str):
            return "string"
        if isinstance(value, list):
            return "array"
        if isinstance(value, dict):
            return "object"
        return "unknown"
```

**默认值填充**：

```python
class DefaultFiller:
    def fill_defaults(self, params: dict, schema: dict) -> dict:
        filled = params.copy()
        for name, prop in schema.get("properties", {}).items():
            if name not in filled and "default" in prop:
                filled[name] = self._evaluate_default(prop["default"])
        return filled

    def _evaluate_default(self, default: Any) -> Any:
        if callable(default):
            return default()
        if isinstance(default, dict) and "$env" in default:
            return os.environ.get(default["$env"])
        return default
```

**范围自动修正**：

```python
class RangeCorrector:
    def correct(self, params: dict, schema: dict) -> tuple[dict, List[Correction]]:
        corrected = params.copy()
        corrections = []

        for name, prop in schema.get("properties", {}).items():
            if name not in corrected:
                continue

            value = corrected[name]
            if "minimum" in prop and value < prop["minimum"]:
                corrections.append(Correction(
                    param=name,
                    original=value,
                    corrected=prop["minimum"],
                    reason="below_minimum"
                ))
                corrected[name] = prop["minimum"]

            if "maximum" in prop and value > prop["maximum"]:
                if prop.get("exclusiveMaximum"):
                    corrections.append(Correction(
                        param=name,
                        original=value,
                        corrected=prop["maximum"],
                        reason="exceeds_maximum"
                    ))
                    corrected[name] = prop["maximum"]
                else:
                    corrections.append(Correction(
                        param=name,
                        original=value,
                        corrected=prop["maximum"],
                        reason="above_maximum"
                    ))
                    corrected[name] = prop["maximum"]

        return corrected, corrections
```

### 第三层：参数修复提示

**智能纠错建议**：

```python
class ParameterSuggestionEngine:
    def __init__(self):
        self.history = ParameterCorrectionHistory()

    def generate_suggestion(
        self,
        error: ParamError,
        schema: dict,
        previous_errors: List[ParamError]
    ) -> str:
        suggestions = []

        if error.error_type == "invalid_enum":
            suggestions.append(
                f"'{error.param}' should be one of: {error.suggestion}"
            )
            similar = self._find_similar_value(error.value, error.suggestion)
            if similar:
                suggestions.append(f"Did you mean '{similar}'?")

        elif error.error_type == "type_mismatch":
            expected_type = self._get_expected_type(error.param, schema)
            suggestions.append(
                f"'{error.param}' expects type '{expected_type}', "
                f"got '{type(error.value).__name__}'"
            )

        elif error.error_type == "required":
            suggestions.append(
                f"Required parameter '{error.param}' is missing"
            )

        return "\n".join(suggestions)

    def _find_similar_value(
        self,
        invalid_value: str,
        valid_options: List[str]
    ) -> Optional[str]:
        invalid_lower = invalid_value.lower()
        for option in valid_options:
            if self._levenshtein_distance(invalid_lower, option.lower()) <= 2:
                return option
        return None

    def _levenshtein_distance(self, s1: str, s2: str) -> int:
        if len(s1) < len(s2):
            return self._levenshtein_distance(s2, s1)
        if len(s2) == 0:
            return len(s1)

        previous_row = range(len(s2) + 1)
        for i, c1 in enumerate(s1):
            current_row = [i + 1]
            for j, c2 in enumerate(s2):
                insertions = previous_row[j + 1] + 1
                deletions = current_row[j] + 1
                substitutions = previous_row[j] + (c1 != c2)
                current_row.append(min(insertions, deletions, substitutions))
            previous_row = current_row

        return previous_row[-1]
```

**错误上下文累积**：

```python
class ParameterErrorHistory:
    def __init__(self):
        self.errors: List[ParamError] = []
        self.corrections: List[Correction] = []

    def add_error(self, error: ParamError):
        self.errors.append(error)

    def add_correction(self, correction: Correction):
        self.corrections.append(correction)

    def get_context_for_model(self) -> str:
        lines = ["Parameter validation history:"]
        for i, (err, corr) in enumerate(zip(self.errors, self.corrections)):
            lines.append(f"  {i+1}. {err.to_model_message()}")
            if corr:
                lines.append(f"     Corrected to: {corr.corrected}")
        return "\n".join(lines)

    def should_suggest_alternatives(self) -> bool:
        return len(self.errors) >= 2 and any(
            e.error_type == "invalid_enum" for e in self.errors[-3:]
        )
```

### 第四层：参数校验优化

**模型友好的错误格式**：

```python
class ModelFriendlyErrorFormatter:
    def format_errors(
        self,
        result: ValidationResult,
        schema: dict
    ) -> str:
        if result.valid:
            return "[Parameters Valid] All parameters passed validation"

        lines = ["[Parameter Validation Failed]"]
        for error in result.errors:
            lines.append(f"  - {self._format_single_error(error, schema)}")

        if result.warnings:
            lines.append("\nWarnings:")
            for warning in result.warnings:
                lines.append(f"  - {warning}")

        return "\n".join(lines)

    def _format_single_error(
        self,
        error: ParamError,
        schema: dict
    ) -> str:
        base = f"{error.param}: {error.message}"

        if error.error_type == "type_mismatch":
            expected = self._get_expected_type(error.param, schema)
            base += f"\n    Expected type: {expected}"
            base += f"\n    Received type: {type(error.value).__name__}"

        elif error.error_type == "invalid_enum":
            base += f"\n    Allowed values: {error.suggestion}"

        elif error.error_type == "below_minimum" or error.error_type == "above_maximum":
            prop = schema.get("properties", {}).get(error.param, {})
            if "minimum" in prop:
                base += f"\n    Minimum value: {prop['minimum']}"
            if "maximum" in prop:
                base += f"\n    Maximum value: {prop['maximum']}"

        return base
```

**自适应校验严格度**：

```python
class AdaptiveValidator:
    def __init__(self):
        self.strict_mode = True
        self.user_preferences = UserPreferences()
        self.history = ParameterErrorHistory()

    def should_use_strict_mode(self) -> bool:
        if not self.user_preferences.allow_coercion:
            return True

        recent_errors = self.history.errors[-5:]
        if any(e.error_type == "type_mismatch" for e in recent_errors):
            return True

        return self.strict_mode

    def validate(self, params: dict, schema: dict) -> ValidationResult:
        if self.should_use_strict_mode():
            return self.strict_validate(params, schema)
        return self.lenient_validate(params, schema)
```

## 工程实践

```python
class ParameterErrorHandler:
    def __init__(self, schema_registry: SchemaRegistry):
        self.validator = ParameterValidator(schema_registry)
        self.coercer = TypeCoercer()
        self.default_filler = DefaultFiller()
        self.range_corrector = RangeCorrector()
        self.suggestion_engine = ParameterSuggestionEngine()
        self.error_history = ParameterErrorHistory()
        self.error_formatter = ModelFriendlyErrorFormatter()

    async def handle(
        self,
        tool_name: str,
        params: dict,
        context: ExecutionContext
    ) -> ParameterHandleResult:
        schema = self.schema_registry.get_schema(tool_name)

        params = self.default_filler.fill_defaults(params, schema)
        params, type_corrections = self.coercer.coerce_all(params, schema)
        params, range_corrections = self.range_corrector.correct(params, schema)

        validation_result = await self.validator.validate(params, schema, context)

        if validation_result.errors:
            self.error_history.add_error(validation_result.errors)

            if self.error_history.should_suggest_alternatives():
                return ParameterHandleResult(
                    status="needs_alternatives",
                    errors=validation_result.errors,
                    suggestions=self.error_formatter.format_errors(
                        validation_result, schema
                    )
                )

            return ParameterHandleResult(
                status="invalid",
                errors=validation_result.errors,
                message=self.error_formatter.format_errors(validation_result, schema)
            )

        return ParameterHandleResult(
            status="valid",
            params=validation_result.params,
            corrections=[*type_corrections, *range_corrections]
        )
```

## 常见误区

- **只做类型校验**：忽略范围、格式、枚举等约束条件
- **校验失败直接报错**：没有尝试自动修复或给出有用提示
- **错误信息不包含schema**：模型无法知道正确格式
- **对所有参数统一校验**：高风险参数和低风险参数应区分处理
- **不记录历史错误**：无法发现系统性参数问题
