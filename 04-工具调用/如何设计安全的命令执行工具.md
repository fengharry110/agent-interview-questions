# 如何设计安全的命令执行工具？

## 面试问题

如何设计安全的命令执行工具？

## 考察重点

- 是否理解命令执行工具的安全风险：命令注入、环境变量注入、权限提升
- 是否能设计安全的命令执行工具，包含命令白名单、参数验证、限时执行
- 是否理解沙箱化命令执行的实现方式
- 是否能处理命令执行的各种边界情况和降级策略

## 核心结论

命令执行工具（Shell/Bash）是智能体系统中最危险的工具之一。设计不安全的命令执行工具可能导致：
1. 命令注入攻击（`; rm -rf /`）
2. 环境变量注入（`LD_PRELOAD` 攻击）
3. 权限提升（通过 sudo 执行危险操作）
4. 资源耗尽（ fork 炸弹）

安全的命令执行工具设计核心是：**命令白名单**（只允许预定义的命令）、**参数严格校验**（防止注入）、**资源限制**（CPU、内存、时间）和**沙箱隔离**（限制文件系统、网络访问）。

## 安全风险分析

### 主要安全风险

| 风险类型 | 攻击方式 | 危害程度 | 防御措施 |
|---------|---------|---------|---------|
| 命令注入 | `; rm -rf /` | 极高 | 参数转义 + 命令白名单 |
| 管道注入 | `\| malicious_command` | 极高 | 参数转义 + 命令白名单 |
| 环境变量注入 | `LD_PRELOAD=malicious` | 高 | 清理环境变量 |
| 超时攻击 | `sleep 999999` | 中 | 执行超时限制 |
| 资源耗尽 | `:(){ :\|:& };:` (fork bomb) | 高 | 资源配额限制 |
| 权限提升 | `sudo rm -rf /` | 极高 | 禁用 sudo 或限制范围 |

## 安全防护实现

### 命令白名单机制

```python
class SafeCommandRegistry:
    def __init__(self):
        self.allowed_commands = {
            "git": {
                "subcommands": ["status", "log", "diff", "branch", "checkout", "pull", "push", "clone"],
                "allowed_flags": {"--help": True, "-C": True, "--no-pager": True},
                "max_args": 10
            },
            "ls": {
                "allowed_flags": {"-la": True, "-l": True, "-a": True, "-R": True},
                "max_args": 2
            },
            "cat": {
                "allowed_flags": {"-n": True, "-b": True},
                "max_args": 1,
                "blocked_paths": ["/etc", "/root", "/var/log"]
            },
            "grep": {
                "allowed_flags": {"-r": True, "-n": True, "-i": True, "-v": True},
                "max_args": 3,
                "blocked_paths": ["/etc", "/root"]
            },
            "npm": {
                "subcommands": ["install", "run", "test", "start"],
                "allowed_flags": {"--save": True, "--save-dev": True},
                "max_args": 5
            }
        }

    def validate_command(self, cmd: str, args: list[str]) -> ValidationResult:
        # 解析命令
        parts = cmd.split()
        base_cmd = parts[0]

        # 检查命令是否在白名单
        if base_cmd not in self.allowed_commands:
            return ValidationResult(valid=False, reason=f"Command '{base_cmd}' not in whitelist")

        cmd_config = self.allowed_commands[base_cmd]

        # 检查子命令
        if "subcommands" in cmd_config:
            if len(parts) < 2:
                return ValidationResult(valid=False, reason=f"Missing subcommand for '{base_cmd}'")
            subcmd = parts[1]
            if subcmd not in cmd_config["subcommands"]:
                return ValidationResult(valid=False, reason=f"Subcommand '{subcmd}' not allowed")

        # 检查参数数量
        if len(args) > cmd_config.get("max_args", 5):
            return ValidationResult(valid=False, reason=f"Too many arguments")

        return ValidationResult(valid=True)
```

### 参数安全校验

```python
class CommandParamValidator:
    DANGEROUS_CHARS = [";", "|", "&", "$", "`", "<", ">", "\n", "\r", "\0"]
    BLOCKED_PATTERNS = [
        r"rm\s+-rf\s+/",
        r"mkfs\.",
        r"dd\s+if=.*of=/dev/",
        r":\(\)\{.*:\|.*&.*\};:",  # Fork bomb
        r"curl.*\|.*sh",
        r"wget.*\|.*sh"
    ]

    def validate(self, param: str) -> ValidationResult:
        # 检查危险字符
        for char in self.DANGEROUS_CHARS:
            if char in param:
                return ValidationResult(
                    valid=False,
                    reason=f"Dangerous character '{char}' in parameter"
                )

        # 检查危险模式
        import re
        for pattern in self.BLOCKED_PATTERNS:
            if re.search(pattern, param):
                return ValidationResult(
                    valid=False,
                    reason=f"Parameter matches blocked pattern: {pattern}"
                )

        # 检查路径遍历
        if ".." in param:
            return ValidationResult(valid=False, reason="Path traversal not allowed")

        return ValidationResult(valid=True)

    def sanitize(self, param: str) -> str:
        # 对允许的字符进行转义
        return param.replace("'", "'\\''")
```

### 资源限制

```python
class ResourceLimiter:
    def __init__(self):
        self.max_cpu_seconds = 60
        self.max_memory_mb = 512
        self.max_output_bytes = 1024 * 1024  # 1MB
        self.max_temp_files = 100

    async def execute_with_limits(
        self,
        cmd: str,
        args: list[str],
        cwd: str,
        timeout_seconds: int = 30
    ) -> ExecutionResult:
        # 构建进程
        process = await asyncio.create_subprocess_exec(
            cmd,
            *args,
            cwd=cwd,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE,
            env=self.clean_environment()
        )

        try:
            # 带超时执行
            stdout, stderr = await asyncio.wait_for(
                process.communicate(),
                timeout=timeout_seconds
            )

            # 检查输出大小
            if len(stdout) > self.max_output_bytes:
                stdout = stdout[:self.max_output_bytes] + b"... [truncated]"

            return ExecutionResult(
                returncode=process.returncode,
                stdout=stdout.decode("utf-8", errors="replace"),
                stderr=stderr.decode("utf-8", errors="replace")
            )

        except asyncio.TimeoutError:
            process.kill()
            return ExecutionResult(
                returncode=-1,
                stdout="",
                stderr=f"Execution timeout after {timeout_seconds} seconds"
            )

    def clean_environment(self) -> dict:
        # 清理危险环境变量
        env = os.environ.copy()
        dangerous_vars = [
            "LD_PRELOAD", "LD_LIBRARY_PATH", "DYLD_INSERT_LIBRARIES",
            "DYLD_LIBRARY_PATH", "BASH_ENV", "ENV", "HISTFILE",
            "HISTFILESIZE", "HOSTALIASES", "IFS", "MAIL", "TERMCAP"
        ]
        for var in dangerous_vars:
            env.pop(var, None)

        # 设置安全的 PATH
        env["PATH"] = "/usr/bin:/bin:/usr/local/bin"

        return env
```

## Claude Code 的 Bash 工具实现

### 命令执行安全检查

Claude Code 的 `Bash` 工具实现了多层安全检查：

```typescript
// bash.ts
async function safeBash(
  command: string,
  context: ExecutionContext
): Promise<ToolResult> {
  // 1. 命令基础检查
  const parsed = parseCommand(command);

  // 2. 白名单检查
  if (!COMMAND_WHITELIST.has(parsed.cmd)) {
    return {
      error: `Command '${parsed.cmd}' not allowed`,
      tool: "Bash"
    };
  }

  // 3. 参数检查
  for (const arg of parsed.args) {
    if (containsDangerousChars(arg)) {
      return {
        error: "Argument contains dangerous characters",
        tool: "Bash"
      };
    }
  }

  // 4. 执行超时
  const result = await execWithTimeout(command, {
    timeout: 30000,
    cwd: context.workingDirectory,
    env: sanitizeEnv()
  });

  return {
    result: result.stdout,
    error: result.stderr || null,
    metadata: {
      exitCode: result.code,
      duration: result.duration
    }
  };
}

const COMMAND_WHITELIST = new Set([
  "git", "npm", "node", "python", "python3", "ls", "cat", "grep",
  "find", "awk", "sed", "sort", "uniq", "head", "tail", "wc",
  "mkdir", "rmdir", "touch", "cp", "mv", "rm", "chmod", "chown"
]);
```

### 环境清理

```typescript
function sanitizeEnv(): NodeJS.ProcessEnv {
  const env = { ...process.env };

  // 移除危险环境变量
  delete env.LD_PRELOAD;
  delete env.LD_LIBRARY_PATH;
  delete env.DYLD_INSERT_LIBRARIES;
  delete env.DYLD_LIBRARY_PATH;
  delete env.BASH_ENV;
  delete env.ENV;
  delete env.HISTFILE;
  delete env.HISTFILESIZE;

  // 限制 PATH
  env.PATH = "/usr/bin:/bin:/usr/local/bin";

  return env;
}
```

### 使用 seccomp 过滤系统调用

```python
class SeccompFilter:
    ALLOWED_SYSCALLS = [
        # 文件操作
        "read", "write", "open", "close", "stat", "fstat",
        # 进程
        "exit", "exit_group", "getpid", "getppid",
        # 内存
        "brk", "mmap", "munmap",
        # 时间
        "gettimeofday", "clock_gettime",
        # 网络（只允许本地）
        "socket", "bind", "listen", "accept",
    ]

    DENIED_SYSCALLS = [
        "ptrace",    # 进程调试/注入
        "syslog",    # 系统日志
        "mount",     # 文件系统挂载
        "umount2",   # 文件系统卸载
        "chroot",    # 改变根目录
        " pivot_root",  # 切换根文件系统
        "reboot",    # 系统重启
    ]

    def apply(self):
        import ctypes
        import ctypes.util

        libseccomp = ctypes.CDLL(ctypes.util.find_library("seccomp"))

        # 创建过滤器
        ctx = libseccomp.scmp_ctx_init(1)  # 1 = SCMP_ACT_KILL

        # 默认拒绝
        libseccomp.scmp_act_add(ctx, 0, 0x7fff00000000)  # SCMP_ACT_KILL

        # 允许特定调用
        for syscall in self.ALLOWED_SYSCALLS:
            try:
                num = self.get_syscall_num(syscall)
                libseccomp.scmp_act_add(ctx, num, 0)
            except:
                pass

        # 应用过滤器
        libseccomp.scmp_ctx_load(ctx)
```

## 工程实践 Checklist

设计安全命令执行工具时的 Checklist：

- [ ] **命令白名单**：是否只允许预定义的命令
- [ ] **参数校验**：是否检查了危险字符和模式
- [ ] **环境清理**：是否移除了 LD_PRELOAD 等危险环境变量
- [ ] **超时限制**：是否设置了执行超时
- [ ] **资源限制**：是否限制了 CPU、内存、输出大小
- [ ] **输出截断**：是否限制了大输出
- [ ] **沙箱隔离**：是否使用了容器或 gVisor 隔离
- [ ] **审计日志**：是否记录了所有命令执行

## 真实踩坑案例

### 案例：命令注入导致数据删除

某团队的 Bash 工具没有过滤分号，导致用户输入 `file.txt; rm -rf /workspace` 时，第二个命令被执行。

**修复方案**：
1. 实现命令白名单机制
2. 对参数进行严格校验，过滤危险字符
3. 使用 subprocess 而非 shell=True 执行

### 案例：环境变量注入

某团队未清理 LD_PRELOAD，导致恶意库被加载并执行任意代码。

**修复方案**：
1. 清理所有危险环境变量
2. 在隔离环境中执行命令
3. 使用 gVisor 或 seccomp 加固
