# 如何减少模型选错工具？

## 面试问题

如何减少模型选错工具？

## 考察重点

- 是否理解模型选错工具的根本原因
- 是否能设计有效的工具描述和选择策略
- 是否能通过上下文管理和示例引导模型
- 是否能结合Claude Code/Codex实现说明工程方法

## 核心结论

减少选错工具的核心是：**准确的工具描述**、**合理的工具数量**、**清晰的上下文**、**针对性的示例**。本质是降低模型的歧义性和理解成本。

## 参考回答

### 模型选错工具的原因分析

| 原因类别 | 具体表现 | 解决方案 |
|---------|---------|---------|
| **描述模糊** | 工具功能重叠、边界不清 | 优化描述、增加区分度 |
| **选择过多** | 数十个工具让模型困惑 | 分组、动态暴露、层级化 |
| **上下文缺失** | 模型不理解任务场景 | Few-shot示例、任务前缀 |
| **参数误导** | 参数名/类型让模型误判 | 参数命名规范、类型明确 |
| **顺序偏差** | 模型倾向选靠前的工具 | 打乱顺序、强调差异 |

### Claude Code的工具选择机制

Claude Code使用函数调用（Function Calling）进行工具选择：

```typescript
interface ToolDefinition {
  name: string;
  description: string;
  input_schema: object;
}

class ToolSelector {
  constructor(
    private tools: ToolDefinition[],
    private examples: ToolExample[]
  ) {}

  async selectTool(
    task: string,
    context: ConversationContext
  ): Promise<ToolSelection> {
    const prompt = this.buildSelectionPrompt(task, context);

    const response = await this.model.call([
      { role: "system", content: SYSTEM_PROMPT },
      { role: "user", content: prompt },
      ...this.formatExamples(),
    ]);

    return this.parseToolSelection(response);
  }

  private buildSelectionPrompt(
    task: string,
    context: ConversationContext
  ): string {
    return `
Given the following task: "${task}"

Available tools:
${this.formatTools()}

Select the most appropriate tool. Consider:
1. Does the tool's description match the task intent?
2. Are the required parameters available?
3. Is this action allowed given the current context?
`;
  }
}
```

### Codex的工具选择策略

Codex采用语义相似度匹配：

```python
class SemanticToolSelector:
    def __init__(
        self,
        tool_registry: ToolRegistry,
        embedding_model: EmbeddingModel
    ):
        self.tool_registry = tool_registry
        self.embedding_model = embedding_model

    async def select(
        self,
        task: str,
        context: AgentContext
    ) -> Optional[ToolSelection]:
        task_embedding = await self.embedding_model.embed(task)

        candidate_tools = await self._filter_by_context(
            self.tool_registry.get_all_tools(),
            context
        )

        scores = []
        for tool in candidate_tools:
            tool_embedding = tool.embedding
            similarity = cosine_similarity(task_embedding, tool_embedding)
            scores.append((tool, similarity))

        scores.sort(key=lambda x: x[1], reverse=True)

        if scores and scores[0][1] > self.threshold:
            return ToolSelection(tool=scores[0][0], confidence=scores[0][1])

        return None

    async def _filter_by_context(
        self,
        tools: List[Tool],
        context: AgentContext
    ) -> List[Tool]:
        filtered = []
        for tool in tools:
            if self._is_available(tool, context):
                filtered.append(tool)
        return filtered
```

## 深度展开

### 第一层：工具描述优化

**描述设计原则**：

1. **功能明确性**：清晰说明工具能做什么、不能做什么
2. **边界区分度**：与其他工具的差异要明确
3. **参数说明**：关键参数的作用要解释
4. **使用场景**：在什么情况下应该调用此工具

**描述模板**：

```markdown
## tool_name

**Purpose**: [一句话说明工具的核心功能]

**When to use**:
- [场景1]
- [场景2]

**Parameters**:
- `param_name` (type, required): [参数作用说明]

**Returns**: [返回值含义]

**Constraints**:
- [使用限制1]
- [使用限制2]

**vs [similar_tool]**: [关键区别]
```

### 第二层：工具数量控制

**分组策略**：

```python
class ToolGroup:
    def __init__(self, name: str, description: str, tools: List[Tool]):
        self.name = name
        self.description = description
        self.tools = tools

    def matches(self, task: str) -> bool:
        return any(tool.matches(task) for tool in self.tools)

class HierarchicalToolRegistry:
    def __init__(self):
        self.groups: Dict[str, ToolGroup] = {}
        self.flat_tools: Dict[str, Tool] = {}

    def register_group(self, group: ToolGroup):
        self.groups[group.name] = group

    async def get_relevant_tools(
        self,
        task: str,
        context: AgentContext
    ) -> List[Tool]:
        relevant_groups = [
            g for g in self.groups.values()
            if g.matches(task)
        ]

        if not relevant_groups:
            return list(self.flat_tools.values())[:10]

        tools = []
        for group in relevant_groups:
            tools.extend(group.tools)

        return tools[:15]

    def expose_group(
        self,
        group_name: str,
        context: AgentContext
    ) -> List[Tool]:
        if self._should_expose_group(group_name, context):
            return self.groups[group_name].tools
        return []
```

**动态暴露策略**：

```python
class DynamicToolExposer:
    def __init__(self, registry: HierarchicalToolRegistry):
        self.registry = registry
        self.exposure_history: Dict[str, List[str]] = defaultdict(list)

    async def get_tools_for_task(
        self,
        task: str,
        context: AgentContext
    ) -> List[Tool]:
        base_tools = await self.registry.get_relevant_tools(task, context)

        if context.task_type == "coding":
            return self._prioritize_coding_tools(base_tools, context)
        elif context.task_type == "research":
            return self._prioritize_research_tools(base_tools, context)

        return base_tools

    def _prioritize_coding_tools(
        self,
        tools: List[Tool],
        context: AgentContext
    ) -> List[Tool]:
        coding_priority = ["Read", "Edit", "Write", "Bash", "Grep"]
        priority_map = {name: i for i, name in enumerate(coding_priority)}

        return sorted(
            tools,
            key=lambda t: priority_map.get(t.name, len(priority_map))
        )
```

### 第三层：上下文增强

**任务前缀注入**：

```python
class TaskContextEnhancer:
    TASK_PREFIXES = {
        "debug": "The user is debugging an issue. Focus on diagnostic tools.",
        "refactor": "The user is refactoring code. Focus on read and edit tools.",
        "explore": "The user is exploring the codebase. Focus on search and read tools.",
        "test": "The user is writing or running tests. Focus on test-related tools."
    }

    def enhance(self, task: str, context: AgentContext) -> str:
        task_lower = task.lower()

        for keyword, prefix in self.TASK_PREFIXES.items():
            if keyword in task_lower:
                return f"{prefix}\n\nTask: {task}"

        return task

    def get_tools_for_context(
        self,
        base_tools: List[Tool],
        task: str
    ) -> List[Tool]:
        if "debug" in task.lower():
            return [t for t in base_tools if t.category in ["read", "execute"]]
        elif "write" in task.lower() or "create" in task.lower():
            return [t for t in base_tools if t.category in ["write", "edit"]]
        return base_tools
```

### 第四层：选择质量优化

**置信度校准**：

```python
class ConfidenceCalibrator:
    def __init__(self):
        self.calibration_data: List[CalibrationPoint] = []

    def record_selection(
        self,
        task: str,
        selected_tool: str,
        tool_confidence: float,
        success: bool
    ):
        self.calibration_data.append(CalibrationPoint(
            task=task,
            selected_tool=selected_tool,
            confidence=tool_confidence,
            success=success
        ))

    def get_calibrated_threshold(self) -> float:
        if len(self.calibration_data) < 100:
            return 0.7

        correct = [c for c in self.calibration_data if c.success]
        if not correct:
            return 0.9

        avg_confidence_correct = sum(c.confidence for c in correct) / len(correct)
        return max(0.6, avg_confidence_correct - 0.1)
```

**歧义检测与提示**：

```python
class AmbiguityDetector:
    def detect_ambiguity(
        self,
        task: str,
        top_tools: List[Tuple[Tool, float]]
    ) -> Optional[AmbiguityWarning]:
        if len(top_tools) < 2:
            return None

        first_score, second_score = top_tools[0][1], top_tools[1][1]
        score_ratio = first_score / second_score if second_score > 0 else float('inf')

        if score_ratio < 1.5:
            return AmbiguityWarning(
                message=f"Multiple tools have similar scores: "
                        f"{top_tools[0][0].name} ({first_score:.2f}) vs "
                        f"{top_tools[1][0].name} ({second_score:.2f})",
                suggested_tools=[t[0].name for t in top_tools[:3]]
            )

        return None

    def format_warning(self, warning: AmbiguityWarning) -> str:
        return f"""[Ambiguity Detected]
{warning.message}

Consider:
- {', '.join(warning.suggested_tools)}

Choose based on:
1. Specific parameters needed
2. Expected return format
3. Task constraints"""
```

**选择结果验证**：

```python
class SelectionValidator:
    def validate(
        self,
        selection: ToolSelection,
        task: str,
        context: AgentContext
    ) -> ValidationResult:
        errors = []

        if not self._is_tool_appropriate(selection.tool, task):
            errors.append(f"Tool '{selection.tool.name}' may not be optimal for this task")

        if not self._has_required_params(selection.tool, context):
            errors.append(f"Required parameters may not be available")

        if not self._is_permitted(selection.tool, context):
            errors.append(f"Tool '{selection.tool.name}' is not permitted in current context")

        return ValidationResult(
            valid=len(errors) == 0,
            errors=errors
        )

    def _is_tool_appropriate(self, tool: Tool, task: str) -> bool:
        task_embedding = self.embedding_model.embed(task)
        tool_embedding = tool.embedding
        similarity = cosine_similarity(task_embedding, tool_embedding)
        return similarity > 0.3
```

## 工程实践

```python
class ToolSelectionOptimizer:
    def __init__(self, config: OptimizerConfig):
        self.selector = ToolSelector(config.tools, config.examples)
        self.enhancer = TaskContextEnhancer()
        self.calibrator = ConfidenceCalibrator()
        self.ambiguity_detector = AmbiguityDetector()
        self.validator = SelectionValidator()

    async def select(
        self,
        task: str,
        context: AgentContext
    ) -> SelectionResult:
        enhanced_task = self.enhancer.enhance(task, context)

        tools = await self.selector.get_candidates(enhanced_task, context)
        top_tools = tools[:5]

        ambiguity = self.ambiguity_detector.detect_ambiguity(
            enhanced_task,
            [(t, t.confidence) for t in top_tools]
        )

        if ambiguity:
            return SelectionResult(
                status="ambiguous",
                warning=self.ambiguity_detector.format_warning(ambiguity),
                top_tools=top_tools
            )

        best_tool = top_tools[0]
        validation = self.validator.validate(best_tool, enhanced_task, context)

        if not validation.valid:
            return SelectionResult(
                status="needs_review",
                tool=best_tool,
                warnings=validation.errors
            )

        return SelectionResult(status="confirmed", tool=best_tool)
```

## 常见误区

- **工具描述过于简洁**：模型无法理解工具间的差异
- **暴露过多工具**：增加选择难度和误选概率
- **缺少示例**：模型无法理解期望的选择逻辑
- **不验证选择结果**：错误选择无法被纠正
- **忽视上下文**：同一任务在不同场景下应选不同工具
