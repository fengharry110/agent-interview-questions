# 如何设计外部接口调用工具？

## 面试问题

如何设计外部接口调用工具？

## 考察重点

- 是否理解外部API调用的安全机制（认证、超时、限流）
- 是否能设计可靠的错误处理和重试策略
- 是否关注接口调用的可观测性和成本控制
- 是否能结合Claude Code/Codex实现说明工程方法

## 核心结论

外部接口调用工具是智能体连接第三方服务的桥梁，设计核心在于：输入校验确保参数安全、超时控制防止资源耗尽、重试策略保证可靠性、错误映射提供可操作的反馈。

## 参考回答

外部API调用工具的设计关键点：

1. **接口定义标准化**：使用JSON Schema定义请求/响应结构，确保模型理解参数含义
2. **安全机制**：认证Token加密存储、参数白名单校验、输出内容安全扫描
3. **可靠性设计**：超时控制（连接超时/读取超时）、重试策略（指数退避+熔断）、限流保护
4. **可观测性**：请求日志（脱敏后）、响应延迟、错误分类统计

### Claude Code的API调用实现

Claude Code使用`mcp-client`库与外部服务通信，核心机制：

```typescript
interface MCPClientConfig {
  timeout_ms: number;
  retries: number;
  retry_delay_ms: number;
}

class MCPClient {
  async callTool(tool: ToolCall): Promise<ToolResult> {
    const start = Date.now();
    try {
      const response = await this.executeWithTimeout(
        this.buildRequest(tool),
        this.config.timeout_ms
      );
      return this.normalizeResponse(response);
    } catch (error) {
      return this.handleError(error, tool);
    }
  }

  private async executeWithTimeout(
    request: Request,
    timeout: number
  ): Promise<Response> {
    const controller = new AbortController();
    const timeoutId = setTimeout(() => controller.abort(), timeout);
    try {
      return await fetch(request.url, {
        method: request.method,
        headers: request.headers,
        body: request.body,
        signal: controller.signal
      });
    } finally {
      clearTimeout(timeoutId);
    }
  }
}
```

### Codex的API调用实现

Codex采用事件驱动的API调用模式：

```python
class APIInterface:
    def __init__(self, config: APIConfig):
        self.timeout = config.timeout
        self.max_retries = config.max_retries
        self.session = aiohttp.ClientSession()

    async def call(
        self,
        endpoint: str,
        params: dict,
        context: RequestContext
    ) -> APIResponse:
        for attempt in range(self.max_retries):
            try:
                async with self.session.post(
                    endpoint,
                    json=params,
                    timeout=aiohttp.ClientTimeout(total=self.timeout)
                ) as resp:
                    if resp.status == 429:  # Rate limit
                        await self.handle_rate_limit(attempt)
                        continue
                    return await self.parse_response(resp)
            except asyncio.TimeoutError:
                if attempt == self.max_retries - 1:
                    raise APIError(f"Timeout after {self.max_retries} attempts")
                await self.exponential_backoff(attempt)
```

## 深度展开

### 第一层：接口安全设计

外部API调用面临的安全威胁：

| 威胁类型 | 防护机制 | 实现方式 |
|---------|---------|---------|
| API Key泄露 | 加密存储+环境变量 | AWS KMS/Vault管理 |
| 参数注入 | 输入校验+类型强制 | JSON Schema验证 |
| 敏感信息外泄 | 输出过滤+脱敏 | 正则替换+字段白名单 |
| 恶意回调 | HTTPS+证书校验 | SSL Pinning |

**认证机制设计**：

```python
class APICredentialManager:
    def __init__(self, vault_url: str):
        self.vault = VaultClient(vault_url)
        self._cache = LRUCache(maxsize=100)

    async def get_credential(self, service: str) -> Credential:
        if service in self._cache:
            return self._cache.get(service)

        cred = await self.vault.get_secret(f"api/{service}")
        self._cache.put(service, cred)
        return cred

    def rotate_if_needed(self, cred: Credential) -> bool:
        if cred.expires_at - datetime.now() < timedelta(hours=24):
            return True  # Trigger rotation
        return False
```

### 第二层：可靠性设计

**超时配置策略**：

```python
class TimeoutStrategy:
    DEFAULT_CONNECT_TIMEOUT = 5.0   # 连接建立超时
    DEFAULT_READ_TIMEOUT = 30.0     # 读取数据超时
    LONG_RUNNING_TIMEOUT = 300.0    # 长任务超时

    def get_timeout(self, operation: str) -> float:
        if operation in ["search", "query"]:
            return self.DEFAULT_READ_TIMEOUT
        elif operation in ["batch_process", "export"]:
            return self.LONG_RUNNING_TIMEOUT
        return self.DEFAULT_READ_TIMEOUT
```

**指数退避重试**：

```python
class ExponentialBackoff:
    def __init__(self, base_delay: float = 1.0, max_delay: float = 60.0):
        self.base_delay = base_delay
        self.max_delay = max_delay
        self.jitter = RandomJitter()

    async def wait(self, attempt: int) -> None:
        delay = min(self.base_delay * (2 ** attempt), self.max_delay)
        await asyncio.sleep(self.jitter.add(delay))

class RandomJitter:
    def add(self, delay: float) -> float:
        return delay * (0.5 + random.random() * 0.5)  # 50%-150%
```

**熔断器模式**：

```python
class CircuitBreaker:
    def __init__(self, failure_threshold: int = 5, timeout: float = 60.0):
        self.failure_threshold = failure_threshold
        self.timeout = timeout
        self.failure_count = 0
        self.last_failure_time = None
        self.state = "closed"  # closed | open | half_open

    async def call(self, func: Callable) -> Any:
        if self.state == "open":
            if time.time() - self.last_failure_time > self.timeout:
                self.state = "half_open"
            else:
                raise CircuitOpenError("Circuit is open")

        try:
            result = await func()
            self.on_success()
            return result
        except Exception as e:
            self.on_failure()
            raise e

    def on_success(self):
        self.failure_count = 0
        self.state = "closed"

    def on_failure(self):
        self.failure_count += 1
        self.last_failure_time = time.time()
        if self.failure_count >= self.failure_threshold:
            self.state = "open"
```

### 第三层：错误处理与反馈

**错误分类与映射**：

```python
class APIErrorClassifier:
    ERROR_CATEGORIES = {
        "auth": {401, 403, 407},
        "rate_limit": {429},
        "client": {400, 422},
        "server": {500, 502, 503, 504},
        "network": {0},  # Connection errors
        "timeout": {"TIMEOUT"}
    }

    def classify(self, error: Exception, status_code: int = None) -> ErrorCategory:
        if isinstance(error, asyncio.TimeoutError):
            return ErrorCategory.TIMEOUT
        if status_code in self.ERROR_CATEGORIES["network"]:
            return ErrorCategory.NETWORK
        if status_code in self.ERROR_CATEGORIES["auth"]:
            return ErrorCategory.AUTH
        if status_code in self.ERROR_CATEGORIES["rate_limit"]:
            return ErrorCategory.RATE_LIMIT
        return ErrorCategory.UNKNOWN

    def get_recovery_action(self, category: ErrorCategory) -> RecoveryAction:
        actions = {
            ErrorCategory.TIMEOUT: RecoveryAction.RETRY_WITH_BACKOFF,
            ErrorCategory.NETWORK: RecoveryAction.RETRY_IMMEDIATELY,
            ErrorCategory.AUTH: RecoveryAction.REFRESH_TOKEN,
            ErrorCategory.RATE_LIMIT: RecoveryAction.RETRY_AFTER_DELAY,
            ErrorCategory.SERVER: RecoveryAction.RETRY_WITH_BACKOFF,
            ErrorCategory.CLIENT: RecoveryAction.FAIL_FAST,
        }
        return actions.get(category, RecoveryAction.FAIL_FAST)
```

### 第四层：成本与限流控制

**令牌桶限流**：

```python
class TokenBucketRateLimiter:
    def __init__(self, rate: float, capacity: int):
        self.rate = rate  # tokens per second
        self.capacity = capacity
        self.tokens = capacity
        self.last_update = time.time()

    async def acquire(self, tokens: int = 1) -> float:
        while self.tokens < tokens:
            self._refill()
            if self.tokens < tokens:
                wait_time = (tokens - self.tokens) / self.rate
                await asyncio.sleep(wait_time)
        self.tokens -= tokens
        return 0

    def _refill(self):
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.capacity, self.tokens + elapsed * self.rate)
        self.last_update = now
```

**API成本追踪**：

```python
class APICostTracker:
    def __init__(self):
        self.costs = defaultdict(CostSummary)

    def record(self, api_name: str, tokens_used: int, latency_ms: float):
        self.costs[api_name].total_calls += 1
        self.costs[api_name].total_tokens += tokens_used
        self.costs[api_name].total_latency += latency_ms

    def get_report(self) -> dict:
        return {
            api: {
                "calls": summary.total_calls,
                "avg_latency_ms": summary.total_latency / summary.total_calls,
                "cost_estimate": summary.total_tokens * TOKEN_PRICE[api]
            }
            for api, summary in self.costs.items()
        }
```

## 工程实践

```python
class ExternalAPITool:
    def __init__(self, config: ExternalAPIToolConfig):
        self.client = MCPClient(config.mcp)
        self.rate_limiter = TokenBucketRateLimiter(
            rate=config.requests_per_second,
            capacity=config.burst_capacity
        )
        self.circuit_breaker = CircuitBreaker(
            failure_threshold=5,
            timeout=60.0
        )
        self.cost_tracker = APICostTracker()
        self.error_classifier = APIErrorClassifier()

    async def execute(self, tool_call: ToolCall) -> ToolResult:
        start_time = time.time()
        await self.rate_limiter.acquire()

        try:
            result = await self.circuit_breaker.call(
                lambda: self.client.callTool(tool_call)
            )
            self.cost_tracker.record(tool_call.name, result.tokens, result.latency)
            return result
        except Exception as e:
            category = self.error_classifier.classify(e)
            action = self.error_classifier.get_recovery_action(category)
            return self.build_error_response(e, category, action)
```

## 常见误区

- **只做网络超时，不做读取超时**：大文件下载可能无限等待
- **重试不考虑幂等性**：POST类接口重复调用可能产生副作用
- **错误信息直接返回原始内容**：可能泄露内部实现细节
- **忽略第三方服务的SLA**：未做降级预案导致级联失败
- **不跟踪API使用成本**：上线后才发现费用超支
