# 哪些操作需要人工确认？

## 面试问题

哪些操作需要人工确认？

## 考察重点

- 是否理解人工确认不是越多越好，需要在安全和体验之间找平衡
- 是否能根据操作的影响范围、可逆性、风险程度来制定确认策略
- 是否了解不同场景的确认策略（代码助手 vs 客服机器人 vs 自动化工具）
- 是否有设计人工确认流程的经验

## 核心结论

需要人工确认的操作取决于三个因素：**影响范围**（影响多少人）、**可逆性**（能否撤回）、**风险程度**（出问题后果多严重）。基本原则是：**不可逆的操作一定需要确认，影响他人的操作一定需要确认，高风险操作一定需要确认**。但确认不是越多越好——过多的确认会让用户麻木，反而不看内容就点确认。

### 确认决策框架

```python
class ConfirmationDecider:
    """
    判断是否需要人工确认
    """
    
    def needs_confirmation(self, operation: Operation) -> bool:
        """
        根据操作的特征判断是否需要确认
        """
        # 规则 1：不可逆操作
        if operation.is_irreversible():
            return True
        
        # 规则 2：影响他人
        if operation.affects_others():
            return True
        
        # 规则 3：高风险
        if operation.risk_score() > 7:
            return True
        
        # 规则 4：涉及资金/数据
        if operation.involves_finance_or_data():
            return True
        
        # 规则 5：操作超出历史模式
        if self._is_unusual_operation(operation):
            return True
        
        # 规则 6：用户明确要求确认的
        if self._user_prefers_confirmation(operation.type):
            return True
        
        return False
    
    def confirmation_type(self, operation: Operation) -> str:
        """
        返回需要的确认类型
        """
        risk = operation.risk_score()
        
        if risk >= 9:
            return "multi_factor"  # 需要多人/多因素确认
        elif risk >= 7:
            return "explicit"      # 需要明确输入 "yes"
        elif risk >= 5:
            return "one_click"    # 一键确认即可
        else:
            return "no_confirm"   # 不需要确认
```

### 确认体验设计

```python
class ConfirmationUI:
    """
    确认交互设计
    """
    
    @staticmethod
    def format_confirmation(operation: DangerousOperation) -> str:
        """格式化工确认信息"""
        if operation.risk_level == "high":
            return f"""
⚠️ **高风险操作确认**

**你将执行：**
{operation.description}

**后果：**
{operation.consequences}

**影响范围：**
{operation.impact_scope}

请输入 `确认执行` 来继续，或输入 `取消` 来中止。
"""
        elif operation.risk_level == "medium":
            return f"""
ℹ️ **操作确认**

**操作：** {operation.description}

请输入 `y` 确认，或直接回车取消。
"""
    
    @staticmethod
    def needs_attention_prompt(
        operation: DangerousOperation
    ) -> str:
        """需要注意的确认提示"""
        if operation.is_unusual:
            return (
                f"⚠️ 这个操作和你平时的操作模式不同：\n"
                f"- 你很少使用 {operation.tool_name} 工具\n"
                f"- 你很少对 {operation.target} 执行此操作\n"
                f"请确认是你本人想执行这个操作。"
            )
        return ""
```

### 确认频率的控制

```python
class ConfirmationFrequencyControl:
    """
    确认频率控制，避免用户疲劳
    """
    
    def __init__(self):
        self.confirmation_history: Dict[str, List] = {}
    
    def should_confirm(self, user_id: str, operation: Operation) -> bool:
        """
        判断本次是否需要确认
        如果用户近期多次确认同类操作，可以减少确认频率
        """
        history = self.confirmation_history.get(user_id, [])
        
        # 同类操作在过去 1 小时内确认过 3 次以上
        recent_similar = [
            h for h in history
            if h['type'] == operation.type
            and h['time'] > time.time() - 3600
        ]
        
        if len(recent_similar) >= 3:
            # 如果用户之前都确认了，降低频率
            if all(h['approved'] for h in recent_similar[-3:]):
                return False
        
        return True
    
    def record_confirmation(
        self,
        user_id: str,
        operation: Operation,
        approved: bool
    ):
        """记录确认历史"""
        if user_id not in self.confirmation_history:
            self.confirmation_history[user_id] = []
        
        self.confirmation_history[user_id].append({
            'type': operation.type,
            'time': time.time(),
            'approved': approved
        })
```

```python
class UserConfirmationPreferences:
    """
    用户可定制的确认偏好
    """
    
    DEFAULT_LEVEL = "balanced"
    
    LEVELS = {
        "strict": {
            "description": "所有写操作都需要确认",
            "use_case": "新手用户/安全敏感场景"
        },
        "balanced": {
            "description": "不可逆/高风险操作需要确认",
            "use_case": "大多数用户"
        },
        "relaxed": {
            "description": "仅极高风险操作需要确认",
            "use_case": "高级用户/信任环境"
        }
    }
    
    def __init__(self):
        self.preferences: Dict[str, str] = {}
    
    def set_level(self, user_id: str, level: str):
        """设置用户的确认级别"""
        if level in self.LEVELS:
            self.preferences[user_id] = level
    
    def get_level(self, user_id: str) -> str:
        """获取用户的确认级别"""
        return self.preferences.get(user_id, self.DEFAULT_LEVEL)
```

## 工程实践

### 确认系统实现

```python
class ConfirmationSystem:
    """
    人工确认系统
    """
    
    async def confirm(self, operation: Operation) -> bool:
        """
        执行确认流程
        """
        # 1. 判断是否需要确认
        if not self.decider.needs_confirmation(operation):
            return True
        
        # 2. 获取确认类型
        confirm_type = self.decider.confirmation_type(operation)
        
        # 3. 执行确认
        if confirm_type == "multi_factor":
            return await self._multi_factor_confirm(operation)
        elif confirm_type == "explicit":
            return await self._explicit_confirm(operation)
        else:
            return await self._one_click_confirm(operation)
    
    async def _explicit_confirm(self, operation) -> bool:
        """需要用户明确输入确认词"""
        message = self.ui.format_confirmation(operation)
        response = await self.wait_input(message, timeout=120)
        
        if response and response.strip() == "确认执行":
            self.audit.record(operation, confirmed=True)
            return True
        
        self.audit.record(operation, confirmed=False)
        return False
```

## 常见误区

- **什么都要确认**：过多的确认让用户麻木，高危确认也被随手点掉。确认只应该针对真正危险的操作
- **确认信息不清晰**：只写"确认执行此操作"不写后果，用户无法做出知情决定
- **所有用户统一策略**：新手和高级用户的风险承受能力不同，应该提供可定制的确认级别
- **确认了就认为安全了**：用户也可能误确认。确认 + 审计日志才是完整方案

## 追问方向

- 你怎么评估一个操作的风险分数？标准是什么？有没有量化公式？
- 如何处理确认超时？默认拒绝还是默认允许？
- 如果在 5 分钟内用户确认了 10 次"删除文件"，你认为这是正常操作还是攻击？系统应该怎么做？
- 你的确认流程对用户体验的影响有多大？有没有测量过"用户因为频繁确认而放弃使用"的比例？

## 学习建议

在你的智能体系统中找出 10 个典型操作，给每个操作打一个"风险分"（1-10 分）。然后将 7 分以上的操作加入确认流程。运行一周后，统计确认量、用户的确认率和误确认率。根据数据调整确认策略——确认太多就提分阈值，确认太少就降分阈值。
