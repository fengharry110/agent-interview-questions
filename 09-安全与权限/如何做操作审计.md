# 如何做操作审计？

## 面试问题

如何做操作审计？

## 考察重点

- 是否理解审计在智能体系统中的特殊需求：不仅记录"谁做了什么"，还要记录"模型为什么这么做"
- 是否能设计完整的审计日志结构，覆盖输入、决策、执行、结果、异常
- 是否了解日志的防篡改、安全存储和生命周期管理
- 是否有过审计系统在实际生产中的设计经验

## 核心结论

智能体系统的审计和传统系统审计最大的区别是：**不仅要记录"操作了什么"，还要记录"为什么操作"**。因为操作是由模型自主决策的，没有"为什么"的审计日志无法追溯故障根因。完整的审计链是：用户输入 → 模型决策过程 → 工具调用链 → 执行结果 → 最终输出。任何一个环节缺失，审计就失去了价值。

### 审计日志数据结构

```python
from dataclasses import dataclass, field
from datetime import datetime
from typing import Optional, List, Dict, Any
import json
import hashlib

@dataclass
class AuditEvent:
    """
    审计事件——智能体系统中的最小审计单元
    """
    event_id: str                    # 唯一事件 ID
    event_type: str                  # 事件类型
    timestamp: datetime              # 事件时间
    session_id: str                  # 会话 ID
    trace_id: str                    # 追踪 ID（关联同一请求的多个事件）
    user_id: str                     # 用户 ID
    user_role: str                   # 用户角色
    
    # 智能体特有的审计字段
    agent_id: str                    # 智能体实例 ID
    model_name: str                  # 使用的模型
    model_input: Optional[str]       # 模型输入摘要
    model_output: Optional[str]      # 模型输出摘要
    reasoning_chain: Optional[str]   # 模型推理链
    
    # 操作字段
    action: str                      # 操作名称
    params: Dict[str, Any]           # 操作参数
    result_summary: str              # 结果摘要
    status: str                      # 状态: success/failure/blocked
    error: Optional[str]             # 错误信息
    
    # 安全字段
    risk_score: Optional[float]      # 风险评分
    permissions_checked: List[str]   # 已检查的权限
    confirmation_level: str          # 确认等级
    
    # 防篡改
    previous_hash: str               # 前一条日志的哈希
    hash: str = ""                   # 当前日志的哈希
```

### 审计记录器实现

```python
class AuditLogger:
    """
    审计日志记录器——不可变、可追溯
    """
    
    def __init__(self, storage_path: str):
        self.storage_path = storage_path
        self.buffer: List[AuditEvent] = []
        self.last_hash = self._load_last_hash()
    
    async def record(self, event: AuditEvent):
        """
        记录审计事件
        """
        # 1. 设置哈希链
        event.previous_hash = self.last_hash
        event.hash = self._compute_hash(event)
        self.last_hash = event.hash
        
        # 2. 写入存储
        await self._write_event(event)
        
        # 3. 实时检测异常
        await self._check_anomalies(event)
    
    def _compute_hash(self, event: AuditEvent) -> str:
        """计算事件哈希（用于防篡改）"""
        content = json.dumps({
            "event_id": event.event_id,
            "timestamp": event.timestamp.isoformat(),
            "session_id": event.session_id,
            "action": event.action,
            "params": event.params,
            "previous_hash": event.previous_hash
        }, sort_keys=True, default=str)
        return hashlib.sha256(content.encode()).hexdigest()
    
    async def _write_event(self, event: AuditEvent):
        """写入事件到存储"""
        log_line = json.dumps({
            "event_id": event.event_id,
            "event_type": event.event_type,
            "timestamp": event.timestamp.isoformat(),
            "session_id": event.session_id,
            "trace_id": event.trace_id,
            "user_id": event.user_id,
            "user_role": event.user_role,
            "agent_id": event.agent_id,
            "action": event.action,
            "params": self._sanitize_params(event.params),
            "status": event.status,
            "risk_score": event.risk_score,
            "permissions_checked": event.permissions_checked,
            "error": event.error,
            "hash": event.hash,
            "previous_hash": event.previous_hash
        }, default=str)
        
        async with aiofiles.open(
            self._get_log_file(), "a", encoding="utf-8"
        ) as f:
            await f.write(log_line + "\n")
    
    def _sanitize_params(self, params: Dict) -> Dict:
        """脱敏参数中的敏感字段"""
        sensitive_keys = {"password", "token", "api_key", "secret", "authorization"}
        sanitized = {}
        for key, value in params.items():
            if key.lower() in sensitive_keys:
                sanitized[key] = "***REDACTED***"
            else:
                sanitized[key] = value
        return sanitized
    
    def _get_log_file(self) -> str:
        """获取当前日志文件路径（按日期轮转）"""
        today = datetime.now().strftime("%Y-%m-%d")
        return os.path.join(self.storage_path, f"audit_{today}.log")
```

### 审计轨迹回放

```python
class AuditTrailReplay:
    """
    审计轨迹回放——按 trace_id 追踪完整执行链
    """
    
    async def replay_trace(self, trace_id: str) -> Trail:
        """
        回放指定 trace_id 的完整执行轨迹
        """
        events = await self._load_events_by_trace_id(trace_id)
        events.sort(key=lambda e: e.timestamp)
        
        trail = Trail(trace_id=trace_id)
        
        for event in events:
            step = self._event_to_step(event)
            trail.add_step(step)
        
        return trail
    
    def _event_to_step(self, event: AuditEvent) -> TrailStep:
        """将审计事件转换为轨迹步骤"""
        base = {
            "step_id": event.event_id,
            "time": event.timestamp,
            "type": event.event_type,
            "status": event.status
        }
        
        if event.event_type == "user.input":
            return UserInputStep(**base, content=event.params.get("content"))
        elif event.event_type == "model.output":
            return ModelDecisionStep(
                **base,
                reasoning=event.reasoning_chain,
                action=event.action
            )
        elif event.event_type == "tool.invocation":
            return ToolCallStep(
                **base,
                tool=event.action,
                params=event.params
            )
        elif event.event_type == "permission.denied":
            return PermissionDeniedStep(
                **base,
                reason=event.error,
                risk_score=event.risk_score
            )
        
        return GenericStep(**base)
    
    async def _load_events_by_trace_id(
        self, trace_id: str
    ) -> List[AuditEvent]:
        """按 trace_id 加载所有关联事件"""
        events = []
        log_files = glob.glob(
            os.path.join(self.storage_path, "audit_*.log")
        )
        
        for log_file in log_files:
            async with aiofiles.open(log_file, "r") as f:
                async for line in f:
                    event = json.loads(line)
                    if event.get("trace_id") == trace_id:
                        events.append(AuditEvent(**event))
        
        return events
```

### 哈希链防篡改机制

```
日志记录 1               日志记录 2               日志记录 3
┌─────────────────┐     ┌─────────────────┐     ┌─────────────────┐
│ event_id: "001"  │     │ event_id: "002"  │     │ event_id: "003"  │
│ timestamp: T1    │     │ timestamp: T2    │     │ timestamp: T3    │
│ data: {...}      │     │ data: {...}      │     │ data: {...}      │
│ prev_hash: ""    │ ──→ │ prev_hash: H1    │ ──→ │ prev_hash: H2    │
│ hash: H1         │     │ hash: H2         │     │ hash: H3         │
└─────────────────┘     └─────────────────┘     └─────────────────┘
       │                       │                       │
       └───────────────────────┼───────────────────────┘
                               ↓
                    如果中间某条被篡改:
                    HASH(H2) != 记录中的 H2
                    → 整条链断裂，篡改可被发现
```

### 实时异常检测

```python
class AuditAnomalyDetector:
    """
    基于审计日志的实时异常检测
    """
    
    def __init__(self):
        self.rules = [
            AnomalyRule(
                name="rapid_tool_calls",
                condition=lambda events:
                    len([e for e in events if "tool" in e.event_type]) > 50,
                severity="warning",
                description="短时间内大量工具调用"
            ),
            AnomalyRule(
                name="permission_denied_spike",
                condition=lambda events:
                    len([e for e in events if e.event_type == "permission.denied"]) > 10,
                severity="critical",
                description="权限拒绝事件激增，可能正在尝试越权"
            ),
            AnomalyRule(
                name="unauthorized_data_access",
                condition=lambda events:
                    any(e.risk_score and e.risk_score > 0.8 for e in events),
                severity="critical",
                description="高风险的越权数据访问"
            ),
            AnomalyRule(
                name="session_too_long",
                condition=self._session_too_long,
                severity="warning",
                description="会话持续时间异常"
            )
        ]
    
    def analyze_events(self, events: List[AuditEvent]) -> List[Anomaly]:
        """分析事件列表，检测异常"""
        anomalies = []
        for rule in self.rules:
            if rule.condition(events):
                anomalies.append(Anomaly(
                    rule_name=rule.name,
                    severity=rule.severity,
                    description=rule.description,
                    timestamp=datetime.utcnow()
                ))
        return anomalies
    
    def _session_too_long(self, events: List[AuditEvent]) -> bool:
        """检查会话是否过长"""
        if not events:
            return False
        first = min(e.timestamp for e in events)
        last = max(e.timestamp for e in events)
        duration = (last - first).total_seconds()
        return duration > 3600  # 超过 1 小时
```

| 阶段 | 存储 | 保留时间 | 访问权限 | 用途 |

| 温存储 | 压缩日志文件 | 90 天 | 安全团队 | 事件调查、轨迹回放 |

### 审计日志查询接口

```python
class AuditQueryService:
    """
    审计日志查询服务
    """
    
    async def query(
        self,
        filters: AuditFilter
    ) -> QueryResult:
        """
        查询审计日志
        """
        query = self._build_query(filters)
        results = await self.search_client.search(query)
        
        return QueryResult(
            total=results.total,
            events=[AuditEvent(**hit) for hit in results.hits],
            page=filters.page,
            page_size=filters.page_size
        )
    
    def _build_query(self, filters: AuditFilter) -> Dict:
        """构建查询 DSL"""
        must = []
        
        if filters.user_id:
            must.append({"term": {"user_id": filters.user_id}})
        if filters.event_type:
            must.append({"term": {"event_type": filters.event_type}})
        if filters.time_range:
            must.append({
                "range": {
                    "timestamp": {
                        "gte": filters.time_range.start,
                        "lte": filters.time_range.end
                    }
                }
            })
        if filters.status:
            must.append({"term": {"status": filters.status}})
        if filters.risk_min:
            must.append({"range": {"risk_score": {"gte": filters.risk_min}}})
        
        return {
            "query": {"bool": {"must": must}},
            "sort": [{"timestamp": {"order": "desc"}}],
            "from": (filters.page - 1) * filters.page_size,
            "size": filters.page_size
        }


@dataclass
class AuditFilter:
    """审计查询过滤器"""
    user_id: Optional[str] = None
    event_type: Optional[str] = None
    time_range: Optional[TimeRange] = None
    status: Optional[str] = None
    risk_min: Optional[float] = None
    page: int = 1
    page_size: int = 50
```

## 常见误区

- **只记录工具调用，不记录模型决策**：不知道模型为什么调用工具，审计日志只能回答"做了什么"不能回答"为什么做"
- **审计日志自身不安全**：日志明文存储，攻击者入侵后可以篡改或删除日志
- **日志脱敏不足**：审计日志中包含完整的用户输入、API Key、密码等敏感信息
- **日志存储无期限管理**：所有日志永久保留或过早删除，不符合合规要求
- **审计和系统耦合过紧**：审计逻辑混杂在业务代码中，审计系统故障影响主流程
- **忽略审计日志的性能开销**：每个工具调用都写日志，高频场景下审计成为性能瓶颈
- **没有异常检测机制**：日志只存不分析，违规操作发生也无法及时发现

## 追问方向

- 你的审计系统如何在高频工具调用的场景下保证性能？批量写入还是异步写入？
- 审计日志的完整性如何验证？如果怀疑日志被篡改了怎么检查？
- 不同合规要求（SOC2、GDPR、HIPAA）对审计日志有不同的保留要求，你的系统如何适配？
- 如果用户要求"删除我的所有数据"（被遗忘权），审计日志中的用户数据如何处理？
- 如何确保审计系统本身不会成为单点故障？审计系统宕机是否影响主流程运行？

## 学习建议

先从"记录完整决策链"入手：每次模型调用时，记录输入、输出、工具调用序列、推理过程。然后加上"哈希链"防止日志被篡改。最后实现"异常检测"——不是所有审计日志都是事后追溯用的，实时检测同样重要。面试中关键要讲清楚"为什么做"比"做了什么"更重要——这是智能体审计和传统审计的核心区别。
