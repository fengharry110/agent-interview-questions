# 如何把线上失败案例转成测试集？

## 面试问题

如何把线上失败案例转成测试集？

## 考察重点

- 是否理解"线上失败案例→测试用例"这个闭环是智能体系统持续改进的核心机制
- 是否能说清楚从原始轨迹到可用测试用例的转换步骤
- 是否了解自动化和人工在这个过程中的分工
- 是否有实际将线上问题沉淀为回归测试的经验

## 核心结论

把线上失败案例转成测试集的核心步骤是：**收集→筛选→标注→格式化→入库**。关键是把"用户说智能体做错了"这样的模糊反馈，转化为一个可重复执行的测试用例，包含明确的输入、期望输出和评估标准。这个测试集是智能体系统持续改进的燃料，也是防止回退的护栏。

## 参考回答

### 转换流程

```
线上失败案例 → 测试用例的完整流程：

1. 收集
   ├── 用户主动反馈（"回答错了"）
   ├── 自动检测（循环、超时、工具失败）
   └── 采样审查（随机抽取 1% 轨迹做人工审查）

2. 筛选
   ├── 去重（排除已知问题）
   ├── 分类（规划/检索/工具/其他）
   └── 定级（P0 紧急 / P1 重要 / P2 一般）

3. 标注
   ├── 提取输入（用户原始请求 + 上下文）
   ├── 标注期望输出（正确的回答/行为）
   └── 定义评估标准（精确匹配/语义相似/步骤检查）

4. 格式化
   ├── 转为标准测试用例格式
   ├── 加入测试套件
   └── 设置自动运行触发器

5. 验证
   ├── 运行测试，确认当前系统确实失败
   ├── 修复后再次运行，确认修复有效
   └── 定期回归，确保没回退
```

### 收集阶段

```python
class FailureCollector:
    """
    线上失败案例的自动收集
    """
    
    async def collect(self) -> List[FailureCase]:
        cases = []
        
        # 1. 用户反馈
        async for feedback in self.user_feedback_stream():
            if feedback.rating < 3:  # 低评分
                trace = await self.trace_store.get(
                    feedback.session_id
                )
                cases.append(FailureCase(
                    source="user_feedback",
                    trace=trace,
                    user_comment=feedback.comment,
                    priority=self._assess_priority(feedback)
                ))
        
        # 2. 自动检测
        async for trace in self.trace_store.stream_latest():
            if detect_looping(trace):
                cases.append(FailureCase(
                    source="auto_detected_looping",
                    trace=trace,
                    priority="high"
                ))
            if trace.summary.tool_failure_rate > 0.5:
                cases.append(FailureCase(
                    source="high_tool_failure",
                    trace=trace,
                    priority="medium"
                ))
        
        return cases
    
    def _assess_priority(self, feedback) -> str:
        """评估优先级"""
        if "wrong answer" in feedback.comment.lower():
            return "high"
        if "security" in feedback.comment.lower():
            return "critical"
        return "medium"
```

### 标注阶段

标注是将"失败的轨迹"转化为"可用的测试用例"最关键的步骤：

```python
class TestCaseAnnotator:
    """
    将失败轨迹标注为测试用例
    """
    
    def annotate(self, trace: Trace, 
                 expected_behavior: str = None) -> TestCase:
        """
        将一条失败轨迹标注为测试用例
        """
        
        # 1. 提取输入：用户的原始请求
        input_data = {
            "user_goal": trace.user_goal,
            "session_context": self._extract_context(trace),
            "available_tools": trace.tools_snapshot
        }
        
        # 2. 如果没有提供期望行为，自动推断
        if not expected_behavior:
            expected_behavior = self._infer_expected(trace)
        
        # 3. 定义评估标准
        evaluation = self._define_evaluation(trace, expected_behavior)
        
        return TestCase(
            id=f"TC-{trace.session_id}",
            source_trace_id=trace.session_id,
            input=input_data,
            expected_output={
                "behavior": expected_behavior,
                "constraints": self._extract_constraints(trace)
            },
            evaluation=evaluation,
            metadata={
                "failure_type": trace.failure_type,
                "original_model": trace.model,
                "tags": ["regression", trace.failure_type]
            }
        )
    
    def _infer_expected(self, trace: Trace) -> str:
        """
        从失败轨迹推断期望行为
        基于对比正常执行的差异
        """
        if trace.failure_type == "looping":
            return "应该在 10 步内完成，而非陷入循环"
        
        if trace.failure_type == "wrong_tool":
            correct_tool = self._find_correct_tool(trace)
            return f"应该使用 {correct_tool} 而非 {trace.used_tool}"
        
        if trace.failure_type == "incomplete":
            return "应该完成任务的所有步骤"
        
        return "需要人工标注期望行为"
    
    def _define_evaluation(self, trace, expected) -> EvaluationCriteria:
        """
        根据失败类型定义评估方法
        """
        criteria = EvaluationCriteria()
        
        if trace.failure_type == "wrong_answer":
            # 用 LLM 判断语义是否匹配期望
            criteria.add(
                method="llm_judge",
                prompt=f"判断回答是否包含以下关键信息：{expected}"
            )
        elif trace.failure_type == "wrong_tool":
            # 精确检查工具选择
            criteria.add(
                method="exact_match",
                field="steps[0].tool_name",
                expected=expected
            )
        elif trace.failure_type == "looping":
            # 检查步骤数上限
            criteria.add(
                method="max_steps",
                max_count=10
            )
        
        return criteria
```

### 测试集的管理

```python
class RegressionTestSuite:
    """
    回归测试集的管理
    """
    
    def __init__(self):
        self.test_cases: List[TestCase] = []
        self.results: Dict[str, TestResult] = {}
    
    def add_case(self, case: TestCase):
        """添加测试用例"""
        self.test_cases.append(case)
    
    def run_all(self, agent) -> TestRunSummary:
        """运行所有回归测试"""
        results = []
        for case in self.test_cases:
            try:
                # 执行智能体
                output = agent.run(case.input.user_goal)
                
                # 评估
                result = case.evaluation.evaluate(output)
                results.append({
                    "case_id": case.id,
                    "passed": result.passed,
                    "score": result.score,
                    "details": result.details
                })
            except Exception as e:
                results.append({
                    "case_id": case.id,
                    "passed": False,
                    "error": str(e)
                })
        
        return TestRunSummary(
            total=len(results),
            passed=sum(1 for r in results if r["passed"]),
            failed=sum(1 for r in results if not r["passed"]),
            pass_rate=sum(1 for r in results if r["passed"]) / len(results)
        )
    
    def get_failure_trend(self, days: int = 30) -> Trend:
        """获取测试集失败趋势"""
        history = self._get_run_history(days=days)
        return Trend(
            daily_pass_rates=history.daily_rates,
            regressions=history.regressions,
            improvement_areas=history.improvements
        )
```

### 测试用例的生命周期

```python
class TestCaseLifecycle:
    """
    测试用例从创建到退役的完整生命周期
    """
    
    STATES = [
        "draft",        # 自动生成的草稿
        "reviewing",    # 人工审核中
        "active",       # 活跃的回归测试
        "flaky",        # 不稳定（偶尔失败）
        "deprecated",   # 场景已变化
        "archived"      # 已归档
    ]
    
    def promote(self, case_id: str):
        """提升状态"""
        case = self.get_case(case_id)
        
        if case.state == "draft":
            # 人工审核通过后进入 active
            if self.review_approved(case):
                case.state = "active"
        
        elif case.state == "active":
            # 连续 3 次运行结果不一致 → flaky
            if self.is_flaky(case):
                case.state = "flaky"
        
        elif case.state == "flaky":
            # 人工确认为无效用例 → deprecated
            if self.is_no_longer_valid(case):
                case.state = "deprecated"
```

```python
class TraceToTestCase:
    """
    将执行轨迹转换为测试用例的命令行工具
    """
    
    def convert(self, trace_id: str) -> TestCase:
        trace = self.load_trace(trace_id)
        
        # 自动转换
        case = TestCase(
            id=f"TC-{trace_id[:8]}",
            input=TestCaseInput(
                user_goal=trace.user_goal,
                context=self.build_context(trace)
            ),
            expected=self.infer_expected(trace),
            evaluation=self.build_evaluation(trace)
        )
        
        # 输出到标准格式
        self.save_case(case)
        print(f"Created test case: {case.id}")
        print(f"Status: draft (needs review)")
        
        return case
    
    def batch_convert(self, trace_ids: List[str], days: int = 7):
        """批量转换最近 N 天的失败轨迹"""
        cases = []
        for tid in trace_ids:
            try:
                case = self.convert(tid)
                cases.append(case)
            except Exception as e:
                print(f"Failed to convert {tid}: {e}")
        
        print(f"\nConverted {len(cases)} traces to test cases")
        print(f"Review at: {self.review_url}")
```

### 自动化验证

```python
class AutoValidator:
    """
    自动验证测试用例的质量
    """
    
    def validate_case(self, case: TestCase) -> ValidationResult:
        issues = []
        
        # 1. 输入是否完整
        if not case.input.user_goal:
            issues.append("Missing user_goal")
        
        # 2. 期望结果是否为有效内容
        if not case.expected.behavior_summary:
            issues.append("Missing expected behavior")
        
        # 3. 评估标准是否能执行
        if not case.evaluation.is_actionable():
            issues.append("Evaluation criteria not actionable")
        
        # 4. 是否与现有用例重复
        if self.is_duplicate(case):
            issues.append("Duplicate test case")
        
        return ValidationResult(
            valid=len(issues) == 0,
            issues=issues
        )
```

## 常见误区

- **把每条失败轨迹都做成测试用例**：很多失败是偶发的（模型随机性），不适合做稳定的测试用例。需要先确认是否可复现
- **只加不减**：测试集会越来越大，需要定期清理不再适用的用例（模型升级了、功能变了）
- **测试用例和原始轨迹脱钩**：测试用例应该保留 source_trace_id，方便溯源和对比修复前后的行为差异
- **没有自动化运行机制**：测试集建好了但没人跑，等于没有。测试集必须集成到 CI/CD 中，每次改动都自动回归

## 学习建议

从今天开始做一件事：下一次你的智能体出了错，不要只修 prompt，而是先把这次失败转成一个测试用例。花 10 分钟写下输入、期望输出和评估标准。当你积累了 20-30 个这样的测试用例后，你会发现每次改动都有了"安全网"，改 prompt 不再靠感觉了。
