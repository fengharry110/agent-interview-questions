# 什么信息不应该放进上下文？

## 面试问题

什么信息不应该放进上下文？

## 考察重点

- 是否理解上下文空间的稀缺性和成本意识
- 是否能识别哪些信息会降低模型决策质量
- 是否理解上下文污染的来源和防范措施
- 是否能结合Claude Code/Codex实现说明工程方法

## 核心结论

上下文是智能体最宝贵的资源。放错信息不仅浪费空间，更会误导模型。应该排除：**过时信息**、**矛盾信息**、**噪音信息**、**敏感信息**、**模型已知的常识**。

## 参考回答

### 不应放入上下文的信息类型

| 类型 | 示例 | 为什么不放 | 替代方案 |
|-----|------|-----------|---------|
| **过时信息** | 昨天已解决的bug状态 | 占用空间且可能误导 | 只保留最终结论 |
| **矛盾信息** | 两个工具返回相反结论 | 让模型困惑 | 明确标记冲突让模型判断 |
| **原始噪音** | 大文件完整内容 | 超过模型处理能力 | 提取摘要或关键片段 |
| **敏感信息** | 用户密码、API Key | 安全风险 | 日志脱敏后不进入上下文 |
| **常识知识** | Git是什么、HTTP协议 | 浪费token | 不放，模型已知道 |

### Claude Code的上下文过滤

Claude Code实现了严格的上下文过滤机制：

```typescript
interface ContextFilter {
  maxTokenBudget: number;
  contentCategories: ContentCategory[];
}

class ContextFilter {
  constructor(config: ContextFilterConfig) {
    this.maxTokenBudget = config.maxTokenBudget;
    this.contentCategories = config.categories;
  }

  filter(content: Message[]): FilteredContext {
    const scored = content.map(msg => ({
      msg,
      score: this.calculateRelevanceScore(msg)
    }));

    scored.sort((a, b) => b.score - a.score);

    const selected: Message[] = [];
    let totalTokens = 0;

    for (const item of scored) {
      const tokens = this.estimateTokens(item.msg);
      if (totalTokens + tokens <= this.maxTokenBudget) {
        selected.push(item.msg);
        totalTokens += tokens;
      } else {
        break;
      }
    }

    return { messages: selected, totalTokens, dropped: content.length - selected.length };
  }

  calculateRelevanceScore(msg: Message): number {
    let score = 0;

    if (msg.importance === "high") score += 100;
    if (msg.recency === "recent") score += 50;
    if (msg.type === "user_intent") score += 30;
    if (msg.type === "tool_result") score += 20;
    if (msg.type === "reasoning") score += 10;
    if (msg.isError) score -= 50;

    return score;
  }
}
```

### Codex的上下文管理

Codex采用对话压缩算法：

```python
class ConversationCompressor:
    def __init__(self, max_tokens: int):
        self.max_tokens = max_tokens

    def compress(self, messages: List[Message]) -> List[Message]:
        if self._estimate_tokens(messages) <= self.max_tokens:
            return messages

        system_msg = messages[0]
        relevant_messages = self._extract_relevant_core(messages[1:])

        if self._estimate_tokens([system_msg] + relevant_messages) > self.max_tokens:
            relevant_messages = self._summarize_groups(relevant_messages)

        return [system_msg] + relevant_messages

    def _extract_relevant_core(
        self,
        messages: List[Message]
    ) -> List[Message]:
        importance_scores = []
        for i, msg in enumerate(messages):
            score = self._calculate_importance(msg, messages, i)
            importance_scores.append((i, score))

        importance_scores.sort(key=lambda x: x[1], reverse=True)

        selected_indices = []
        total_tokens = 0
        for idx, score in importance_scores:
            msg_tokens = self._estimate_tokens([messages[idx]])
            if total_tokens + msg_tokens <= self.max_tokens * 0.8:
                selected_indices.append(idx)
                total_tokens += msg_tokens

        selected_indices.sort()
        return [messages[i] for i in selected_indices]
```

## 深度展开

### 第一层：敏感信息过滤

**敏感信息识别与过滤**：

```python
class SensitiveInfoFilter:
    SENSITIVE_PATTERNS = {
        "api_key": r'[a-zA-Z0-9]{20,}(?:key|token|secret)',
        "password": r'password\s*[=:]\s*\S+',
        "ssh_key": r'-----BEGIN (?:RSA|DSA|EC|OPENSSH) PRIVATE KEY-----',
        "jwt": r'eyJ[a-zA-Z0-9_-]*\.eyJ[a-zA-Z0-9_-]*\.[a-zA-Z0-9_-]*',
        "aws_key": r'AKIA[0-9A-Z]{16}',
        "credit_card": r'\d{4}[-\s]?\d{4}[-\s]?\d{4}[-\s]?\d{4}',
    }

    def filter(self, text: str) -> tuple[str, List[SensitiveMatch]]:
        matches = []
        masked = text

        for pattern_name, pattern in self.SENSITIVE_PATTERNS.items():
            for match in re.finditer(pattern, text):
                matches.append(SensitiveMatch(
                    type=pattern_name,
                    start=match.start(),
                    end=match.end(),
                    value=match.group()
                ))
                masked = masked.replace(match.group(), f"[{pattern_name.upper()}]")

        return masked, matches

    def should_block_message(self, message: Message) -> bool:
        if message.type == "tool_result":
            for key, value in message.data.items():
                if isinstance(value, str):
                    _, matches = self.filter(value)
                    if any(m.type in ("api_key", "password", "ssh_key") for m in matches):
                        return True
        return False
```

**数据最小化原则**：

```python
class DataMinimizer:
    def minimize(self, data: dict, schema: dict) -> dict:
        minimized = {}
        for key, value in data.items():
            if key in schema.get("sensitive_fields", []):
                minimized[key] = "[REDACTED]"
            elif isinstance(value, dict):
                minimized[key] = self.minimize(value, schema.get("properties", {}).get(key, {}))
            else:
                minimized[key] = value
        return minimized

    def truncate_long_content(self, content: str, max_length: int = 4000) -> str:
        if len(content) <= max_length:
            return content
        return content[:max_length] + f"\n... [Truncated {len(content) - max_length} characters]"
```

### 第二层：噪音信息过滤

**无用信息识别**：

```python
class NoiseFilter:
    NOISE_PATTERNS = [
        r'^Loading\.\.\.$',
        r'^Processing\.\.\.$',
        r'^\.{3,}$',
        r'^(?:Debug|Trace|Verbose) info:.*',
    ]

    def is_noise(self, message: Message) -> bool:
        if message.type not in ("log", "debug", "trace"):
            return False

        for pattern in self.NOISE_PATTERNS:
            if re.match(pattern, message.content):
                return True

        if self._is_repetitive(message):
            return True

        return False

    def _is_repetitive(self, message: Message) -> bool:
        if message.type != "log":
            return False

        recent_similar = sum(
            1 for m in self.recent_messages
            if m.type == "log" and m.content == message.content
        )

        return recent_similar >= 3
```

**日志级别过滤**：

```python
class LogLevelFilter:
    def __init__(self, min_level: str = "INFO"):
        self.levels = {"DEBUG": 0, "INFO": 1, "WARNING": 2, "ERROR": 3}
        self.min_level = self.levels.get(min_level, 1)

    def filter(self, messages: List[Message]) -> List[Message]:
        return [
            msg for msg in messages
            if self._get_level(msg) >= self.min_level
        ]

    def _get_level(self, msg: Message) -> int:
        if hasattr(msg, "level"):
            return self.levels.get(msg.level, 1)
        return 1
```

### 第三层：过时信息处理

**信息时效性管理**：

```python
class TemporalRelevance:
    def __init__(self):
        self.message_timestamps: Dict[str, datetime] = {}
        self.fact_validity: Dict[str, datetime] = {}

    def is_stale(self, fact: str, current_time: datetime) -> bool:
        if fact not in self.fact_validity:
            return False
        return current_time > self.fact_validity[fact]

    def mark_fact_as_stale(self, fact: str):
        self.fact_validity[fact] = datetime.now()

    def update_context(
        self,
        context: List[Message],
        current_time: datetime
    ) -> List[Message]:
        updated = []
        for msg in context:
            if self._is_stale_message(msg, current_time):
                if msg.type in ("user_intent", "plan"):
                    continue
                summary = await self._summarize_stale_message(msg)
                updated.append(summary)
            else:
                updated.append(msg)
        return updated

    def _is_stale_message(self, msg: Message, current_time: datetime) -> bool:
        msg_time = self.message_timestamps.get(msg.id)
        if not msg_time:
            return False
        age = (current_time - msg_time).total_seconds()
        return age > 3600 and msg.type in ("tool_result", "log")
```

### 第四层：信息优先级排序

**Token预算分配**：

```python
class TokenBudgetAllocator:
    def __init__(self, max_tokens: int):
        self.max_tokens = max_tokens
        self.budget分配 = {
            "system_prompt": int(max_tokens * 0.1),
            "user_intent": int(max_tokens * 0.15),
            "recent_history": int(max_tokens * 0.35),
            "retrieved_context": int(max_tokens * 0.25),
            "tool_definitions": int(max_tokens * 0.1),
            "working_memory": int(max_tokens * 0.05),
        }

    def allocate(self, context: ContextRequest) -> TokenAllocation:
        allocated = {}

        if context.user_intent:
            intent_tokens = self._estimate_tokens(context.user_intent)
            allocated["user_intent"] = min(
                intent_tokens,
                self.budget分配["user_intent"]
            )

        if context.retrieved_docs:
            doc_tokens = sum(self._estimate_tokens(doc) for doc in context.retrieved_docs)
            allocated["retrieved_context"] = min(
                doc_tokens,
                self.budget分配["retrieved_context"]
            )

        return TokenAllocation(
            budget=self.budget分配,
            allocated=allocated,
            remaining=self.max_tokens - sum(allocated.values())
        )
```

## 工程实践

```python
class ContextPurifier:
    def __init__(self, config: PurifierConfig):
        self.sensitive_filter = SensitiveInfoFilter()
        self.noise_filter = NoiseFilter()
        self.temporal_relevance = TemporalRelevance()
        self.budget_allocator = TokenBudgetAllocator(config.max_tokens)

    async def purify(
        self,
        messages: List[Message],
        context_request: ContextRequest
    ) -> PurifiedContext:
        filtered = []
        sensitive_matches = []

        for msg in messages:
            if self.sensitive_filter.should_block_message(msg):
                continue
            masked_msg, matches = self.sensitive_filter.filter(msg)
            sensitive_matches.extend(matches)
            if not self.noise_filter.is_noise(masked_msg):
                filtered.append(masked_msg)

        deprioritized = await self.temporal_relevance.update_context(
            filtered, datetime.now()
        )

        allocation = self.budget_allocator.allocate(context_request)
        final_context = self._fit_to_budget(deprioritized, allocation)

        return PurifiedContext(
            messages=final_context,
            dropped_count=len(messages) - len(final_context),
            sensitive_matches=sensitive_matches,
            token_allocation=allocation
        )
```

## 常见误区

- **什么信息都往里塞**：导致上下文膨胀，模型难以聚焦关键信息
- **不做敏感信息过滤**：可能泄露密码、API Key等
- **保留完整日志**：日志噪音严重影响模型理解
- **忽视信息时效性**：用过时的中间结果误导模型
- **不进行Token预算控制**：超出窗口导致截断重要信息
