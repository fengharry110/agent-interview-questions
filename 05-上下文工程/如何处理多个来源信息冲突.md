# 如何处理多个来源信息冲突？

## 面试问题

如何处理多个来源信息冲突？

## 考察重点

- 是否能识别信息冲突的不同类型和原因
- 是否能设计冲突检测和解决机制
- 是否理解何时应该让模型处理而非系统自动解决
- 是否能结合Claude Code/Codex实现说明工程方法

## 核心结论

处理信息冲突的核心是：**检测-评估-解决**。系统应能自动解决明显的冲突（如过时信息），但将复杂判断留给模型，并提供足够的上下文让模型做出明智决定。

## 参考回答

### 冲突类型分析

| 冲突类型 | 示例 | 解决策略 |
|---------|------|---------|
| **时效冲突** | 文档A说最新，文档B说已过时 | 信任最新来源 |
| **权威冲突** | 官方文档vs社区文档 | 官方优先 |
| **语义冲突** | 两个检索结果含义相反 | 需人工判断 |
| **结构性冲突** | 不同格式表示同一实体 | 标准化后比较 |

### Claude Code的冲突处理

```typescript
class InformationConflictResolver {
  async resolve(
    conflictingItems: ContextItem[],
    task: Task
  ): Promise<ResolvedContext> {
    const conflicts = this.detectConflicts(conflictingItems);

    const resolved: ResolvedItem[] = [];
    const uncertainties: UnresolvedConflict[] = [];

    for (const conflict of conflicts) {
      if (this.canAutoResolve(conflict)) {
        const resolved_item = await this.autoResolve(conflict);
        resolved.push(resolved_item);
      } else {
        uncertainties.push({
          conflict,
          context: this.buildConflictContext(conflict)
        });
      }
    }

    return {
      resolved,
      uncertainties,
      needsModelJudgment: uncertainties.length > 0
    };
  }

  private detectConflicts(items: ContextItem[]): Conflict[] {
    const conflicts: Conflict[] = [];

    const claims = this.extractClaims(items);

    const claimGroups = this.groupByMeaning(claims);

    for (const group of claimGroups) {
      if (group.length > 1 && this.hasContradiction(group)) {
        conflicts.push({
          id: uuid(),
          items: group.map(c => c.source),
          claim_type: group[0].type,
          contradiction: this.explainContradiction(group)
        });
      }
    }

    return conflicts;
  }

  private canAutoResolve(conflict: Conflict): boolean {
    if (conflict.claim_type === "temporal_fact") {
      return true;
    }

    if (conflict.claim_type === "numeric_value") {
      return true;
    }

    return false;
  }

  private async autoResolve(conflict: Conflict): Promise<ResolvedItem> {
    if (conflict.claim_type === "temporal_fact") {
      const byTime = conflict.items.sort(
        (a, b) => new Date(b.metadata.timestamp).getTime() -
                  new Date(a.metadata.timestamp).getTime()
      );
      return {
        resolved_to: byTime[0],
        reason: "Most recent source"
      };
    }

    if (conflict.claim_type === "numeric_value") {
      const values = conflict.items.map(i => ({
        item: i,
        value: this.extractNumericValue(i.content)
      })).filter(x => x.value !== null);

      values.sort((a, b) => Math.abs(a.value - b.value) < 0.001 ? 0 : -1);

      return {
        resolved_to: values[0].item,
        reason: "Most confident measurement"
      };
    }
  }
}
```

### Codex的冲突处理

```python
class CodexConflictResolver:
    def resolve(
        self,
        retrieved_docs: List[RetrievedDoc],
        query: str
    ) -> ResolutionResult:
        conflicts = self._find_conflicts(retrieved_docs)

        if not conflicts:
            return ResolutionResult(
                content=self._combine_docs(retrieved_docs),
                resolved=True,
                method="no_conflict"
            )

        resolvable = []
        unresolvable = []

        for conflict in conflicts:
            if self._is_auto_resolvable(conflict):
                resolvable.append(self._resolve_conflict(conflict))
            else:
                unresolvable.append(conflict)

        if unresolvable:
            return ResolutionResult(
                content=self._build_conflict_aware_content(
                    retrieved_docs, resolvable, unresolvable
                ),
                resolved=False,
                conflicts=unresolvable,
                method="partial_auto"
            )

        return ResolutionResult(
            content=self._build_resolved_content(retrieved_docs, resolvable),
            resolved=True,
            method="auto_resolved"
        )

    def _is_auto_resolvable(self, conflict: Conflict) -> bool:
        if conflict.type == "source_authority":
            return True
        if conflict.type == "recency":
            return True
        if conflict.type == "format":
            return True
        return False

    def _resolve_conflict(self, conflict: Conflict) -> ResolvedConflict:
        if conflict.type == "recency":
            newer = max(conflict.items, key=lambda x: x.timestamp)
            return ResolvedConflict(
                original=conflict,
                resolved_to=newer,
                method="recency"
            )
```

### 第一层：冲突检测

**基于语义的冲突检测**：

```python
class SemanticConflictDetector:
    def __init__(self, embedder: Embedder):
        self.embedder = embedder

    async def detect(
        self,
        items: List[ContextItem]
    ) -> List[Conflict]:
        conflicts = []

        claims = self._extract_claims(items)

        claim_embeddings = await self.embedder.batch_embed(
            [c.content for c in claims]
        )

        for i in range(len(claims)):
            for j in range(i + 1, len(claims)):
                similarity = cosine_similarity(
                    claim_embeddings[i],
                    claim_embeddings[j]
                )

                if similarity < 0.7 and self._contradicts(claims[i], claims[j]):
                    conflicts.append(Conflict(
                        items=[claims[i], claims[j]],
                        similarity=similarity,
                        conflict_type="semantic"
                    ))

        return conflicts

    def _extract_claims(self, items: List[ContextItem]) -> List[Claim]:
        claims = []
        for item in items:
            extracted = self._parse_claims(item)
            for claim in extracted:
                claim.source = item
                claims.append(claim)
        return claims

    def _contradicts(self, claim1: Claim, claim2: Claim) -> bool:
        contradiction_pairs = [
            ("is", "is not"),
            ("supports", "opposes"),
            ("greater than", "less than"),
            ("before", "after"),
        ]

        c1_lower = claim1.content.lower()
        c2_lower = claim2.content.lower()

        for pos, neg in contradiction_pairs:
            if pos in c1_lower and pos in c2_lower:
                if (pos in c1_lower and neg in c2_lower) or \
                   (neg in c1_lower and pos in c2_lower):
                    return True

        return False
```

**结构化冲突检测**：

```python
class StructuredConflictDetector:
    def detect_structural_conflicts(
        self,
        items: List[ContextItem]
    ) -> List[Conflict]:
        conflicts = []

        entities = self._extract_entities(items)

        entity_groups = self._group_by_entity(entities)

        for entity, mentions in entity_groups.items():
            if len(mentions) > 1:
                conflicting = self._check_attribute_conflicts(mentions)
                if conflicting:
                    conflicts.append(Conflict(
                        entity=entity,
                        mentions=mentions,
                        type="structural",
                        details=conflicting
                    ))

        return conflicts

    def _check_attribute_conflicts(
        self,
        mentions: List[EntityMention]
    ) -> List[AttributeConflict]:
        conflicts = []
        attributes = defaultdict(list)

        for mention in mentions:
            for attr, value in mention.attributes.items():
                attributes[attr].append((mention, value))

        for attr, values in attributes.items():
            unique_values = set(v for _, v in values)
            if len(unique_values) > 1:
                conflicts.append(AttributeConflict(
                    attribute=attr,
                    values=values,
                    count=len(unique_values)
                ))

        return conflicts
```

### 第二层：冲突解决策略

**基于权威性的解决**：

```python
class AuthorityBasedResolver:
    AUTHORITY_LEVELS = {
        "official_documentation": 100,
        "api_documentation": 90,
        "technical_specification": 85,
        "community_wiki": 50,
        "forum_post": 30,
        "unknown": 10
    }

    def resolve(self, conflict: Conflict) -> ResolvedItem:
        if conflict.type != "authority":
            raise ValueError("Cannot resolve non-authority conflict")

        scored = []
        for item in conflict.items:
            authority = self._get_authority(item.source)
            scored.append((item, authority))

        scored.sort(key=lambda x: x[1], reverse=True)

        return ResolvedItem(
            selected=scored[0][0],
            reason=f"Highest authority: {scored[0][1]}",
            method="authority"
        )

    def _get_authority(self, source: Source) -> int:
        if hasattr(source, "authority_level"):
            return source.authority_level

        source_type = source.type.lower()
        return self.AUTHORITY_LEVELS.get(source_type, 10)
```

**基于时间的解决**：

```python
class TimeBasedResolver:
    def resolve(self, conflict: Conflict) -> ResolvedItem:
        if not all(hasattr(item, "timestamp") for item in conflict.items):
            raise ValueError("Not all items have timestamps")

        sorted_items = sorted(
            conflict.items,
            key=lambda x: x.timestamp or datetime.min,
            reverse=True
        )

        return ResolvedItem(
            selected=sorted_items[0],
            reason=f"Most recent: {sorted_items[0].timestamp}",
            method="recency"
        )
```

**基于置信度的解决**：

```python
class ConfidenceBasedResolver:
    def resolve(self, conflict: Conflict) -> ResolvedItem:
        scored = []
        for item in conflict.items:
            confidence = self._calculate_confidence(item)
            scored.append((item, confidence))

        scored.sort(key=lambda x: x[1], reverse=True)

        return ResolvedItem(
            selected=scored[0][0],
            reason=f"Highest confidence: {scored[0][1]:.2f}",
            method="confidence"
        )

    def _calculate_confidence(self, item: ContextItem) -> float:
        confidence = 0.5

        if hasattr(item, "relevance_score"):
            confidence += item.relevance_score * 0.3

        if hasattr(item, "source_quality"):
            confidence += item.source_quality * 0.3

        if hasattr(item, "retrieval_count"):
            confidence += min(item.retrieval_count / 10, 0.2)

        return min(confidence, 1.0)
```

### 第三层：让模型处理冲突

**冲突上下文构建**：

```python
class ModelConflictHandler:
    def build_prompt_for_model(
        self,
        conflict: Conflict,
        context: str
    ) -> str:
        return f"""You need to resolve a conflict between multiple information sources.

Context:
{context}

Conflict:
The following sources provide contradictory information:

{self._format_conflict_items(conflict)}

Analyze each source and determine which is most likely correct based on:
1. Source authority and reliability
2. Recency of information
3. Specificity to your current task
4. Internal consistency of the information

Provide your resolution and reasoning."""

    def _format_conflict_items(self, conflict: Conflict) -> str:
        lines = []
        for i, item in enumerate(conflict.items, 1):
            lines.append(f"\n--- Source {i} ---")
            lines.append(f"Content: {item.content[:500]}")
            if hasattr(item, "source"):
                lines.append(f"Source: {item.source}")
            if hasattr(item, "timestamp"):
                lines.append(f"Time: {item.timestamp}")
        return "\n".join(lines)

    async def request_model_resolution(
        self,
        conflict: Conflict,
        context: str,
        model: LLM
    ) -> str:
        prompt = self.build_prompt_for_model(conflict, context)
        response = await model.complete([
            {"role": "system", "content": "You are a careful information analyst."},
            {"role": "user", "content": prompt}
        ])
        return response.text
```

```python
@dataclass
class ConflictMetadata:
    conflict_id: str
    conflict_type: str
    source_items: List[str]
    resolution_method: str
    resolved_by: str
    timestamp: datetime
    model_reasoning: str = None

class ConflictTracker:
    def __init__(self):
        self.conflicts: List[ConflictMetadata] = []

    def record_resolution(
        self,
        conflict: Conflict,
        resolution: ResolvedItem,
        method: str,
        resolved_by: str = "system"
    ):
        self.conflicts.append(ConflictMetadata(
            conflict_id=conflict.id,
            conflict_type=conflict.type,
            source_items=[item.id for item in conflict.items],
            resolution_method=method,
            resolved_by=resolved_by,
            timestamp=datetime.now()
        ))

    def get_conflict_report(self) -> dict:
        return {
            "total_conflicts": len(self.conflicts),
            "by_type": self._count_by_type(),
            "by_method": self._count_by_method(),
            "unresolved": self._get_unresolved()
        }
```

```python
class ConflictUIPresenter:
    def present_to_user(self, conflicts: List[Conflict]) -> str:
        if not conflicts:
            return "No conflicts detected."

        lines = ["[Information Conflicts Detected]\n"]

        for i, conflict in enumerate(conflicts, 1):
            lines.append(f"\n--- Conflict {i} ---")
            lines.append(f"Type: {conflict.type}")
            lines.append(f"\nConflicting claims:")

            for j, item in enumerate(conflict.items, 1):
                lines.append(f"  {j}. {item.content[:200]}...")
                if hasattr(item, "source"):
                    lines.append(f"     Source: {item.source}")

            if conflict.resolution:
                lines.append(f"\nResolution: {conflict.resolution}")
            else:
                lines.append("\nResolution: Pending")

        return "\n".join(lines)
```

## 工程实践

```python
class ConflictResolutionPipeline:
    def __init__(self, config: ConflictConfig):
        self.semantic_detector = SemanticConflictDetector(config.embedder)
        self.structured_detector = StructuredConflictDetector()
        self.authority_resolver = AuthorityBasedResolver()
        self.time_resolver = TimeBasedResolver()
        self.confidence_resolver = ConfidenceBasedResolver()
        self.model_handler = ModelConflictHandler()
        self.tracker = ConflictTracker()

    async def resolve(
        self,
        items: List[ContextItem],
        context: str = None
    ) -> ResolutionResult:
        all_conflicts = []

        semantic_conflicts = await self.semantic_detector.detect(items)
        all_conflicts.extend(semantic_conflicts)

        structural_conflicts = self.structured_detector.detect_structural_conflicts(items)
        all_conflicts.extend(structural_conflicts)

        resolved = []
        needs_model = []

        for conflict in all_conflicts:
            if self._is_auto_resolvable(conflict):
                resolution = await self._auto_resolve(conflict)
                resolved.append(resolution)
                self.tracker.record_resolution(
                    conflict, resolution, resolution.method
                )
            else:
                needs_model.append(conflict)

        if needs_model and context:
            for conflict in needs_model:
                resolution_text = await self.model_handler.request_model_resolution(
                    conflict, context, self.model
                )
                resolved.append(ResolvedConflict(
                    original=conflict,
                    selected_content=resolution_text,
                    method="model"
                ))

        return ResolutionResult(
            original_items=items,
            conflicts=all_conflicts,
            resolutions=resolved,
            unresolved=len(needs_model),
            conflict_report=self.tracker.get_conflict_report()
        )
```

## 常见误区

- **强行自动解决所有冲突**：复杂语义冲突需要模型判断
- **不追踪冲突历史**：无法改进冲突解决策略
- **对用户隐藏冲突**：用户可能对结果有疑虑
- **冲突检测过于敏感**：微小差异也被当作冲突
- **不考虑上下文**：同样的冲突在不同场景下可能需要不同解决方式
