# 工具执行结果应该如何进入上下文？

## 面试问题

工具执行结果应该如何进入上下文？

## 考察重点

- 是否理解工具返回值的结构化处理机制
- 能否说清楚信息筛选、压缩和格式化的策略
- 是否能结合具体场景说明实现细节
- 是否理解工具结果对模型决策的影响

## 核心结论

工具执行结果进入上下文的方式直接影响模型对工具使用效果的理解。核心问题是：如何在有限上下文空间内，提供足够信息让模型判断是否需要继续调用工具或结束任务。

class ClaudeCodeToolResultProcessor:
    MAX_RESULT_TOKENS = 8000
    MIN_RELEVANT_TOKENS = 100
    
    def process(self, tool_result: ToolResult, tool_call: ToolCall) -> ProcessedResult:
        # 1. 根据工具类型选择处理策略
        strategy = self.get_strategy(tool_call.tool_name)
        
        # 2. 执行基础处理
        processed = strategy.process(tool_result.data)
        
        # 3. 检查是否需要截断

            processed = self.truncate_with_summary(processed, self.MAX_RESULT_TOKENS)

        return ProcessedResult(
            content=processed,
            tool_name=tool_call.tool_name,
            status=tool_result.status,
            truncated=tokens > self.MAX_RESULT_TOKENS,
            relevance_score=self.calculate_relevance(processed, tool_call.task)
        )
    
    def format_for_context(self, processed: ProcessedResult) -> str:
        return f"""
<tool_result name="{processed.tool_name}" status="{processed.status}">
{processed.content}
</tool_result>
"""

## 信息筛选策略

### 基于相关性的筛选

```python
class RelevanceBasedFilter:
    def __init__(self, relevance_threshold: float = 0.6):
        self.threshold = relevance_threshold
    
    def filter(self, tool_result: Any, task: str) -> FilteredResult:
        # 1. 分析任务的关键实体
        task_entities = self.extract_entities(task)
        
        # 2. 分析结果内容
        result_parts = self.split_result(tool_result)
        
        # 3. 计算每个部分的相关性
        scored_parts = []
        for part in result_parts:
            score = self.calculate_relevance(part, task_entities)
            scored_parts.append((part, score))
        
        # 4. 选择高相关性部分
        selected = [p for p, s in scored_parts if s >= self.threshold]
        
        # 5. 如果选择太少，放宽阈值
        if len(selected) < 3 and self.threshold > 0.3:
            self.threshold *= 0.8
            selected = [p for p, s in scored_parts if s >= self.threshold]
        
        return FilteredResult(
            content='\n'.join(selected),
            included_ratio=len(selected) / len(result_parts),
            scores={i: s for i, (p, s) in enumerate(scored_parts)}
        )
```

```python
class StructuredResultExtractor:
    def extract(self, tool_result: Any, tool_type: str) -> ExtractedContent:
        if tool_type == "filesystem":
            return self.extract_file_info(tool_result)
        elif tool_type == "bash":
            return self.extract_command_output(tool_result)
        elif tool_type == "web_search":
            return self.extract_search_results(tool_result)
        elif tool_type == "database":
            return self.extract_query_results(tool_result)
    
    def extract_file_info(self, result: dict) -> ExtractedContent:
        # 文件搜索结果：只保留路径和关键元信息
        files = result.get('files', [])
        extracted = []
        for f in files:
            extracted.append({
                'path': f.get('path'),
                'size': f.get('size'),
                'modified': f.get('modified')
            })
        return ExtractedContent(
            summary=f"Found {len(files)} files",
            details=extracted,
            full_content=str(result)  # 保留完整内容用于调试
        )
    
    def extract_command_output(self, result: dict) -> ExtractedContent:
        # 命令输出：保留最后N行和错误信息
        stdout = result.get('stdout', '')
        stderr = result.get('stderr', '')
        lines = stdout.split('\n')
        
        if len(lines) > 100:
            head = '\n'.join(lines[:50])
            tail = '\n'.join(lines[-50:])
            content = f"{head}\n... [{len(lines) - 100} lines truncated] ...\n{tail}"
        else:
            content = stdout
        
        if stderr:
            content += f"\n[stderr]: {stderr}"
        
        return ExtractedContent(
            summary=self.summarize_output(stdout),
            details=content,
            has_error=bool(stderr)
        )
```

## Codex的处理模式

```typescript
class CodexToolResultIngester {
    private readonly MAX_TOKENS_PER_RESULT = 6000;
    private readonly MIN_LINES_OF_OUTPUT = 5;
    
    async ingest(toolCall: ToolCall, result: ToolResult): Promise<ContextEntry> {
        const entry: ContextEntry = {
            type: 'tool_result',
            toolName: toolCall.name,
            timestamp: Date.now(),
            content: await this.processResult(result, toolCall),
            metadata: {
                success: result.status === 'success',
                tokens: await this.countTokens(result),
                executionTime: result.metadata?.execution_time_ms
            }
        };
        
        // 记录到执行历史
        this.executionHistory.push({
            toolCall,
            result: entry,
            task: toolCall.task
        });
        
        return entry;
    }
    
    private async processResult(result: ToolResult, toolCall: ToolCall): Promise<string> {
        if (result.status === 'failure') {
            return this.formatError(result.error);
        }
        
        let content = this.extractContent(result.data, toolCall.name);
        
        // 如果内容过长，进行智能压缩
        if (await this.countTokens(content) > this.MAX_TOKENS_PER_RESULT) {
            content = await this.compress(content, toolCall);
        }
        
        // 添加执行摘要
        return `Tool: ${toolCall.name}\nResult: ${content}`;
    }
    
    private async compress(content: string, toolCall: ToolCall): Promise<string> {
        const lines = content.split('\n');
        
        // 策略1：保留开头和结尾（大多数命令输出）
        if (lines.length > 200) {
            const head = lines.slice(0, 100);
            const tail = lines.slice(-100);
            return [
                ...head,
                `... [${lines.length - 200} lines truncated] ...`,
                ...tail
            ].join('\n');
        }
        
        // 策略2：提取关键行
        const keyLines = this.extractKeyLines(lines, toolCall.name);
        return keyLines.join('\n');
    }
}
```

## 大结果处理

### 流式截断

```python
class StreamingTruncator:
    def __init__(self, max_tokens: int = 8000):
        self.max_tokens = max_tokens
    
    def truncate(self, content: str, context_hint: str = "") -> TruncatedContent:
        tokens = self.tokenize(content)
        
        if len(tokens) <= self.max_tokens:
            return TruncatedContent(
                content=content,
                truncated=False,
                truncation_point=None
            )
        
        # 策略1：如果有上下文提示，优先保留相关部分
        if context_hint:
            relevant = self.find_relevant_sections(content, context_hint)
            if len(relevant) > self.max_tokens * 0.6:
                return self.merge_and_truncate(relevant, self.max_tokens)
        
        # 策略2：默认保留开头和结尾
        return self.head_tail_truncate(content, self.max_tokens)
    
    def head_tail_truncate(self, content: str, max_tokens: int) -> TruncatedContent:
        tokens = self.tokenize(content)
        half = max_tokens // 2
        
        head_tokens = tokens[:half]
        tail_tokens = tokens[-half:]
        
        head_content = self.detokenize(head_tokens)
        tail_content = self.detokenize(tail_tokens)
        
        return TruncatedContent(
            content=f"{head_content}\n... [{len(tokens) - max_tokens} tokens truncated] ...\n{tail_content}",
            truncated=True,
            truncation_point=half,
            head_content=head_content,
            tail_content=tail_content
        )
```

### 摘要生成

```python
class ToolResultSummarizer:
    def summarize(self, tool_result: ToolResult, max_tokens: int = 500) -> str:
        if tool_result.status == "failure":
            return f"[Error] {tool_result.error.message}"
        
        content = str(tool_result.data)
        
        # 提取关键统计信息
        stats = self.extract_stats(content)
        
        # 生成自然语言摘要
        prompt = f"""Summarize the following tool result in {max_tokens} tokens.
Focus on: what succeeded, key data points, and any important findings.

Tool Result:
{content}

Summary:"""
        
        summary = self.model.complete(prompt)
        return f"{summary}\n\nKey Stats: {stats}" if stats else summary
    
    def extract_stats(self, content: str) -> str:
        # 提取数字统计信息
        numbers = re.findall(r'\d+(?:\.\d+)?', content)
        if len(numbers) > 10:
            return f"Found {len(numbers)} numeric values"
        return ""
```

## 失败结果处理

### 错误信息格式化

```python
class ErrorFormatter:
    def format(self, error: ErrorInfo, tool_schema: dict) -> str:
        # 结构化错误信息，帮助模型理解失败原因
        error_type = self.classify_error(error)
        
        return f"""[Tool Execution Failed]
Error Type: {error_type}
Message: {error.message}

{self.format_recovery_suggestion(error_type, tool_schema)}"""
    
    def format_recovery_suggestion(self, error_type: str, tool_schema: dict) -> str:
        suggestions = {
            "invalid_parameters": f"Check parameter types. Required: {tool_schema.get('required', [])}",
            "permission_denied": "Consider using a different tool with lower permission requirements",
            "timeout": "Try simplifying the request or breaking it into smaller steps",
            "not_found": "Verify the resource exists and the path is correct",
            "rate_limit": "Wait before retrying or use a rate-limited version of the tool"
        }
        return f"Suggestion: {suggestions.get(error_type, 'Review the error and try a different approach')}"
```

## 上下文注入时机

### 即时 vs 批量

```python
class ContextInjectionStrategy:
    def __init__(self, strategy: str = "auto"):
        self.strategy = strategy
    
    def should_inject_immediately(self, tool_result: ToolResult) -> bool:
        if self.strategy == "immediate":
            return True
        elif self.strategy == "batch":
            return False
        else:  # auto
            # 高优先级工具立即注入
            if tool_result.metadata.priority == "high":
                return True
            # 失败结果立即注入（让模型及时调整）
            if tool_result.status == "failure":
                return True
            return False
    
    def batch_inject(self, results: List[ToolResult]) -> str:
        # 批量注入时，按工具类型分组
        grouped = defaultdict(list)
        for r in results:
            grouped[r.metadata.tool_name].append(r)
        
        formatted = []
        for tool_name, tool_results in grouped.items():
            formatted.append(self.format_batch(tool_name, tool_results))
        
        return '\n'.join(formatted)
```

## 失败模式与处理

| 失败模式 | 表现 | 解决方案 |
|---------|------|---------|
| 结果过长 | 上下文溢出 | 智能截断 + 摘要 |
| 关键信息丢失 | 模型决策失误 | 相关性保留策略 |
| 错误信息不清 | 模型无法恢复 | 结构化错误格式化 |
| 格式混乱 | 模型误读结果 | XML隔离 + 类型标注 |
