# 智能体上下文里应该放什么？

## 面试问题

智能体上下文里应该放什么？

## 考察重点

- 是否理解上下文内容的分层和优先级机制
- 能否说清楚不同类型信息的组织方式
- 是否能结合具体实现说明上下文构建策略
- 是否理解上下文容量与信息密度的权衡

## 核心结论

上下文不是越多越好，而是要在有限容量内提供模型决策所需的关键信息。核心原则是：放模型"不知道"但"需要知道"的信息，放"正在发生"和"刚发生"的信息，放"最可能相关"的信息。

## 上下文内容分类

### Claude Code的上下文结构

```python
class ClaudeCodeContextBuilder:
    def build(self, session: Session, task: str) -> List[dict]:
        context = []
        
        # Layer 1: System Prompt (不可变)
        context.append({
            "role": "system",
            "content": SYSTEM_PROMPT
        })
        
        # Layer 2: Tool Definitions (高优先级)
        context.append({
            "role": "system",
            "content": self.format_tools(session.available_tools)
        })
        
        # Layer 3: User Intent Summary (从历史中提取)
        intent = self.extract_user_intent(session.history)
        context.append({
            "role": "system",
            "content": f"User's current goal: {intent}"
        })
        
        # Layer 4: Relevant History (语义检索)
        relevant = self.retrieve_relevant_history(session, task)
        context.extend(relevant)
        
        # Layer 5: Current Task
        context.append({
            "role": "user",
            "content": task
        })
        
        return context
    
    def format_tools(self, tools: List[Tool]) -> str:
        formatted = ["# Available Tools", ""]
        for tool in tools:
            formatted.append(f"## {tool.name}")
            formatted.append(f"{tool.description}")
            formatted.append(f"Parameters: {json.dumps(tool.parameters)}")
            formatted.append("")
        return '\n'.join(formatted)
```

### Codex的上下文组织

```typescript
class CodexContextBuilder {
    private readonly MAX_CONTEXT_TOKENS = 200000;
    
    build(task: Task, session: Session): Message[] {
        const context: Message[] = [];
        const remaining = this.MAX_CONTEXT_TOKENS;
        
        // 1. 核心指令（占10%）
        const coreInstructions = this.getCoreInstructions();
        context.push(...coreInstructions);
        
        // 2. 工具Schema（占15%）
        const toolSchemas = this.getToolSchemas(session.availableTools);
        context.push(...toolSchemas);
        
        // 3. 任务上下文（动态分配）
        const taskContext = this.buildTaskContext(task, remaining * 0.4);
        context.push(...taskContext);
        
        // 4. 执行历史（最近N轮）
        const history = this.getRecentHistory(session, remaining * 0.3);
        context.push(...history);
        
        return context;
    }
    
    private buildTaskContext(task: Task, maxTokens: number): Message[] {
        return [
            { role: 'system', content: `Current Task: ${task.description}` },
            { role: 'system', content: `Task Type: ${task.type}` },
            { role: 'system', content: `Constraints: ${JSON.stringify(task.constraints)}` }
        ];
    }
}
```

## 上下文分层模型

### 分层优先级

```
┌─────────────────────────────────────────────────────────────┐
│  L0: 系统约束 (System Constraints)                    ~2K   │
│  - 安全策略、权限边界、不可覆盖指令                        │
├─────────────────────────────────────────────────────────────┤
│  L1: 工具定义 (Tool Definitions)                    ~3K    │
│  - 工具名称、参数schema、使用约束                          │
├─────────────────────────────────────────────────────────────┤
│  L2: 任务目标 (Task Objective)                      ~1K    │
│  - 用户意图摘要、当前任务描述、约束条件                     │
├─────────────────────────────────────────────────────────────┤
│  L3: 执行状态 (Execution State)                      ~5K    │
│  - 已完成步骤、当前进度、关键决策点                        │
├─────────────────────────────────────────────────────────────┤
│  L4: 外部知识 (External Knowledge)                 动态    │
│  - 检索结果、工具返回、相关记忆                           │
├─────────────────────────────────────────────────────────────┤
│  L5: 会话历史 (Conversation History)               动态    │
│  - 近期交互、关键用户反馈                                │
└─────────────────────────────────────────────────────────────┘
```

## 应该排除的内容

### 1. 冗余重复信息

```python
class RedundancyFilter:
    def filter(self, context_items: List[ContextItem]) -> List[ContextItem]:
        filtered = []
        for item in context_items:
            # 检查是否与已有内容高度重复
            is_duplicate = False
            for existing in filtered:
                if self.semantic_similarity(item.content, existing.content) > 0.9:
                    is_duplicate = True
                    break
            
            if not is_duplicate:
                filtered.append(item)
        
        return filtered
```

### 3. 过时信息

```python
class StalenessFilter:
    def __init__(self, max_age_hours: int = 24):
        self.max_age = max_age_hours * 3600
    
    def filter(self, items: List[ContextItem]) -> List[ContextItem]:
        now = time.time()
        return [
            item for item in items
            if now - item.timestamp < self.max_age
            or item.importance > 0.8  # 高重要性内容不受时间限制
        ]
```

## 内容选择算法

### 重要性评分

```python
class ImportanceScorer:
    def score(self, item: ContextItem, current_task: Task) -> float:
        scores = {}
        
        # 相关性分数（与当前任务）
        scores['relevance'] = self.calculate_relevance(item, current_task)
        
        # 时效性分数
        scores['recency'] = self.calculate_recency(item)
        
        # 唯一性分数
        scores['uniqueness'] = self.calculate_uniqueness(item)
        
        # 重要性固有分数
        scores['inherent'] = item.inherent_importance
        
        # 加权求和
        weights = {'relevance': 0.4, 'recency': 0.2, 'uniqueness': 0.1, 'inherent': 0.3}
        return sum(scores[k] * weights[k] for k in weights)
```

### 容量分配

```python
class ContextCapacityAllocator:
    def __init__(self, max_total_tokens: int = 200000):
        self.max = max_total_tokens
    
    def allocate(self, items: List[ContextItem]) -> List[ContextItem]:
        # 按重要性排序
        sorted_items = sorted(items, key=lambda x: x.importance, reverse=True)
        
        allocated = []
        used_tokens = 0
        
        for item in sorted_items:
            item_tokens = self.count_tokens(item)
            if used_tokens + item_tokens <= self.max:
                allocated.append(item)
                used_tokens += item_tokens
            elif len(allocated) < 10:  # 至少保留10个最重要的
                continue
            else:
                break
        
        return allocated
```

## 失败模式与处理

| 失败模式 | 表现 | 解决方案 |
|---------|------|---------|
| 上下文不足 | 模型缺乏决策必要信息 | 动态补充关键上下文 |
| 上下文冗余 | 关键信息被稀释 | 优先级过滤 + 容量控制 |
| 信息过时 | 模型基于过期信息决策 | 定期清理 + 有效性标注 |
| 类型混杂 | 模型混淆不同来源信息 | 分层隔离 + 格式标准化 |
