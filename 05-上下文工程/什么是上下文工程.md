# 什么是上下文工程？

## 面试问题

什么是上下文工程？

## 考察重点

- 是否理解上下文工程的核心概念和目标
- 是否能说明上下文管理在智能体系统中的地位
- 是否能解释上下文与模型输出质量的关系

## 核心结论

上下文工程是**管理智能体"看什么"**的系统性方法。核心挑战是：在有限的上下文窗口内，动态选择最相关的信息，让模型做出正确决策。

## 参考回答

### 上下文工程的本质

上下文工程解决三个核心问题：

1. **信息选择**：从海量信息中挑选最相关的
2. **信息组织**：按对模型决策最有帮助的方式组织
3. **信息压缩**：在有限空间内保留最多价值

### Claude Code的上下文架构

Claude Code的上下文管理采用了分层架构：

```typescript
interface ContextLayer {
  name: string;
  priority: number;
  maxTokens: number;
  ttl?: number;
}

class ClaudeCodeContextManager {
  private layers: ContextLayer[] = [
    { name: "system", priority: 100, maxTokens: 2000 },
    { name: "user_intent", priority: 90, maxTokens: 3000 },
    { name: "recent_history", priority: 80, maxTokens: 8000 },
    { name: "retrieved", priority: 70, maxTokens: 6000 },
    { name: "working_memory", priority: 60, maxTokens: 4000 },
    { name: "tool_schemas", priority: 50, maxTokens: 2000 },
  ];

  async buildContext(request: AgentRequest): Promise<LLMMessage[]> {
    const available = await this.getAvailableContent(request);
    const scored = this.scoreAndSort(available, request.task);

    const selected: LLMMessage[] = [];
    let tokensUsed = 0;

    for (const item of scored) {
      if (tokensUsed + item.tokens > this.getBudget(item.layer)) {
        if (item.canSummarize) {
          const summary = await this.summarize(item);
          tokensUsed += this.countTokens(summary);
          selected.push(summary);
        }
        continue;
      }
      tokensUsed += item.tokens;
      selected.push(item.content);
    }

    return this.formatForModel(selected);
  }

  private scoreAndSort(content: Content[], task: Task): ScoredContent[] {
    return content.map(item => ({
      item,
      score: this.calculateRelevance(item, task) +
             (item.layer.priority * 0.1) +
             (item.recencyScore * 0.2)
    })).sort((a, b) => b.score - a.score);
  }
}
```

### Codex的上下文管理

Codex使用动态上下文窗口：

```python
class DynamicContextWindow:
    def __init__(self, model: str):
        self.model = model
        self.model_context_limits = {
            "gpt-4": 128000,
            "gpt-4-turbo": 128000,
            "gpt-3.5-turbo": 16385,
        }
        self.max_tokens = self.model_context_limits.get(model, 16385)

    def build(
        self,
        task: Task,
        history: List[Message],
        knowledge: VectorStore,
        config: ContextConfig
    ) -> Context:
        system_prompt = self._build_system_prompt(task)
        available = self.max_tokens - self.count_tokens(system_prompt)

        retrieved = self._retrieve_knowledge(knowledge, task, available * 0.4)

        remaining = available - self.count_tokens(retrieved)
        recent = self._select_recent_history(history, remaining * 0.5)

        return Context(
            system=system_prompt,
            retrieved=retrieved,
            recent=recent,
            metadata={
                "total_tokens": self.count_tokens(system_prompt + retrieved + recent),
                "strategy": "dynamic_window"
            }
        )

    def _retrieve_knowledge(
        self,
        knowledge: VectorStore,
        task: Task,
        token_budget: int
    ) -> str:
        query_embedding = self.embed(task.description)
        candidates = knowledge.similarity_search(
            query_embedding,
            filter=lambda d: d.metadata.get("relevance_score", 0) > 0.7
        )

        selected = []
        tokens_used = 0
        for doc in candidates:
            doc_tokens = self.count_tokens(doc.content)
            if tokens_used + doc_tokens <= token_budget:
                selected.append(doc)
                tokens_used += doc_tokens
            elif len(selected) > 3:
                break

        return "\n\n".join(doc.content for doc in selected)
```

## 深度展开

### 第一层：上下文分层

**信息分层模型**：

```python
class ContextLayer(Enum):
    SYSTEM = 1        # 系统级指令、角色定义
    USER_INTENT = 2   # 用户当前意图、目标
    TASK_CONTEXT = 3  # 任务相关背景
    RETRIEVED = 4     # 检索到的知识
    HISTORY = 5       # 对话历史
    TOOL_SCHEMA = 6   # 工具定义
    WORKING = 7       # 工作记忆

class LayeredContextManager:
    def __init__(self):
        self.layers = {
            ContextLayer.SYSTEM: LayerConfig(max_tokens=2000, ttl=None),
            ContextLayer.USER_INTENT: LayerConfig(max_tokens=3000, ttl=None),
            ContextLayer.TASK_CONTEXT: LayerConfig(max_tokens=5000, ttl=3600),
            ContextLayer.RETRIEVED: LayerConfig(max_tokens=10000, ttl=1800),
            ContextLayer.HISTORY: LayerConfig(max_tokens=15000, ttl=None),
            ContextLayer.TOOL_SCHEMA: LayerConfig(max_tokens=3000, ttl=None),
            ContextLayer.WORKING: LayerConfig(max_tokens=5000, ttl=300),
        }

    def add_to_layer(self, layer: ContextLayer, content: str):
        config = self.layers[layer]
        tokens = self.count_tokens(content)

        if tokens > config.max_tokens:
            if layer.can_summarize:
                content = self.summarize(content, config.max_tokens)
            else:
                content = content[:config.max_tokens]

        self.context[layer] = content
        self.context_timestamps[layer] = time.time()
```

**分层的重要性**：

| 层级 | 特点 | 更新频率 | 压缩方式 |
|-----|------|---------|---------|
| SYSTEM | 稳定，几乎不变 | 低 | 不压缩 |
| USER_INTENT | 随用户输入变化 | 每次请求 | 不压缩 |
| TASK_CONTEXT | 任务相关背景 | 中 | 可摘要 |
| RETRIEVED | 检索获得 | 每次检索 | 可截断 |
| HISTORY | 对话历史 | 持续增长 | 摘要/丢弃 |
| TOOL_SCHEMA | 工具定义 | 低 | 不压缩 |

### 第二层：上下文构建

**动态上下文构建**：

```python
class ContextBuilder:
    def __init__(self, config: ContextConfig):
        self.config = config
        self.embedder = Embedder(config.embedding_model)
        self.retriever = Retriever(config.vector_store)

    async def build(self, task: Task, session: Session) -> BuiltContext:
        components = []

        system = self._build_system_component(task)
        components.append(("system", system, self.count_tokens(system)))

        intent = self._build_intent_component(task, session)
        components.append(("intent", intent, self.count_tokens(intent)))

        retrieved = await self._build_retrieved_component(task)
        components.append(("retrieved", retrieved, self.count_tokens(retrieved)))

        history = self._build_history_component(session)
        components.append(("history", history, self.count_tokens(history)))

        fitted = self._fit_to_budget(components)

        return BuiltContext(
            content=fitted,
            total_tokens=sum(t for _, _, t in fitted),
            layers_used=[name for name, _, _ in fitted]
        )

    def _fit_to_budget(
        self,
        components: List[Tuple[str, str, int]]
    ) -> List[Tuple[str, str, int]]:
        budget = self.config.max_tokens
        fitted = []
        total = 0

        priority_order = ["system", "intent", "retrieved", "history"]
        component_map = {name: (c, t) for name, c, t in components}

        for name in priority_order:
            if name not in component_map:
                continue
            content, tokens = component_map[name]
            if total + tokens <= budget:
                fitted.append((name, content, tokens))
                total += tokens
            else:
                remaining = budget - total
                if remaining > 1000:
                    fitted.append((name, content[:remaining*4], remaining))
                break

        return fitted
```

### 第三层：上下文压缩

**智能摘要策略**：

```python
class ContextCompressor:
    def __init__(self):
        self.summary_model = load_model("gpt-3.5-turbo")

    async def compress(
        self,
        content: str,
        max_tokens: int,
        preserve_key_info: List[str] = None
    ) -> str:
        current_tokens = self.count_tokens(content)
        if current_tokens <= max_tokens:
            return content

        prompt = f"""Summarize the following text to approximately {max_tokens} tokens.
Keep the following key information: {preserve_key_info or ['main points', 'decisions made', 'important details']}

Text:
{content}

Summary:"""

        response = await self.summary_model.complete(prompt)
        return response.text

    def compress_history(
        self,
        messages: List[Message],
        max_tokens: int
    ) -> List[Message]:
        grouped = self._group_by_task(messages)
        compressed = []

        for group in grouped:
            tokens = self.count_tokens(group.content)
            if tokens <= max_tokens / len(grouped):
                compressed.append(group)
            else:
                summary = await self.compress(
                    group.content,
                    max_tokens // len(grouped),
                    group.key_points
                )
                compressed.append(Message(
                    type="summary",
                    content=summary,
                    metadata={"original_count": len(group.messages)}
                ))

        return compressed
```

### 第四层：上下文质量评估

**质量指标体系**：

```python
@dataclass
class ContextQualityMetrics:
    relevance_score: float      # 上下文与任务的相关性
    completeness_score: float   # 关键信息完整度
    conciseness_score: float     # 信息冗余度
    freshness_score: float       # 信息时效性

class ContextQualityEvaluator:
    def evaluate(
        self,
        context: BuiltContext,
        task: Task
    ) -> ContextQualityMetrics:
        return ContextQualityMetrics(
            relevance_score=self._evaluate_relevance(context, task),
            completeness_score=self._evaluate_completeness(context, task),
            conciseness_score=self._evaluate_conciseness(context),
            freshness_score=self._evaluate_freshness(context)
        )

    def _evaluate_relevance(
        self,
        context: BuiltContext,
        task: Task
    ) -> float:
        task_embedding = self.embed(task.description)
        context_embedding = self.embed(context.content)
        return cosine_similarity(task_embedding, context_embedding)

    def _evaluate_completeness(
        self,
        context: BuiltContext,
        task: Task
    ) -> float:
        required = task.required_information or []
        covered = sum(1 for req in required if req in context.content)
        return covered / len(required) if required else 1.0
```

## 工程实践

```python
class ContextEngineeringSystem:
    def __init__(self, config: ContextConfig):
        self.layer_manager = LayeredContextManager()
        self.builder = ContextBuilder(config)
        self.compressor = ContextCompressor()
        self.evaluator = ContextQualityEvaluator()

    async def prepare_context(
        self,
        task: Task,
        session: Session
    ) -> ContextResult:
        context = await self.builder.build(task, session)

        quality = self.evaluator.evaluate(context, task)

        if quality.relevance_score < 0.6:
            context = await self._improve_relevance(context, task)

        if quality.conciseness_score < 0.5:
            context = await self._improve_conciseness(context, task)

        return ContextResult(
            context=context,
            quality=quality,
            metrics={
                "total_tokens": context.total_tokens,
                "compression_ratio": context.compression_ratio
            }
        )
```

## 常见误区

- **把上下文当存储**：不是把所有信息塞进去，而是选择最相关的
- **忽视信息时效性**：旧信息可能比没信息更危险
- **不进行分层管理**：混合不同类型信息降低检索效率
- **压缩损失关键信息**：粗暴截断导致重要细节丢失
- **不评估上下文质量**：无法持续优化上下文策略
