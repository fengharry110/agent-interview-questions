# 如何设计长期记忆？

## 面试问题

如何设计长期记忆？

## 考察重点

- 是否理解向量存储、索引结构和检索策略
- 能否说清楚记忆持久化的时机和内容选择
- 是否能结合具体系统说明实现细节（Mem0、RecMemory等）
- 是否理解记忆检索与任务的相关性计算

## 核心结论

长期记忆的核心问题不是"存储"，而是"检索"——如何在海量记忆中快速找到与当前任务相关的内容。向量检索是基础，但要解决索引更新、一致性、遗忘机制等工程挑战。

## 记忆存储架构

### 分层记忆系统

```
┌─────────────────────────────────────────────────┐
│                 Working Memory                   │
│         (当前任务上下文, ~10K tokens)           │
└─────────────────────────────────────────────────┘
                      ↓ 定期持久化
┌─────────────────────────────────────────────────┐
│               Short-Term Memory                  │
│           (最近100条交互, ~50K tokens)          │
└─────────────────────────────────────────────────┘
                      ↓ 选择性升级
┌─────────────────────────────────────────────────┐
│               Long-Term Memory                  │
│           (向量数据库, 百万级记忆)               │
└─────────────────────────────────────────────────┘
```

### 记忆的生命周期

```python
class MemoryLifecycle:
    def __init__(self):
        self.working = WorkingMemory()
        self.short_term = ShortTermMemory()
        self.long_term = VectorStore()
    
    def on_interaction(self, interaction: Interaction):
        self.working.add(interaction)
        
        if self.working.should_persist():
            self.short_term.add(self.working.get_summary())
        
        if self.short_term.should_upgrade(interaction):
            self.long_term.add(
                self.extract_memories(interaction),
                metadata=self.build_metadata(interaction)
            )
    
    def retrieve(self, query: str, limit: int = 10) -> List[Memory]:
        # 1. 先从工作记忆检索
        working_results = self.working.search(query, limit=3)
        # 2. 再从短期记忆检索
        short_results = self.short_term.search(query, limit=5)
        # 3. 最后从长期记忆检索
        long_results = self.long_term.search(query, limit=limit)
        return self.merge_results([working_results, short_results, long_results])
```

### 索引结构设计

```python
class MemoryIndex:
    def __init__(self, vector_dim: int = 1536):
        self.index = faiss.IndexFlatIP(vector_dim)  # 内积相似度
        self.memories = []
        self.metadata = []
    
    def add(self, text: str, embedding: np.ndarray, meta: dict):
        # L2归一化使内积等价于余弦相似度
        normalized = embedding / np.linalg.norm(embedding)
        self.index.add(normalized.reshape(1, -1))
        self.memories.append(text)
        self.metadata.append(meta)
    
    def search(self, query_embedding: np.ndarray, k: int = 10) -> List[SearchResult]:
        normalized = query_embedding / np.linalg.norm(query_embedding)
        scores, indices = self.index.search(normalized.reshape(1, -1), k)
        return [
            SearchResult(
                text=self.memories[i],
                score=float(scores[0][j]),
                metadata=self.metadata[i]
            )
            for j, i in enumerate(indices[0])
        ]
```

### HNSW索引优化

对于大规模记忆，使用HNSW分层可导航小世界图：

```python
class HNSWMemoryIndex:
    def __init__(self, vector_dim: int, ef_construction: int = 200):
        self.index = faiss.IndexHNSWFlat(vector_dim, 32)
        self.index.hnsw.efConstruction = ef_construction
        self.index.hnsw.efSearch = 50  # 检索时探索点数
    
    def search(self, query: np.ndarray, k: int = 10) -> Tuple[np.ndarray, np.ndarray]:
        return self.index.search(query, k)
```

HNSW参数选择：
- `M`：每个节点的连接数（16-64），越大越精确但内存占用越高
- `efConstruction`：构建时探索点数，越大索引质量越高但构建越慢
- `efSearch`：检索时探索点数，越大越精确但延迟越高

## 记忆提取策略

### 什么信息值得存入长期记忆

```python
class MemoryExtractor:
    def __init__(self, importance_threshold: float = 0.7):
        self.threshold = importance_threshold
    
    def should_persist(self, interaction: Interaction) -> bool:
        signals = {
            'user_explicit_instruction': interaction.has_explicit_preference(),
            'repeated_pattern': self.is_repeated(interaction),
            'error_recovery': interaction.contains_error_recovery(),
            'complex_task': self.is_complex_task(interaction),
            'decision_point': interaction.has_significant_decision()
        }
        importance = sum(s for s in signals.values()) / len(signals)
        return importance >= self.threshold
    
    def extract_memories(self, interaction: Interaction) -> List[str]:
        memories = []
        if self.is_repeated(interaction):
            memories.append(f"用户偏好: {self.extract_preference(interaction)}")
        if interaction.contains_error_recovery():
            memories.append(f"问题解决: {self.extract_solution(interaction)}")
        if interaction.has_significant_decision():
            memories.append(f"决策: {self.extract_decision(interaction)}")
        return memories
```

### 记忆的粒度控制

```python
class MemoryChunker:
    def chunk(self, text: str, max_tokens: int = 256) -> List[Chunk]:
        sentences = self.split_sentences(text)
        chunks = []
        current_chunk = []
        current_tokens = 0
        
        for sentence in sentences:
            sentence_tokens = self.count_tokens(sentence)
            if current_tokens + sentence_tokens > max_tokens:
                if current_chunk:
                    chunks.append(self.create_chunk(current_chunk))
                current_chunk = [sentence]
                current_tokens = sentence_tokens
            else:
                current_chunk.append(sentence)
                current_tokens += sentence_tokens
        
        if current_chunk:
            chunks.append(self.create_chunk(current_chunk))
        return chunks
    
    def create_chunk(self, sentences: List[str]) -> Chunk:
        return Chunk(
            content=' '.join(sentences),
            summary=self.summarize(sentences),
            keywords=self.extract_keywords(sentences)
        )
```

## 相关系统实现

### Mem0架构

Mem0是一个专门为AI智能体设计的记忆层：

```python
class Mem0Memory:
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.vector_store = QdrantClient()
        self.graph_store = Neo4jClient()
    
    def add(self, memory: str, type: str = "interaction"):
        embedding = self.embedding_model.encode(memory)
        self.vector_store.add(
            collection=self.user_id,
            vector=embedding,
            payload={"content": memory, "type": type}
        )
        
        # 关系型记忆存入图数据库
        entities = self.extract_entities(memory)
        for entity in entities:
            self.graph_store.create_entity(self.user_id, entity)
    
    def search(self, query: str, limit: int = 5) -> List[Memory]:
        query_embedding = self.embedding_model.encode(query)
        results = self.vector_store.search(
            collection=self.user_id,
            vector=query_embedding,
            limit=limit
        )
        return [self.reconstruct(r) for r in results]
    
    def get_recent(self, limit: int = 10) -> List[Memory]:
        return self.vector_store.get_recent(self.user_id, limit)
```

### RAG系统的记忆扩展

```typescript
class ExtendedContextService {
    constructor(
        private workingMemory: WorkingMemory,
        private shortTermMemory: ShortTermMemory,
        private longTermMemory: VectorStore,
        private graphMemory: GraphStore
    ) {}
    
    async buildContext(task: string, maxTokens: number): Promise<Context> {
        const context: ContextEntry[] = [];
        let remaining = maxTokens;
        
        // 1. 检索长期记忆（语义相关）
        const longTerm = await this.longTermMemory.search(task, remaining * 0.4);
        context.push(...longTerm);
        remaining -= this.countTokens(longTerm);
        
        // 2. 补充短期记忆（时间近）
        const shortTerm = await this.shortTermMemory.getRecent(remaining * 0.3);
        context.push(...shortTerm);
        remaining -= this.countTokens(shortTerm);
        
        // 3. 补充工作记忆（当前任务）
        const working = this.workingMemory.getCurrent(remaining * 0.2);
        context.push(working);
        
        // 4. 补充图谱关系（结构化知识）
        const graphRelations = await this.graphMemory.getRelatedEntities(task);
        context.push(...graphRelations.slice(0, remaining * 0.1));
        
        return this.prioritizeAndTruncate(context, maxTokens);
    }
}
```

## 遗忘机制

### 基于重要性的遗忘

```python
class ImportanceBasedForgetting:
    def __init__(self, decay_rate: float = 0.01):
        self.decay_rate = decay_rate
    
    def calculate_importance(self, memory: Memory) -> float:
        base = memory.initial_importance
        age = (now() - memory.created_at) / (30 * 24 * 3600)  # 月
        access_count = memory.access_count
        return base * math.exp(-self.decay_rate * age) * math.log(access_count + 2)
    
    def should_forget(self, memory: Memory) -> bool:
        return self.calculate_importance(memory) < 0.1
```

### 基于访问模式的遗忘

```python
class AccessPatternForgetting:
    def __init__(self):
        self.access_log = defaultdict(list)  # memory_id -> [(timestamp, accessed)]
    
    def record_access(self, memory_id: str):
        self.access_log[memory_id].append((now(), True))
        # 清理超过一年的记录
        self.access_log[memory_id] = [
            (t, a) for t, a in self.access_log[memory_id]
            if now() - t < 365 * 24 * 3600
        ]
    
    def get_access_frequency(self, memory_id: str) -> float:
        accesses = self.access_log.get(memory_id, [])
        if not accesses:
            return 0.0
        # 计算最近30天的访问频率
        recent = [(t, a) for t, a in accesses if now() - t < 30 * 24 * 3600]
        return len(recent) / 30
```

## 失败模式与处理

| 失败模式 | 表现 | 解决方案 |
|---------|------|---------|
| 索引膨胀 | 检索延迟增加 | 定期压缩 + 分层索引 |
| 语义漂移 | 相似记忆无法召回 | 定期重新索引 |
| 记忆冲突 | 同一事实多个版本 | 版本管理 + 冲突解决 |
| 隐私泄露 | 敏感信息被检索 | 差分隐私 + 访问控制 |
