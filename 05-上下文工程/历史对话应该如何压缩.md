# 历史对话应该如何压缩？

## 面试问题

历史对话应该如何压缩？

## 考察重点

- 是否理解对话历史压缩的必要性和策略
- 是否能设计有效的对话摘要算法
- 是否能平衡压缩率和信息保留度
- 是否能结合Claude Code/Codex实现说明工程方法

## 核心结论

对话压缩的核心是**选择性保留**：保留用户意图和关键决策，丢弃中间过程噪音。好的压缩策略能让模型在有限上下文中看到完整的任务轨迹。

## 参考回答

### 压缩策略分类

| 策略 | 适用场景 | 压缩率 | 信息损失 |
|-----|---------|-------|---------|
| **截断** | 对话早期无关内容 | 50-70% | 可能丢失隐含依赖 |
| **摘要** | 长对话段/低价值交互 | 60-80% | 细节可能丢失 |
| **选择性保留** | 高价值决策点 | 30-50% | 需要准确判断价值 |
| **层次化** | 超长对话 | 70-90% | 需要合理分层 |

### Claude Code的对话管理

Claude Code使用层次化对话压缩：

```typescript
class ConversationCompressor {
  private maxTokens: number;
  private minRelevanceScore = 0.3;

  compress(messages: Message[]): CompressedResult {
    if (this.estimateTokens(messages) <= this.maxTokens) {
      return { messages, compressed: false };
    }

    const system = messages[0];
    const rest = messages.slice(1);

    const grouped = this.groupByTopic(rest);
    const scored = this.scoreGroups(grouped);

    const selected: Message[] = [system];
    let tokens = this.countTokens(system);

    for (const group of scored) {
      const groupTokens = this.countTokens(group.content);
      if (tokens + groupTokens <= this.maxTokens * 0.9) {
        selected.push(...group.messages);
        tokens += groupTokens;
      } else {
        selected.push(this.summarizeGroup(group));
        tokens += this.countTokens(selected[selected.length - 1]);
      }
    }

    return {
      messages: selected,
      compressed: selected.length < messages.length,
      droppedCount: messages.length - selected.length
    };
  }

  private groupByTopic(messages: Message[]): MessageGroup[] {
    const groups: MessageGroup[] = [];
    let currentGroup: Message[] = [];
    let lastTopic = "";

    for (const msg of messages) {
      const topic = this.extractTopic(msg);
      if (topic !== lastTopic && currentGroup.length > 0) {
        groups.push(new MessageGroup(currentGroup));
        currentGroup = [];
      }
      currentGroup.push(msg);
      lastTopic = topic;
    }

    if (currentGroup.length > 0) {
      groups.push(new MessageGroup(currentGroup));
    }

    return groups;
  }

  private scoreGroups(groups: MessageGroup[]): MessageGroup[] {
    return groups
      .map(g => ({
        group: g,
        score: this.calculateGroupScore(g)
      }))
      .sort((a, b) => b.score - a.score);
  }

  private calculateGroupScore(group: MessageGroup): number {
    let score = 0;

    if (group.hasUserDecision) score += 50;
    if (group.hasToolCall) score += 20;
    if (group.hasError) score += 10;
    if (group.messageCount < 3) score += 5;

    score += group.recencyScore * 30;

    return score;
  }
}
```

### Codex的对话压缩

```python
class CodexConversationCompressor:
    def __init__(self, max_tokens: int):
        self.max_tokens = max_tokens

    def compress(self, messages: List[Message]) -> CompressionResult:
        if self.estimate_tokens(messages) <= self.max_tokens:
            return CompressionResult(messages=messages, compressed=False)

        messages_by_turn = self._group_by_turn(messages)

        importance_scores = []
        for turn in messages_by_turn:
            score = self._score_turn(turn)
            importance_scores.append((turn, score))

        importance_scores.sort(key=lambda x: x[1], reverse=True)

        selected = []
        tokens_used = 0
        for turn, score in importance_scores:
            turn_tokens = self._estimate_turn_tokens(turn)
            if tokens_used + turn_tokens <= self.max_tokens * 0.85:
                selected.append(turn)
                tokens_used += turn_tokens
            elif score > 0.7:
                summary = self._summarize_turn(turn)
                tokens_used += self._estimate_tokens(summary)
                selected.append([summary])
            else:
                continue

        selected.sort(key=lambda t: t[0].timestamp if hasattr(t[0], 'timestamp') else 0)

        return CompressionResult(
            messages=self._flatten(selected),
            compressed=True,
            original_count=len(messages),
            final_count=len(self._flatten(selected))
        )

    def _group_by_turn(self, messages: List[Message]) -> List[List[Message]]:
        turns = []
        current_turn = []
        last_user = None

        for msg in messages:
            if msg.role == "user" and msg != last_user:
                if current_turn:
                    turns.append(current_turn)
                current_turn = [msg]
                last_user = msg
            else:
                current_turn.append(msg)

        if current_turn:
            turns.append(current_turn)

        return turns
```

### 第一层：对话分块策略

**基于话题的分块**：

```python
class TopicBasedChunker:
    def __init__(self):
        self.topic_extractor = TopicExtractor()

    def chunk(self, messages: List[Message]) -> List[Chunk]:
        chunks = []
        current_chunk_messages = []
        current_topic = None

        for msg in messages:
            topic = self.topic_extractor.extract(msg)

            if current_topic and topic != current_topic:
                if current_chunk_messages:
                    chunks.append(Chunk(
                        messages=current_chunk_messages,
                        topic=current_topic,
                        importance=self._calculate_chunk_importance(current_chunk_messages)
                    ))
                current_chunk_messages = []

            current_chunk_messages.append(msg)
            current_topic = topic

        if current_chunk_messages:
            chunks.append(Chunk(
                messages=current_chunk_messages,
                topic=current_topic,
                importance=self._calculate_chunk_importance(current_chunk_messages)
            ))

        return chunks

    def _calculate_chunk_importance(self, messages: List[Message]) -> float:
        if not messages:
            return 0.0

        score = 0.0

        has_decision = any("decide" in m.content.lower() or "conclusion" in m.content.lower()
                          for m in messages)
        if has_decision:
            score += 0.4

        has_error = any(m.type == "error" for m in messages)
        if has_error:
            score += 0.2

        has_tool_call = any(m.type == "tool_call" for m in messages)
        if has_tool_call:
            score += 0.1

        score += min(len(messages) * 0.02, 0.2)

        return min(score, 1.0)
```

**基于时间的分块**：

```python
class TimeBasedChunker:
    def __init__(self, max_gap_seconds: int = 300):
        self.max_gap = max_gap_seconds

    def chunk(self, messages: List[Message]) -> List[Chunk]:
        chunks = []
        current_chunk = []
        last_timestamp = None

        for msg in messages:
            if not hasattr(msg, 'timestamp') or not msg.timestamp:
                current_chunk.append(msg)
                continue

            if last_timestamp and (msg.timestamp - last_timestamp).seconds > self.max_gap:
                chunks.append(Chunk(messages=current_chunk))
                current_chunk = []

            current_chunk.append(msg)
            last_timestamp = msg.timestamp

        if current_chunk:
            chunks.append(Chunk(messages=current_chunk))

        return chunks
```

### 第三层：压缩质量控制

**重要性评分**：

```python
class MessageImportanceScorer:
    def __init__(self):
        self.importance_keywords = {
            "high": ["decide", "conclusion", "important", "critical", "must", "should"],
            "medium": ["consider", "might", "could", "suggestion", "recommend"],
            "low": ["maybe", "perhaps", "possibly", "might not"]
        }

    def score(self, message: Message) -> float:
        base_score = 0.5

        if message.role == "system":
            return 1.0

        if message.role == "user":
            base_score = 0.7

        content_lower = message.content.lower()

        for keyword in self.importance_keywords["high"]:
            if keyword in content_lower:
                base_score += 0.15

        for keyword in self.importance_keywords["low"]:
            if keyword in content_lower:
                base_score -= 0.1

        if message.type == "tool_result":
            base_score += 0.1

        if message.is_error:
            base_score += 0.2

        return max(0.0, min(1.0, base_score))
```

**信息密度评估**：

```python
class InformationDensityAnalyzer:
    def analyze(self, message: Message) -> float:
        tokens = self.count_tokens(message.content)
        if tokens == 0:
            return 0.0

        entities = self._count_entities(message)
        numbers = self._count_numbers(message)
        technical_terms = self._count_technical_terms(message)

        density = (entities + numbers + technical_terms) / tokens * 100

        return min(density / 10, 1.0)

    def should_preserve_detailed_content(self, message: Message) -> bool:
        return self.analyze(message) > 0.3 or message.type == "code"
```

### 第四层：自适应压缩

**基于对话阶段的压缩**：

```python
class AdaptiveCompressor:
    def __init__(self):
        self.phase_detector = ConversationPhaseDetector()
        self.base_compressor = ConversationCompressor()

    def compress(
        self,
        messages: List[Message],
        max_tokens: int
    ) -> CompressedResult:
        phase = self.phase_detector.detect(messages)

        if phase == "exploration":
            return self._compress_exploration(messages, max_tokens)
        elif phase == "execution":
            return self._compress_execution(messages, max_tokens)
        elif phase == "resolution":
            return self._compress_resolution(messages, max_tokens)
        else:
            return self.base_compressor.compress(messages, max_tokens)

    def _compress_exploration(self, messages: List[Message], max_tokens: int):
        preserved = [messages[0]]

        for msg in messages[1:]:
            if msg.role == "user" and "?" in msg.content:
                preserved.append(msg)

        if len(preserved) < len(messages):
            preserved.append(self._create_summary_of_discarded(messages[len(preserved)-1:]))

        return CompressedResult(messages=preserved, compressed=True)

    def _compress_execution(self, messages: List[Message], max_tokens: int):
        return self.base_compressor.compress(messages, max_tokens)

    def _compress_resolution(self, messages: List[Message], max_tokens: int):
        preserved = messages[-10:] if len(messages) > 10 else messages

        summary = self._create_summary_of_discarded(messages[:-10])
        preserved.insert(0, summary)

        return CompressedResult(messages=preserved, compressed=True)
```

## 工程实践

```python
class ConversationCompressionPipeline:
    def __init__(self, config: CompressionConfig):
        self.max_tokens = config.max_tokens
        self.chunker = TopicBasedChunker()
        self.summarizer = StructuredSummarizer()
        self.importance_scorer = MessageImportanceScorer()
        self.density_analyzer = InformationDensityAnalyzer()

    async def compress(
        self,
        messages: List[Message],
        context: CompressionContext = None
    ) -> CompressionResult:
        if self.estimate_tokens(messages) <= self.max_tokens:
            return CompressionResult(messages=messages, compressed=False)

        chunks = self.chunker.chunk(messages)

        scored_chunks = []
        for chunk in chunks:
            importance = sum(self.importance_scorer.score(m) for m in chunk.messages) / len(chunk.messages)
            density = sum(self.density_analyzer.analyze(m) for m in chunk.messages) / len(chunk.messages)
            scored_chunks.append((chunk, importance, density))

        scored_chunks.sort(key=lambda x: (x[1], x[2]), reverse=True)

        result = []
        tokens_used = 0

        for chunk, importance, density in scored_chunks:
            chunk_tokens = self.estimate_tokens(chunk.messages)

            if tokens_used + chunk_tokens <= self.max_tokens:
                result.extend(chunk.messages)
                tokens_used += chunk_tokens
            elif importance > 0.6 and density > 0.3:
                summary = self.summarizer.summarize(chunk.messages)
                summary_text = self.summarizer.to_text(summary)
                tokens_used += self.estimate_tokens(summary_text)
                if tokens_used <= self.max_tokens:
                    result.append(Message(type="summary", content=summary_text))
            else:
                summary_text = await self.summarizer.create_summary(chunk.messages)
                tokens_used += self.estimate_tokens(summary_text)
                if tokens_used <= self.max_tokens:
                    result.append(Message(type="summary", content=summary_text))

        result.sort(key=lambda m: m.timestamp if hasattr(m, 'timestamp') else 0)

        return CompressionResult(
            messages=result,
            compressed=True,
            original_count=len(messages),
            final_count=len(result),
            summary_count=sum(1 for m in result if m.type == "summary")
        )
```

## 常见误区

- **压缩率过高**：导致关键信息丢失，模型无法理解任务进展
- **粗暴截断**：丢失对话早期的重要上下文依赖
- **忽略对话结构**：破坏用户意图和响应的对应关系
- **不保留决策点**：丢失关键结论影响后续推理
- **压缩后不验证**：无法保证压缩质量
