# 如何处理上下文窗口不够的问题？

## 面试问题

如何处理上下文窗口不够的问题？

## 考察重点

- 是否理解上下文窗口限制的根本原因
- 是否能设计多层次的解决方案
- 是否能平衡成本、性能和信息完整性
- 是否能结合Claude Code/Codex实现说明工程方法

## 核心结论

处理上下文窗口不够的核心思路是：**多级取舍**。优先保留最相关的，压缩次相关的，丢弃最不重要的。配合分层记忆架构，实现近远期信息的分离存储和检索。

## 参考回答

### 问题分析

上下文窗口不够的本质是：**信息供给 > 信息容量**。解决方案可以从三个维度入手：

| 维度 | 策略 | 效果 |
|-----|------|-----|
| **输入端** | 信息过滤/压缩 | 减少进入窗口的内容量 |
| **窗口端** | 动态窗口管理 | 优化窗口利用率 |
| **架构端** | 分层记忆 | 扩展有效容量 |

### Claude Code的窗口管理

Claude Code实现了智能窗口管理：

```typescript
class ContextWindowManager {
  private model: string;
  private modelLimits: Record<string, number> = {
    "claude-3-opus": 200000,
    "claude-3-sonnet": 200000,
    "claude-3-haiku": 200000,
  };

  private reserveTokens: number = 2000;

  constructor(model: string) {
    this.model = model;
  }

  getEffectiveLimit(): number {
    return this.modelLimits[this.model] - this.reserveTokens;
  }

  async fitToWindow(context: ContextComponents): Promise<FittedContext> {
    const limit = this.getEffectiveLimit();
    let remaining = limit;

    const prioritized: PrioritizedContent[] = [];

    for (const component of context.components) {
      const priority = this.calculatePriority(component);
      prioritized.push({ component, priority });
    }

    prioritized.sort((a, b) => b.priority - a.priority);

    const selected: ContextComponent[] = [];
    for (const { component } of prioritized) {
      const tokens = this.estimateTokens(component);

      if (tokens <= remaining) {
        selected.push(component);
        remaining -= tokens;
      } else if (component.canSummarize) {
        const summarized = await this.summarize(component, remaining);
        selected.push(summarized);
        remaining = 0;
        break;
      } else if (remaining > 500) {
        const truncated = this.truncate(component, remaining);
        selected.push(truncated);
        remaining = 0;
        break;
      }
    }

    return {
      components: selected,
      droppedCount: context.components.length - selected.length,
      totalTokens: limit - remaining
    };
  }

  private calculatePriority(component: ContextComponent): number {
    let priority = component.basePriority || 0;

    if (component.type === "system") priority += 100;
    if (component.type === "user_intent") priority += 90;
    if (component.type === "retrieved") priority += 70;
    if (component.isRecent) priority += 30;

    return priority;
  }
}
```

### Codex的层级记忆

```python
class HierarchicalMemory:
    def __init__(self):
        self.working_memory = WorkingMemory(capacity=10000)
        self.short_term = ShortTermMemory(capacity=50000)
        self.long_term = LongTermMemory()

    def get_context(
        self,
        task: str,
        max_tokens: int
    ) -> str:
        query_embedding = self.embed(task)

        relevant_short = self.short_term.retrieve(
            query_embedding,
            max_tokens=max_tokens * 0.6
        )

        if len(relevant_short) < max_tokens * 0.6:
            relevant_long = self.long_term.retrieve(
                query_embedding,
                max_tokens=max_tokens * 0.3
            )
        else:
            relevant_long = ""

        current = self.working_memory.get_content(max_tokens * 0.1)

        return self.combine([relevant_long, relevant_short, current])

    def store(self, content: str, metadata: dict):
        tokens = self.count_tokens(content)

        self.working_memory.add(content, metadata)

        if tokens > 5000 or self.working_memory.is_full():
            summarized = self.short_term.add(content, metadata)
            self.working_memory.clear()
        else:
            summarized = None

        if summarized:
            self.long_term.add(summarized, metadata)
```

### 第一层：信息过滤与选择

**相关性评分与过滤**：

```python
class RelevanceBasedFilter:
    def __init__(self, embedder: Embedder):
        self.embedder = embedder
        self.relevance_threshold = 0.5

    async def filter(
        self,
        items: List[ContextItem],
        task: str,
        max_tokens: int
    ) -> List[ContextItem]:
        task_embedding = await self.embedder.embed(task)

        scored_items = []
        for item in items:
            item_embedding = await self.embedder.embed(item.content)
            score = cosine_similarity(task_embedding, item_embedding)
            scored_items.append((item, score))

        scored_items.sort(key=lambda x: x[1], reverse=True)

        selected = []
        tokens_used = 0

        for item, score in scored_items:
            if score < self.relevance_threshold:
                continue

            item_tokens = self.count_tokens(item.content)
            if tokens_used + item_tokens <= max_tokens:
                selected.append(item)
                tokens_used += item_tokens
            elif score > 0.7 and item_tokens > max_tokens * 0.3:
                summarized = await self.summarize(item, max_tokens - tokens_used)
                selected.append(summarized)
                break

        return selected
```

```python
class LayeredInformationSelector:
    def __init__(self):
        self.budget分配 = {
            "current_task": 0.4,
            "recent_history": 0.3,
            "retrieved_knowledge": 0.2,
            "system_prompt": 0.1,
        }

    def select(
        self,
        context: ContextData,
        task: Task,
        max_tokens: int
    ) -> SelectedContext:
        selections = {}

        selections["system_prompt"] = self._select_system(
            context.system_prompt,
            int(max_tokens * self.budget分配["system_prompt"])
        )

        current_tokens = int(max_tokens * self.budget分配["current_task"])
        selections["current_task"] = self._select_current_task(
            context.task_context,
            task,
            current_tokens
        )

        recent_tokens = int(max_tokens * self.budget分配["recent_history"])
        selections["recent_history"] = self._select_recent(
            context.recent_history,
            task,
            recent_tokens
        )

        retrieved_tokens = int(max_tokens * self.budget分配["retrieved_knowledge"])
        selections["retrieved_knowledge"] = self._select_retrieved(
            context.retrieved,
            task,
            retrieved_tokens
        )

        return SelectedContext(**selections)
```

```python
class SemanticCompressor:
    def __init__(self):
        self.summary_model = load_model("gpt-4-turbo")

    async def compress(
        self,
        content: str,
        max_tokens: int,
        preserve_types: List[str] = None
    ) -> str:
        preserve_types = preserve_types or ["decision", "error", "user_intent"]

        preserve_instruction = ""
        if preserve_types:
            preserve_instruction = f"\n\nIMPORTANT: Preserve the following types of information: {', '.join(preserve_types)}"

        prompt = f"""Compress the following text to approximately {max_tokens} tokens while retaining the most important information.

{preserve_instruction}

Original text:
{content}

Compressed version:"""

        response = await self.summary_model.complete(prompt)
        return response.text
```

```python
        self,

        return "\n".join(lines)
```

### 第三层：分层记忆架构

**工作记忆管理**：

```python
class WorkingMemory:
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.items: List[MemoryItem] = []
        self.current_tokens = 0

    def add(self, content: str, metadata: dict = None):
        tokens = self.count_tokens(content)

        while self.current_tokens + tokens > self.capacity and self.items:
            removed = self.items.pop(0)
            self.current_tokens -= self.count_tokens(removed.content)

        self.items.append(MemoryItem(content=content, metadata=metadata))
        self.current_tokens += tokens

    def get_content(self, max_tokens: int) -> str:
        result = []
        tokens_used = 0

        for item in reversed(self.items):
            item_tokens = self.count_tokens(item.content)
            if tokens_used + item_tokens <= max_tokens:
                result.insert(0, item.content)
                tokens_used += item_tokens
            else:
                break

        return "\n\n".join(result)

    def clear(self):
        self.items = []
        self.current_tokens = 0
```

**长期记忆存储与检索**：

```python
class LongTermMemoryStore:
    def __init__(self, vector_store: VectorStore):
        self.vector_store = vector_store
        self.entity_index = {}
        self.fact_index = {}

    def add(self, content: str, metadata: dict = None):
        content_embedding = self.embed(content)
        self.vector_store.add(content_embedding, content, metadata)

        entities = self._extract_entities(content)
        for entity in entities:
            if entity.text not in self.entity_index:
                self.entity_index[entity.text] = []
            self.entity_index[entity.text].append({
                "content": content,
                "embedding": content_embedding
            })

    def retrieve(
        self,
        query_embedding: List[float],
        max_tokens: int,
        entity_filter: List[str] = None
    ) -> str:
        if entity_filter:
            candidates = []
            for entity_name in entity_filter:
                if entity_name in self.entity_index:
                    candidates.extend(self.entity_index[entity_name])

            candidates.sort(
                key=lambda x: cosine_similarity(query_embedding, x["embedding"]),
                reverse=True
            )

            results = []
            tokens_used = 0
            for candidate in candidates:
                if tokens_used + self.count_tokens(candidate["content"]) <= max_tokens:
                    results.append(candidate["content"])
                    tokens_used += self.count_tokens(candidate["content"])

            return "\n\n".join(results)

        return self.vector_store.search(query_embedding, max_tokens)
```

### 第四层：窗口优化策略

**滑动窗口管理**：

```python
class SlidingWindowManager:
    def __init__(self, window_size: int, step_size: int):
        self.window_size = window_size
        self.step_size = step_size
        self.position = 0

    def get_window(
        self,
        messages: List[Message],
        task: str = None
    ) -> List[Message]:
        if self.position >= len(messages):
            self.position = max(0, len(messages) - self.window_size)

        window_start = self.position
        window_end = min(window_start + self.window_size, len(messages))

        window = messages[window_start:window_end]

        if task and len(window) < self.window_size:
            self.position = max(0, window_start - self.step_size)

        return window

    def advance(self):
        self.position += self.step_size

    def reset(self):
        self.position = 0
```

**基于重要性的非均匀采样**：

```python
class ImportanceBasedSampler:
    def __init__(self, importance_scorer: ImportanceScorer):
        self.scorer = importance_scorer

    def sample(
        self,
        messages: List[Message],
        max_count: int
    ) -> List[Message]:
        if len(messages) <= max_count:
            return messages

        scored = []
        for i, msg in enumerate(messages):
            score = self.scorer.score(msg)
            scored.append((i, msg, score))

        scored.sort(key=lambda x: x[2], reverse=True)

        selected_indices = [idx for idx, _, _ in scored[:max_count]]
        selected_indices.sort()

        return [messages[idx] for idx in selected_indices]
```

## 工程实践

```python
class ContextWindowHandler:
    def __init__(self, config: WindowConfig):
        self.model = config.model
        self.max_tokens = self.model_limits[config.model]
        self.reserve_tokens = config.reserve_tokens

        self.filter = RelevanceBasedFilter(config.embedder)
        self.compressor = SemanticCompressor()
        self.memory = HierarchicalMemory()
        self.selector = LayeredInformationSelector()

    async def prepare(
        self,
        context: ContextData,
        task: Task
    ) -> PreparedContext:
        effective_limit = self.max_tokens - self.reserve_tokens

        if context.total_tokens <= effective_limit:
            return PreparedContext(
                content=context.messages,
                total_tokens=context.total_tokens,
                truncated=False
            )

        selections = self.selector.select(context, task, effective_limit)

        compressed_parts = []
        for part_name, part_content in selections.items():
            part_tokens = self.count_tokens(part_content)
            if part_tokens > self.get_budget(part_name, effective_limit):
                compressed = await self.compressor.compress(
                    part_content,
                    self.get_budget(part_name, effective_limit)
                )
                compressed_parts.append(compressed)
            else:
                compressed_parts.append(part_content)

        return PreparedContext(
            content=compressed_parts,
            total_tokens=self.count_tokens(compressed_parts),
            truncated=True,
            truncation_info=self._get_truncation_info(selections, compressed_parts)
        )
```

## 常见误区

- **暴力截断**：丢失关键信息导致任务失败
- **不加区分地压缩**：重要细节和噪音同样被压缩
- **忽视局部性**：最近的信息往往更重要
- **没有层次感**：把所有信息放在同一层管理
- **不考虑任务相关性**：对所有任务用同样的过滤策略
