# 如何处理用户不断修改目标？

## 面试问题

如何处理用户不断修改目标？

## 考察重点

- 是否理解目标漂移（Goal Drift）的问题
- 是否能设计有效的目标管理机制
- 是否了解何时该接受修改、何时该拒绝
- 是否有处理目标变更和历史记录的经验

## 核心结论

用户修改目标是正常的交互模式，核心处理策略是**目标版本控制 + 变更检测 + 增量执行**。不要每次修改都从头开始，而是尽量复用之前的执行结果。

---

## 目标变更的类型

### 类型一：参数更新

只修改参数，意图不变。

```
用户："帮我订一张去北京的机票"
  目标：book_flight(destination=北京)

用户："改一下，是去上海不是北京"
  目标：book_flight(destination=上海)
  变更类型：参数更新
```

### 类型二：意图微调

意图大类不变，但范围或约束变了。

```
用户："帮我查一下这个月的订单"
  目标：query_orders(time_range=这个月)

用户："不对，是查最近三个月的"
  目标：query_orders(time_range=最近三个月)
  变更类型：约束变更
```

### 类型三：完全重置

新目标和旧目标完全无关。

```
用户："帮我查一下订单"
用户："算了，帮我看看有什么优惠活动"
  变更类型：完全重置
```

## 目标管理机制

### 目标版本控制

```python
class GoalVersion:
    def __init__(self, goal_id: str, initial_goal: Goal):
        self.goal_id = goal_id
        self.version = 1
        self.current_goal = initial_goal
        self.history: list[GoalSnapshot] = [GoalSnapshot(goal=initial_goal, version=1)]

    def update(self, new_goal: Goal) -> GoalChange:
        change = self.detect_change(new_goal)

        if change.type == "param_update":
            # 参数更新，只增加版本
            self.version += 1
            self.current_goal = new_goal
            self.history.append(GoalSnapshot(goal=new_goal, version=self.version))

        elif change.type == "reset":
            # 完全重置，开始新目标
            self.goal_id = generate_new_goal_id()
            self.version = 1
            self.current_goal = new_goal
            self.history = [GoalSnapshot(goal=new_goal, version=1)]

        return change

    def detect_change(self, new_goal: Goal) -> GoalChange:
        # 检测变更类型
        if self.is_param_update(new_goal):
            return GoalChange(type="param_update", changes=self.get_changed_params(new_goal))
        elif self.is_constraint_change(new_goal):
            return GoalChange(type="constraint_change", changes=self.get_changed_constraints(new_goal))
        else:
            return GoalChange(type="reset")
```

### 变更检测

```python
def detect_goal_change(old_goal: Goal, new_goal: Goal) -> GoalChange:
    changes = {
        "params": {},
        "constraints": {},
        "intent": None
    }

    # 检查意图是否变化
    if old_goal.intent != new_goal.intent:
        changes["intent"] = (old_goal.intent, new_goal.intent)
        return GoalChange(type="reset", changes=changes)

    # 检查参数变化
    for key in set(old_goal.params.keys()) | set(new_goal.params.keys()):
        old_val = old_goal.params.get(key)
        new_val = new_goal.params.get(key)
        if old_val != new_val:
            changes["params"][key] = (old_val, new_val)

    # 检查约束变化
    for key in set(old_goal.constraints.keys()) | set(new_goal.constraints.keys()):
        old_val = old_goal.constraints.get(key)
        new_val = new_goal.constraints.get(key)
        if old_val != new_val:
            changes["constraints"][key] = (old_val, new_val)

    if changes["params"] or changes["constraints"]:
        return GoalChange(type="param_update", changes=changes)

    return GoalChange(type="no_change")
```

## 增量执行策略

### 策略一：复用已完成步骤

```python
class IncrementalExecutor:
    def __init__(self):
        self.completed_steps: list[Step] = []

    async def execute_incrementally(
        self,
        old_goal: Goal,
        new_goal: Goal,
        old_result: ExecutionResult
    ) -> ExecutionResult:
        # 分析哪些步骤可以复用
        reusable_steps = self.find_reusable_steps(old_goal, new_goal, old_result)

        # 从可复用步骤之后开始执行
        start_from_step = len(reusable_steps)

        new_steps = []
        for i in range(start_from_step, len(new_goal.steps)):
            step = new_goal.steps[i]
            # 注入之前的结果
            step.context = inject_prior_results(reusable_steps)
            result = await step.execute()
            new_steps.append(result)

        return ExecutionResult(
            steps=reusable_steps + new_steps,
            final_state=new_steps[-1] if new_steps else None
        )

    def find_reusable_steps(self, old_goal: Goal, new_goal: Goal, old_result: ExecutionResult) -> list[Step]:
        reusable = []

        for i, (old_step, result) in enumerate(zip(old_goal.steps, old_result.steps)):
            # 检查这一步是否仍然需要
            if self.is_step_still_needed(old_step, new_goal):
                # 检查依赖是否仍然满足
                if self.check_dependencies(old_step, old_result, new_goal):
                    reusable.append(old_step)

        return reusable
```

### 策略二：检查点恢复

```python
class CheckpointManager:
    def __init__(self):
        self.checkpoints: dict[str, Checkpoint] = {}

    def save_checkpoint(self, goal_id: str, state: ExecutionState):
        self.checkpoints[goal_id] = Checkpoint(
            goal_id=goal_id,
            state=state,
            timestamp=datetime.now(),
            completed_steps=state.completed_steps
        )

    async def resume_from_checkpoint(
        self,
        goal_id: str,
        new_goal: Goal
    ) -> ExecutionResult:
        checkpoint = self.checkpoints.get(goal_id)

        if not checkpoint:
            # 没有检查点，从头开始
            return await execute_goal(new_goal)

        # 检查点之后的步骤
        remaining_steps = new_goal.steps[len(checkpoint.completed_steps):]

        # 执行剩余步骤
        results = []
        for step in remaining_steps:
            result = await step.execute(context=checkpoint.state)
            results.append(result)

        return ExecutionResult(
            completed_steps=checkpoint.completed_steps,
            new_steps=results
        )
```

## 用户交互策略

### 确认变更

```python
def confirm_goal_change(change: GoalChange, old_goal: Goal, new_goal: Goal) -> str:
    if change.type == "param_update":
        params_str = ", ".join([
            f"{k}: {v[0]} → {v[1]}"
            for k, v in change.changes["params"].items()
        ])
        return f"""检测到以下变更：
        {params_str}

        是否继续？
        1. 是，应用变更
        2. 否，保持原样
        3. 开始新任务"""

    elif change.type == "reset":
        return """检测到新的任务。

        当前任务将被取消。
        是否继续？
        1. 是，开始新任务
        2. 否，继续当前任务"""
```

### 变更历史展示

```python
def show_goal_history(versions: list[GoalVersion]) -> str:
    lines = ["任务历史："]

    for i, version in enumerate(versions):
        lines.append(f"\n版本 {version.version}：")
        lines.append(f"  意图：{version.current_goal.intent}")
        lines.append(f"  参数：{version.current_goal.params}")

    return "\n".join(lines)
```

## Claude Code 中的目标处理

Claude Code 处理目标变更的方式：

```typescript
interface GoalState {
  currentGoal: string;
  originalGoal: string;
  modifications: GoalModification[];
  completedSteps: Step[];
  canResume: boolean;
}

async function handleGoalModification(
    newGoal: string,
    state: GoalState
): Promise<GoalState> {
    const change = analyzeGoalChange(state.currentGoal, newGoal);

    if (change.type === 'parameter_update') {
        // 参数更新：复用之前的步骤
        const reusableSteps = findReusableSteps(
            state.completedSteps,
            newGoal
        );

        return {
            ...state,
            currentGoal: newGoal,
            modifications: [...state.modifications, change],
            completedSteps: reusableSteps,
            canResume: true
        };
    } else if (change.type === 'reset') {
        // 完全重置：开始新目标
        return {
            currentGoal: newGoal,
            originalGoal: newGoal,
            modifications: [],
            completedSteps: [],
            canResume: false
        };
    }

    return state;
}
```

## 特殊情况处理

### 用户在执行过程中修改目标

```python
async def handle_modification_during_execution(
    current_execution: Execution,
    new_goal: Goal
) -> str:
    # 如果当前执行可以暂停
    if current_execution.can_pause:
        # 保存当前状态
        checkpoint = current_execution.save_checkpoint()

        # 询问用户
        return f"""当前任务进行到第 {current_execution.progress} 步。

        检测到目标变更。
        您希望：
        1. 暂停当前任务，应用新目标（新目标会在当前进度上继续）
        2. 取消当前任务，开始新任务
        3. 继续当前任务，忽略新目标"""

    else:
        # 无法暂停，必须选择
        return """当前任务无法暂停。
        请选择：
        1. 取消当前任务，开始新任务
        2. 继续当前任务（新的变更将被忽略）"""
```

## 面试追问方向

1. **"用户修改了3次目标，你怎么决定是继续执行还是重新开始？"**

   考察候选人对增量执行 vs 重新执行的理解。

2. **"用户修改目标时，之前的执行结果怎么保留？"**

   考察候选人对状态管理和检查点的理解。

3. **"如果用户修改目标导致之前的一些步骤变得无效，你怎么处理？"**

   考察候选人对依赖分析和回滚的理解。

4. **"你怎么防止用户频繁修改目标导致系统资源浪费？"**

   考察候选人对资源管理和用户体验的权衡。
