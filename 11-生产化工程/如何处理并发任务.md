# 如何处理并发任务？

## 面试问题

智能体系统需要同时处理大量用户请求，你会如何设计并发处理方案？

## 考察重点

- 是否理解智能体并发和传统 Web 并发处理的根本差异
- 是否能区分 IO 密集型（模型调用）和 CPU 密集型（检索）的并发策略
- 是否了解 Python/Node.js 的异步并发模型及其局限性
- 是否考虑过并发下的资源竞争、限流、队列积压

## 核心结论

智能体并发处理的核心是区分 IO 等待和计算负载：模型 API 调用是 IO 密集型，适合异步并发；但 Token 解码和向量检索是计算密集型，需要线程池或进程池隔离。

### 并发模型对比

```python
import asyncio
import threading
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor


class AgentConcurrencyModel:
    """智能体并发模型选型"""

    @staticmethod
    def io_bound_task(task_id: int):
        """模拟模型 API 调用 - IO 密集型"""
        import time
        time.sleep(2)
        return f"task_{task_id}_done"

    @staticmethod
    def cpu_bound_task(task_id: int):
        """模拟向量检索/推理 - CPU 密集型"""
        import time
        start = time.time()
        while time.time() - start < 2:
            _ = [i ** 2 for i in range(10000)]
        return f"task_{task_id}_done"


class ConcurrencyComparison:
    """不同并发模型的性能对比"""

    @staticmethod
    async def async_model(tasks: list):
        """异步 IO 模型 - 适合模型 API 调用"""
        async def call_api(task):
            await asyncio.sleep(2)
            return f"{task}_done"

        return await asyncio.gather(*[call_api(t) for t in tasks])

    @staticmethod
    def thread_pool_model(tasks: list, max_workers: int = 4):
        """线程池模型 - 适合混合负载"""
        with ThreadPoolExecutor(max_workers=max_workers) as executor:
            return list(executor.map(lambda t: f"{t}_done", tasks))

    @staticmethod
    def process_pool_model(tasks: list, max_workers: int = 2):
        """进程池模型 - 适合 CPU 密集"""
        with ProcessPoolExecutor(max_workers=max_workers) as executor:
            return list(executor.map(lambda t: f"{t}_done", tasks))
```

### 异步 Agent 实现

```python
import asyncio
import logging
from dataclasses import dataclass, field
from typing import Optional


@dataclass
class TaskStatus:
    task_id: str
    status: str  # pending / running / completed / failed
    result: Optional[str] = None
    error: Optional[str] = None


class AsyncAgent:
    def __init__(self, max_concurrent: int = 10):
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.active_tasks: dict[str, asyncio.Task] = {}
        self.task_status: dict[str, TaskStatus] = {}

    async def submit_task(self, request: UserRequest) -> str:
        task_id = self._generate_id()
        self.task_status[task_id] = TaskStatus(
            task_id=task_id,
            status="pending",
        )

        # 立即返回 task_id，异步执行
        task = asyncio.create_task(self._execute_task(task_id, request))
        self.active_tasks[task_id] = task
        task.add_done_callback(lambda _: self.active_tasks.pop(task_id, None))

        return task_id

    async def _execute_task(self, task_id: str, request: UserRequest):
        async with self.semaphore:
            self.task_status[task_id].status = "running"
            try:
                result = await self._agent_loop(request)
                self.task_status[task_id].status = "completed"
                self.task_status[task_id].result = result
            except Exception as e:
                self.task_status[task_id].status = "failed"
                self.task_status[task_id].error = str(e)
                logging.error(f"Task {task_id} failed: {e}")

    async def _agent_loop(self, request: UserRequest) -> str:
        max_steps = 10
        context = []

        for step in range(max_steps):
            thought = await self._call_llm(context, request)
            context.append(thought)

            if thought.get("type") == "final_answer":
                return thought["content"]

            if thought.get("type") == "tool_call":
                tool_result = await self._execute_tool(thought)
                context.append({"role": "tool", "content": tool_result})

        return "Max steps reached"

    async def _call_llm(self, context: list, request: UserRequest) -> dict:
        await asyncio.sleep(0.5)
        return {"type": "final_answer", "content": "mocked_result"}

    async def _execute_tool(self, thought: dict) -> str:
        await asyncio.sleep(0.3)
        return "tool_result"

    def get_status(self, task_id: str) -> Optional[TaskStatus]:
        return self.task_status.get(task_id)

    def _generate_id(self) -> str:
        import uuid
        return str(uuid.uuid4())
```

### 并发限流实现

```python
class RateLimiter:
    def __init__(self, rates: dict[str, int]):
        self.rates = rates  # { "model_api": 100, "database": 500 }
        self.tokens: dict[str, list[float]] = {}

    async def acquire(self, resource: str):
        if resource not in self.tokens:
            self.tokens[resource] = []

        now = time.time()
        self.tokens[resource] = [
            t for t in self.tokens[resource]
            if now - t < 60
        ]

        if len(self.tokens[resource]) >= self.rates.get(resource, 100):
            wait_time = 60 - (now - self.tokens[resource][0])
            if wait_time > 0:
                await asyncio.sleep(wait_time)

        self.tokens[resource].append(now)


class ConcurrentAgentPool:
    def __init__(
        self,
        max_workers: int = 5,
        rate_limiter: Optional[RateLimiter] = None,
    ):
        self.max_workers = max_workers
        self.rate_limiter = rate_limiter or RateLimiter({})
        self.queue: asyncio.Queue = asyncio.Queue()
        self.workers: list[asyncio.Task] = []
        self.results: dict[str, asyncio.Future] = {}

    async def start(self):
        self.workers = [
            asyncio.create_task(self._worker(i))
            for i in range(self.max_workers)
        ]

    async def stop(self):
        for w in self.workers:
            w.cancel()
        await asyncio.gather(*self.workers, return_exceptions=True)

    async def submit(self, request: UserRequest) -> asyncio.Future:
        future = asyncio.get_event_loop().create_future()
        task_id = str(uuid.uuid4())
        self.results[task_id] = future
        await self.queue.put((task_id, request))
        return future

    async def _worker(self, worker_id: int):
        while True:
            try:
                task_id, request = await self.queue.get()
                await self.rate_limiter.acquire("model_api")

                agent = AsyncAgent(max_concurrent=3)
                result_task_id = await agent.submit_task(request)
                result = await asyncio.wait_for(
                    self._poll_result(agent, result_task_id),
                    timeout=120,
                )

                if task_id in self.results:
                    self.results[task_id].set_result(result)

            except asyncio.CancelledError:
                break
            except Exception as e:
                if task_id in self.results:
                    self.results[task_id].set_exception(e)
            finally:
                self.queue.task_done()

    async def _poll_result(
        self,
        agent: AsyncAgent,
        task_id: str,
        interval: float = 0.5,
    ):
        while True:
            status = agent.get_status(task_id)
            if status.status in ("completed", "failed"):
                if status.status == "completed":
                    return status.result
                raise Exception(status.error)
            await asyncio.sleep(interval)
```

## 深度展开

### 第一层：智能体并发 vs 传统 Web 并发

| 维度 | 传统 Web 服务 | 智能体服务 |
|------|-------------|-----------|
| 请求处理时间 | 50-500ms | 5-60s（一次对话可能多轮模型调用） |
| 主要瓶颈 | DB / 网络 IO | 模型 API 延迟 + Token 消耗 |
| 状态需求 | 通常无状态 | 有状态（对话上下文、记忆） |
| 并发模型 | 多线程/协程 | 异步 + 队列 + 任务分离 |
| 每个请求资源消耗 | 稳定、可预测 | 差异巨大（取决于任务复杂度） |
| 限流策略 | QPS 限流 | QPS + Token 双维度限流 |

### 第二层：Python asyncio 在 Agent 中的最佳实践

```python
import asyncio
import time


class AsyncAgentExecutor:
    """asyncio 在智能体中的正确用法"""

    def __init__(self):
        self.session = None
        self._lock = asyncio.Lock()

    async def process_batch(self, requests: list[UserRequest]) -> list[dict]:
        """并发处理一批请求"""
        semaphore = asyncio.Semaphore(5)

        async def limited_process(req):
            async with semaphore:
                return await self._process_single(req)

        tasks = [limited_process(r) for r in requests]
        return await asyncio.gather(*tasks, return_exceptions=True)

    async def _process_single(self, request: UserRequest) -> dict:
        """单个请求处理流程"""
        async with self._lock:
            context = await self._build_context(request)

        result = await self._agent_loop(context)
        return result

    async def _build_context(self, request: UserRequest) -> Context:
        """构建上下文 - 包含多个并发子任务"""
        async def get_history():
            await asyncio.sleep(0.1)
            return ["history_1", "history_2"]

        async def retrieve_knowledge():
            await asyncio.sleep(0.3)
            return ["knowledge_1", "knowledge_2"]

        async def get_user_profile():
            await asyncio.sleep(0.2)
            return {"name": "user", "role": "admin"}

        history, knowledge, profile = await asyncio.gather(
            get_history(),
            retrieve_knowledge(),
            get_user_profile(),
        )

        return Context(
            history=history,
            knowledge=knowledge,
            user_profile=profile,
        )

    async def _agent_loop(self, context: Context) -> dict:
        """Agent 主循环 - 串行调用 LLM，因为每一步依赖上一步"""
        for step in range(5):
            response = await self._call_llm(context)
            if response["type"] == "final":
                return response

            if response["type"] == "tool":
                result = await self._call_tool(response["tool"])
                context.add_tool_result(result)

        return {"type": "final", "content": "max_steps"}

    async def _call_llm(self, context: Context) -> dict:
        """模型调用 - 使用 httpx 异步请求"""
        import httpx
        async with httpx.AsyncClient(timeout=30) as client:
            resp = await client.post(
                "https://api.llm.com/v1/chat",
                json={"messages": context.to_messages()},
            )
            return resp.json()

    async def _call_tool(self, tool_call: dict) -> str:
        """工具调用 - 支持异步工具"""
        tool_name = tool_call["name"]
        if tool_name == "search":
            return await self._async_search(tool_call["params"])
        elif tool_name == "execute_sql":
            return await self._async_execute_sql(tool_call["params"])
        return "unknown_tool"

    async def _async_search(self, params: dict) -> str:
        await asyncio.sleep(0.5)
        return "search_result"

    async def _async_execute_sql(self, params: dict) -> str:
        async with self._lock:
            result = await self._run_query(params["sql"])
            return str(result)

    async def _run_query(self, sql: str) -> list:
        import aiosqlite
        async with aiosqlite.connect(":memory:") as db:
            cursor = await db.execute(sql)
            return await cursor.fetchall()
```

```python
import multiprocessing
from concurrent.futures import ProcessPoolExecutor


class HybridConcurrencyEngine:
    """混合并发引擎 - 同时支持 IO 和 CPU 密集型任务"""

    def __init__(self, max_io_workers=20, max_cpu_workers=2):
        self.io_semaphore = asyncio.Semaphore(max_io_workers)
        self.cpu_executor = ProcessPoolExecutor(max_workers=max_cpu_workers)
        self.loop = asyncio.get_event_loop()

    async def process(self, request: UserRequest) -> dict:
        async with self.io_semaphore:
            intent = await self._classify_intent(request)

            if intent.requires_heavy_computation:
                vector_result = await self.loop.run_in_executor(
                    self.cpu_executor,
                    self._heavy_vector_search,
                    intent.query,
                )
            else:
                vector_result = await self._lightweight_retrieve(intent)

            return await self._generate_response(intent, vector_result)

    def _heavy_vector_search(self, query: str) -> list:
        """CPU 密集型向量检索 - 在子进程中执行"""
        import numpy as np
        query_vec = self._embed(query)
        scores = np.dot(self.index, query_vec)
        top_k = np.argsort(scores)[-5:][::-1]
        return [self.documents[i] for i in top_k]

    def _embed(self, text: str) -> np.ndarray:
        """耗时较长的 embedding 计算"""
        return np.random.rand(768)

    async def _classify_intent(self, request: UserRequest) -> Intent:
        await asyncio.sleep(0.1)
        return Intent(query=request.message, requires_heavy_computation=False)

    async def _lightweight_retrieve(self, intent: Intent) -> list:
        await asyncio.sleep(0.05)
        return ["light_result"]

    async def _generate_response(self, intent: Intent, context: list) -> dict:
        await asyncio.sleep(1)
        return {"response": "generated"}
```

## 工程实践

```python
class ProductionConcurrencyManager:
    """生产级并发管理器"""

    def __init__(self, config: dict):
        self.max_concurrent = config.get("max_concurrent_tasks", 20)
        self.max_concurrent_per_user = config.get("max_per_user", 3)
        self.task_timeout = config.get("task_timeout", 120)
        self.user_quotas: dict[str, int] = {}
        self.active_tasks: dict[str, asyncio.Task] = {}
        self.semaphore = asyncio.Semaphore(self.max_concurrent)
        self.rate_limiter = RateLimiter(config.get("rate_limits", {}))

    async def execute_with_concurrency_control(
        self,
        user_id: str,
        request: UserRequest,
    ) -> str:
        # 1. 用户级别并发控制
        if self.user_quotas.get(user_id, 0) >= self.max_concurrent_per_user:
            raise QuotaExceeded(f"User {user_id} has too many active tasks")

        # 2. 全局信号量控制
        async with self.semaphore:
            self.user_quotas[user_id] = self.user_quotas.get(user_id, 0) + 1
            try:
                # 3. 限流检查
                await self.rate_limiter.acquire("model_api")

                # 4. 超时控制
                task_id = str(uuid.uuid4())
                task = asyncio.create_task(
                    self._agent_execute(request)
                )
                self.active_tasks[task_id] = task

                try:
                    result = await asyncio.wait_for(
                        task,
                        timeout=self.task_timeout,
                    )
                    return result
                except asyncio.TimeoutError:
                    task.cancel()
                    raise TaskTimeout(f"Task {task_id} timed out")
                finally:
                    self.active_tasks.pop(task_id, None)

            finally:
                self.user_quotas[user_id] -= 1

    async def _agent_execute(self, request: UserRequest) -> str:
        await asyncio.sleep(2)
        return "completed"
```

## 常见误区

- **忽略 GIL 的影响**：Python 多线程无法利用多核，遇到 CPU 密集型检索时性能大幅下降，需要用进程池
- **事件循环阻塞**：在 async 代码中同步调用 requests.get() 或 time.sleep()，阻塞事件循环
- **无限创建协程**：不限制并发数直接 asyncio.gather，导致模型 API 被限流或 OOM
- **任务超时不处理**：不设置 await 超时，部分任务 hang 住后导致 worker 泄露
- **忽略协程安全**：多个协程同时修改同一个共享对象（如列表、字典），导致数据竞争
