# 如何处理中断、恢复和重试？

## 面试问题

智能体在执行任务过程中如果发生中断（进程崩溃、网络超时、模型 API 失败），你的系统如何恢复和重试？

## 考察重点

- 是否理解智能体中断恢复和传统数据库事务恢复的区别
- 是否能设计任务的检查点（checkpoint）和状态持久化机制
- 是否了解重试策略（指数退避、抖动、截止时间）
- 是否考虑过重试的幂等性和副作用控制

## 核心结论

智能体的中断恢复核心是检查点机制：在 ReAct 循环的每一步（思考前、工具调用前、工具返回后）保存完整快照，重启后从最近检查点恢复执行。

## 参考回答

### 中断恢复架构

```
用户请求进入
    |
    v
[检查点: 初始状态]
    |
    v
[LLM 思考] -----> 如果中断 -----> 从检查点恢复
    |
    v
[检查点: 思考完成]
    |
    v
[工具调用前检查] -----> 工具幂等检查
    |
    v
[工具调用] -----> 如果失败 -----> 重试 / 跳过 / 回退
    |
    v
[检查点: 工具完成]
    |
    v
[继续循环...]
    |
    v
[最终输出]
```

### 检查点持久化实现

```python
import json
import time
import uuid
from dataclasses import dataclass, field
from typing import Any, Optional


@dataclass
class Checkpoint:
    task_id: str
    session_id: str
    step_index: int
    phase: str  # before_think / after_think / before_tool / after_tool
    context: dict
    tool_calls: list[dict]
    tool_results: list[dict]
    metadata: dict = field(default_factory=dict)
    created_at: float = field(default_factory=time.time)

    def to_dict(self) -> dict:
        return {
            "task_id": self.task_id,
            "session_id": self.session_id,
            "step_index": self.step_index,
            "phase": self.phase,
            "context": self.context,
            "tool_calls": self.tool_calls,
            "tool_results": self.tool_results,
            "metadata": self.metadata,
            "created_at": self.created_at,
        }


class CheckpointManager:
    def __init__(self, store):
        self.store = store

    async def save(self, cp: Checkpoint):
        key = f"checkpoint:{cp.task_id}:{cp.step_index}:{cp.phase}"
        await self.store.set(key, json.dumps(cp.to_dict()))
        await self.store.expire(key, 86400 * 7)

        await self.store.set(
            f"checkpoint:latest:{cp.task_id}",
            json.dumps({
                "step_index": cp.step_index,
                "phase": cp.phase,
                "timestamp": time.time(),
            }),
        )

    async def load_latest(self, task_id: str) -> Optional[Checkpoint]:
        latest_key = f"checkpoint:latest:{task_id}"
        latest_data = await self.store.get(latest_key)
        if not latest_data:
            return None

        latest = json.loads(latest_data)
        cp_key = f"checkpoint:{task_id}:{latest['step_index']}:{latest['phase']}"
        cp_data = await self.store.get(cp_key)
        if not cp_data:
            return None

        return Checkpoint(**json.loads(cp_data))

    async def load_checkpoint(
        self,
        task_id: str,
        step_index: int,
        phase: str,
    ) -> Optional[Checkpoint]:
        key = f"checkpoint:{task_id}:{step_index}:{phase}"
        data = await self.store.get(key)
        if data:
            return Checkpoint(**json.loads(data))
        return None

    async def cleanup(self, task_id: str):
        keys = await self.store.keys(f"checkpoint:{task_id}:*")
        if keys:
            await self.store.delete(*keys)
```

### 可恢复的 Agent 实现

```python
class ResumableAgent:
    def __init__(
        self,
        checkpoint_manager: CheckpointManager,
        config: AgentConfig,
    ):
        self.cp_manager = checkpoint_manager
        self.config = config
        self.max_steps = config.max_steps

    async def execute(
        self,
        task: UserRequest,
        session_id: str,
    ) -> AgentResult:
        existing = await self.cp_manager.load_latest(task.id)

        if existing:
            return await self._resume(task, existing)

        return await self._fresh_execute(task, session_id)

    async def _fresh_execute(
        self,
        task: UserRequest,
        session_id: str,
    ) -> AgentResult:
        context = {"task": task.message, "history": [], "steps_completed": []}

        for step in range(self.max_steps):
            cp = Checkpoint(
                task_id=task.id,
                session_id=session_id,
                step_index=step,
                phase="before_think",
                context=context,
                tool_calls=[],
                tool_results=[],
            )
            await self.cp_manager.save(cp)

            thought = await self._call_llm(context)

            cp.phase = "after_think"
            cp.context["last_thought"] = thought
            await self.cp_manager.save(cp)

            if thought.get("type") == "final_answer":
                await self.cp_manager.cleanup(task.id)
                return AgentResult(
                    success=True,
                    answer=thought["content"],
                    steps=context["steps_completed"],
                )

            if thought.get("type") == "tool_call":
                tool_call = thought["tool_call"]
                tool_call["idempotency_key"] = str(uuid.uuid4())

                cp.phase = "before_tool"
                cp.tool_calls.append(tool_call)
                await self.cp_manager.save(cp)

                result = await self._execute_tool_with_retry(tool_call)

                cp.phase = "after_tool"
                cp.tool_results.append({"call": tool_call, "result": result})
                await self.cp_manager.save(cp)

                context["history"].append({
                    "role": "assistant",
                    "thought": thought,
                })
                context["history"].append({
                    "role": "tool",
                    "result": result,
                })
                context["steps_completed"].append(step)

        await self.cp_manager.cleanup(task.id)
        return AgentResult(success=False, answer="Max steps reached")

    async def _resume(
        self,
        task: UserRequest,
        checkpoint: Checkpoint,
    ) -> AgentResult:
        context = checkpoint.context
        step = checkpoint.step_index

        for remaining_step in range(step, self.max_steps):
            if checkpoint.phase == "before_think":
                pass
            elif checkpoint.phase == "after_think":
                pass
            elif checkpoint.phase == "before_tool":
                last_call = checkpoint.tool_calls[-1]
                result = await self._execute_tool_with_retry(last_call)
                context["history"].append({
                    "role": "tool",
                    "result": result,
                })
                continue
            elif checkpoint.phase == "after_tool":
                pass

            cp = Checkpoint(
                task_id=task.id,
                session_id=checkpoint.session_id,
                step_index=remaining_step,
                phase="before_think",
                context=context,
                tool_calls=[],
                tool_results=[],
            )
            await self.cp_manager.save(cp)

            thought = await self._call_llm(context)

            cp.phase = "after_think"
            cp.context["last_thought"] = thought
            await self.cp_manager.save(cp)

            if thought.get("type") == "final_answer":
                await self.cp_manager.cleanup(task.id)
                return AgentResult(
                    success=True,
                    answer=thought["content"],
                    steps=context["steps_completed"],
                )

            if thought.get("type") == "tool_call":
                tool_call = thought["tool_call"]
                tool_call["idempotency_key"] = str(uuid.uuid4())

                result = await self._execute_tool_with_retry(tool_call)

                context["history"].append({
                    "role": "assistant",
                    "thought": thought,
                })
                context["history"].append({
                    "role": "tool",
                    "result": result,
                })
                context["steps_completed"].append(remaining_step)

        return AgentResult(success=False, answer="Max steps reached")

    async def _call_llm(self, context: dict) -> dict:
        import httpx
        async with httpx.AsyncClient(timeout=30) as client:
            resp = await client.post(
                self.config.model_endpoint,
                json={"context": context},
            )
            return resp.json()

    async def _execute_tool_with_retry(self, tool_call: dict) -> Any:
        retryer = RetryWithBackoff(
            max_retries=self.config.tool_retries,
            base_delay=1.0,
            max_delay=30.0,
        )
        return await retryer.execute(
            self._execute_tool,
            tool_call,
        )

    async def _execute_tool(self, tool_call: dict) -> Any:
        idempotency_check = await self._check_idempotency(
            tool_call["idempotency_key"],
        )
        if idempotency_check:
            return idempotency_check

        import httpx
        async with httpx.AsyncClient(timeout=10) as client:
            resp = await client.post(
                f"{self.config.tool_endpoint}/{tool_call['name']}",
                json=tool_call["parameters"],
            )
            result = resp.json()

            await self._record_idempotency(
                tool_call["idempotency_key"],
                result,
            )
            return result

    async def _check_idempotency(self, key: str) -> Optional[Any]:
        result = await self.cp_manager.store.get(f"idempotency:{key}")
        if result:
            return json.loads(result)
        return None

    async def _record_idempotency(self, key: str, result: Any):
        await self.cp_manager.store.set(
            f"idempotency:{key}",
            json.dumps(result),
            ex=86400,
        )
```

### 重试策略实现

```python
import asyncio
import random
import time


class RetryWithBackoff:
    def __init__(
        self,
        max_retries: int = 3,
        base_delay: float = 1.0,
        max_delay: float = 60.0,
        jitter: bool = True,
    ):
        self.max_retries = max_retries
        self.base_delay = base_delay
        self.max_delay = max_delay
        self.jitter = jitter

    async def execute(
        self,
        func: callable,
        *args,
        **kwargs,
    ) -> Any:
        last_exception = None

        for attempt in range(self.max_retries + 1):
            try:
                return await func(*args, **kwargs)
            except NonRetryableError as e:
                raise e
            except RetryableError as e:
                last_exception = e
                if attempt < self.max_retries:
                    delay = self._calculate_delay(attempt)
                    await asyncio.sleep(delay)
            except Exception as e:
                last_exception = e
                if attempt < self.max_retries:
                    delay = self._calculate_delay(attempt)
                    await asyncio.sleep(delay)

        raise MaxRetriesExceeded(
            f"Failed after {self.max_retries} retries",
            last_exception,
        )

    def _calculate_delay(self, attempt: int) -> float:
        delay = min(
            self.base_delay * (2 ** attempt),
            self.max_delay,
        )
        if self.jitter:
            delay = delay * (0.5 + random.random() * 0.5)
        return delay


class RetryableError(Exception):
    """可重试的异常"""
    pass


class NonRetryableError(Exception):
    """不可重试的异常"""
    pass


class MaxRetriesExceeded(Exception):
    """超过最大重试次数"""
    pass


# 错误分类
class ModelRateLimitError(RetryableError):
    """模型限流 - 可重试"""
    pass


class ModelTimeoutError(RetryableError):
    """模型超时 - 可重试"""
    pass


class ToolTransientError(RetryableError):
    """工具临时错误 - 可重试"""
    pass


class ToolPermissionError(NonRetryableError):
    """工具权限错误 - 不可重试"""
    pass


class InvalidParameterError(NonRetryableError):
    """无效参数 - 不可重试"""
    pass
```

### 重试策略对比

| 策略 | 延迟计算 | 适用场景 | 优点 | 缺点 |
|------|---------|---------|------|------|
| 固定间隔 | delay = 常数 | 网络超时 | 简单 | 容易导致惊群效应 |
| 指数退避 | delay = base * 2^n | 限流、服务过载 | 避免雪崩 | 尾部延迟大 |
| 指数退避 + 抖动 | delay * (0.5~1.5) | 通用场景 | 减少碰撞 | 稍复杂 |
| 即时重试 | delay = 0 | 竞争条件 | 延迟最低 | 可能无效 |
| 截止时间 | 超过 deadline 不再重试 | 实时交互 | 保证 SLA | 可能提前放弃 |

```python
class FailureHandler:
    """智能体中的失败分类处理"""

    FAILURE_STRATEGIES = {
        "model_api_timeout": {
            "retryable": True,
            "max_retries": 3,
            "backoff": "exponential",
            "fallback": "use_alternative_model",
        },
        "model_api_rate_limit": {
            "retryable": True,
            "max_retries": 5,
            "backoff": "exponential_with_jitter",
            "fallback": "queue_and_wait",
        },
        "model_api_500": {
            "retryable": True,
            "max_retries": 2,
            "backoff": "immediate",
            "fallback": "switch_region",
        },
        "tool_timeout": {
            "retryable": True,
            "max_retries": 2,
            "backoff": "fixed",
            "fallback": "skip_tool",
        },
        "tool_invalid_params": {
            "retryable": False,
            "fallback": "inform_model",
        },
        "tool_permission_denied": {
            "retryable": False,
            "fallback": "notify_user",
        },
        "agent_loop_timeout": {
            "retryable": True,
            "max_retries": 1,
            "backoff": "none",
            "fallback": "return_partial",
        },
    }

    async def handle_failure(
        self,
        failure_type: str,
        context: dict,
    ) -> FailureResolution:
        strategy = self.FAILURE_STRATEGIES.get(failure_type)
        if not strategy:
            return FailureResolution(
                action="abort",
                message=f"Unknown failure type: {failure_type}",
            )

        if not strategy["retryable"]:
            return FailureResolution(
                action=strategy["fallback"],
                message=f"Non-retryable failure: {failure_type}",
            )

        retry_count = context.get("retry_count", 0)
        max_retries = strategy["max_retries"]

        if retry_count >= max_retries:
            return FailureResolution(
                action=strategy["fallback"],
                message=f"Max retries ({max_retries}) reached for {failure_type}",
            )

        return FailureResolution(
            action="retry",
            delay=self._calculate_delay(
                strategy["backoff"],
                retry_count,
            ),
        )
```

## 深度展开

### 第一层：中断恢复的三种粒度

| 粒度 | 恢复点 | 恢复时间 | 实现复杂度 | 适用场景 |
|------|-------|---------|-----------|---------|
| 任务级 | 从头执行 | 高（整个任务重做） | 低 | 简单任务（<5s） |
| 步骤级（检查点） | 从当前步骤恢复 | 中 | 中 | 标准 ReAct 循环 |
| 动作级 | 从工具调用后恢复 | 低 | 高 | 长流程、昂贵工具调用 |

### 第二层：幂等性设计

```python
class IdempotencyManager:
    def __init__(self, redis_client):
        self.redis = redis_client

    async def execute_idempotent(
        self,
        idempotency_key: str,
        func: callable,
        ttl: int = 86400,
    ) -> Any:
        result_key = f"idempotent:result:{idempotency_key}"
        lock_key = f"idempotent:lock:{idempotency_key}"

        existing = await self.redis.get(result_key)
        if existing:
            return json.loads(existing)

        lock = await self.redis.set(lock_key, "1", nx=True, ex=60)
        if not lock:
            raise ConcurrentExecutionError(
                "Another execution in progress for this key"
            )

        try:
            result = await func()

            await self.redis.set(
                result_key,
                json.dumps(result),
                ex=ttl,
            )
            return result
        finally:
            await self.redis.delete(lock_key)

    def generate_key(self, task_id: str, step: int, action: str) -> str:
        return f"{task_id}:{step}:{action}"
```

class PartialResultHandler:
    def handle_partial_success(
        self,
        completed_steps: list,
        failed_steps: list,
        context: dict,
    ) -> AgentResult:
        if len(completed_steps) == 0:
            return AgentResult(
                success=False,
                answer="Task failed at initial stage",
                partial_results=[],

        partial_output = self._synthesize_partial_output(
            completed_steps,
            context,
        )

        return AgentResult(
            success=True,
            answer=partial_output,
            partial_results=completed_steps,
            warnings=[s.error for s in failed_steps],
            message=(
                f"Completed {len(completed_steps)}/{"
                f"{len(completed_steps) + len(failed_steps)}} steps"
            ),
        )

## 工程实践

```python
class RecoveryOrchestrator:
    def __init__(self, config: dict):
        self.checkpoint_manager = CheckpointManager(config["store"])
        self.failure_handler = FailureHandler()
        self.idempotency = IdempotencyManager(config["redis"])

    async def safe_execute(
        self,
        task: AgentTask,
    ) -> AgentResult:
        attempt = 0
        max_attempts = 3

        while attempt < max_attempts:
            try:
                agent = ResumableAgent(
                    self.checkpoint_manager,
                    task.config,
                )
                return await agent.execute(task.request, task.session_id)
            except NonRetryableError as e:
                return AgentResult(
                    success=False,
                    answer=str(e),
                    error_type="non_retryable",
                )
            except (TimeoutError, ConnectionError) as e:
                attempt += 1
                if attempt >= max_attempts:
                    return AgentResult(
                        success=False,
                        answer=str(e),
                        error_type="max_attempts",
                    )
                await asyncio.sleep(2 ** attempt)
            except Exception as e:
                return AgentResult(
                    success=False,
                    answer=str(e),
                    error_type="unexpected",
                )
```

## 常见误区

- **重试不做幂等检查**：工具调用重试时同一 SQL 执行两次，或同一笔支付扣款两次
- **没有截止时间**：任务一直重试直到成功，但用户已经在 30 秒前放弃等待了
- **重试时不做抖动**：大量任务同时重试导致下游系统雪崩
- **进程重启后不恢复**：只做了内存中的状态管理，进程挂了就丢失所有进度
- **恢复后不验证上下文有效性**：检查点恢复后发现 session 已过期或工具参数已变化
- **忽略非重试性错误**：所有错误都重试，权限错误、参数错误重试也没用，浪费资源
