# 如何优化智能体成本和延迟？

## 面试问题

智能体系统上线后，模型 API 调用成本和响应延迟都很高，你会如何优化？

## 考察重点

- 是否理解 Token 消耗的构成和优化空间
- 是否能从系统架构层面优化延迟（缓存、预加载、模型路由）
- 是否了解不同模型的成本-质量-延迟权衡
- 是否能量化优化效果，用数据驱动决策

## 核心结论

智能体成本优化的本质是减少不必要的 Token 消耗，延迟优化的本质是减少串行等待。核心指标是"每次有效输出消耗的 Token 数"，优化目标是在质量不降低的前提下最大化这个效率。

## 参考回答

### 成本构成分析

```
单次智能体请求的成本构成：

+-------------------------------------------------+
|  系统提示词 (System Prompt)                     |
|  固定消耗: ~2000 tokens                         |
|  优化方向: 精简、复用、分层                      |
+-------------------------------------------------+
|  用户输入 (User Input)                          |
|  可变消耗: ~500 tokens                          |
|  优化方向: 输入压缩                             |
+-------------------------------------------------+
|  历史上下文 (History)                           |
|  递增消耗: 每轮 ~2000 tokens                    |
|  优化方向: 摘要压缩、选择性保留                  |
+-------------------------------------------------+
|  工具定义 (Tool Schema)                         |
|  固定消耗: ~1000 tokens (取决于工具数量)         |
|  优化方向: 懒加载、按需注入                     |
+-------------------------------------------------+
|  模型输出 (Model Output)                        |
|  可变消耗: ~500 tokens                          |
|  优化方向: 控制输出长度                         |
+-------------------------------------------------+

典型成本占比 (4次 ReAct 循环):
- 系统提示词: 20%
- 历史上下文: 45%
- 工具定义: 15%
- 模型输出: 10%
- 用户输入: 10%
```

### Token 消耗优化

```python
class TokenOptimizer:
    def __init__(self, config: TokenConfig):
        self.config = config

    async def optimize_context(self, context: AgentContext) -> OptimizedContext:
        optimized = await self._compress_history(context.history)
        optimized = await self._prune_tool_schemas(optimized, context.current_step)
        optimized = await self._summarize_long_term(optimized)
        return optimized

    async def _compress_history(
        self,
        history: list[Message],
        max_tokens: int = 8000,
    ) -> list[Message]:
        total = self._count_tokens(history)
        if total <= max_tokens:
            return history

        recent = history[-5:]
        older = history[:-5]

        summary = await self._summarize_messages(older, max_tokens // 2)
        compressed = [Message(role="summary", content=summary)] + recent

        while self._count_tokens(compressed) > max_tokens:
            compressed.pop(1)

        return compressed

    async def _prune_tool_schemas(
        self,
        context: OptimizedContext,
        step: int,
    ) -> OptimizedContext:
        """按需注入工具定义，不用的工具不占 Token"""
        all_tools = context.available_tools
        if not self.config.lazy_tool_loading:
            return context

        needed_tools = await self._predict_needed_tools(
            context.current_goal,
            step,
        )

        context.tool_schemas = [
            t for t in all_tools if t.name in needed_tools
        ]

        return context

    async def _predict_needed_tools(
        self,
        goal: str,
        step: int,
    ) -> set[str]:
        """根据当前目标和步骤预测需要的工具"""
        if "search" in goal.lower() or step == 0:
            return {"search", "read"}
        if "write" in goal.lower():
            return {"write", "read", "list"}
        return {"search", "read", "write"}

    async def _summarize_messages(
        self,
        messages: list[Message],
        max_tokens: int,
    ) -> str:
        """使用轻量模型做摘要，降低成本"""
        return await self.config.summary_model.summarize(
            messages,
            max_tokens=max_tokens,
        )

    def _count_tokens(self, messages: list[Message]) -> int:
        return sum(len(m.content) // 2 for m in messages)
```

### 延迟优化

```python
import asyncio
import time


class LatencyOptimizer:
    def __init__(self, config: LatencyConfig):
        self.config = config
        self.cache = TokenCache(config.cache_size)
        self.model_router = ModelRouter(config.models)
        self.connection_pool = ConnectionPool(config.pool_size)

    async def optimize_request(
        self,
        request: CompletionRequest,
    ) -> OptimizedRequest:
        cached = await self.cache.get(request)
        if cached:
            return OptimizedRequest(
                from_cache=True,
                result=cached,
                latency_ms=0,
            )

        model = self.model_router.select(request)
        optimized = OptimizedRequest(
            original=request,
            selected_model=model,
            early_stop_after=self._estimate_stop_condition(request),
        )

        return optimized

    async def execute_agent_loop(self, task: AgentTask) -> AgentResult:
        start = time.time()
        timeout = self.config.request_timeout

        prefetch_tasks = []

        for step in range(self.config.max_steps):
            step_start = time.time()

            llm_task = asyncio.create_task(
                self._call_llm_with_cache(task.context)
            )

            if step == 0:
                prefetch = asyncio.create_task(
                    self._prefetch_next_tools(task)
                )
                prefetch_tasks.append(prefetch)

            llm_result = await asyncio.wait_for(
                llm_task,
                timeout=max(1, timeout - (time.time() - start)),
            )

            if time.time() - start > timeout:
                return AgentResult(
                    success=False,
                    error="timeout",
                    partial=llm_result,
                )

        return AgentResult(success=True, result=llm_result)

    async def _call_llm_with_cache(
        self,
        context: AgentContext,
    ) -> LLMResult:
        cache_key = self.cache.make_key(context.messages)
        cached = await self.cache.get_by_key(cache_key)
        if cached:
            return cached

        async with self.connection_pool.get_connection() as conn:
            result = await conn.complete(
                messages=context.messages,
                model=context.selected_model,
                max_tokens=context.max_tokens,
            )

            await self.cache.set(cache_key, result)
            return result

    async def _prefetch_next_tools(self, task: AgentTask):
        """预加载下一步可能需要的数据"""
        if task.type == "search":
            await self._preload_search_index(task.query)
        elif task.type == "data_analysis":
            await self._preload_dataset(task.dataset_id)

    def _estimate_stop_condition(self, request: CompletionRequest) -> int:
        """根据请求复杂度估计需要多少步"""
        if request.requires_tools:
            return 8
        if request.is_simple_query:
            return 1
        return 4
```

### 模型路由策略

```python
class ModelRouter:
    def __init__(self, config: dict):
        self.models = {
            "claude-sonnet-4": ModelConfig(
                cost_per_1k_input=0.003,
                cost_per_1k_output=0.015,
                latency_p50=1.5,
                quality_score=9.5,
            ),
            "gpt-4o": ModelConfig(
                cost_per_1k_input=0.005,
                cost_per_1k_output=0.015,
                latency_p50=2.0,
                quality_score=9.0,
            ),
            "gpt-4o-mini": ModelConfig(
                cost_per_1k_input=0.00015,
                cost_per_1k_output=0.0006,
                latency_p50=0.8,
                quality_score=7.0,
            ),
            "claude-haiku": ModelConfig(
                cost_per_1k_input=0.00025,
                cost_per_1k_output=0.00125,
                latency_p50=0.6,
                quality_score=7.5,
            ),
        }

    def select(self, request: CompletionRequest) -> str:
        if request.requires_reasoning or request.requires_tools:
            return "claude-sonnet-4"

        if request.is_simple:
            return self._select_cheapest_fastest()

        return "gpt-4o-mini"

    def _select_cheapest_fastest(self) -> str:
        return min(
            self.models.items(),
            key=lambda x: (
                x[1].cost_per_1k_input + x[1].cost_per_1k_output,
                x[1].latency_p50,
            ),
        )[0]

    def estimate_cost(self, request: CompletionRequest) -> CostEstimate:
        model = self.select(request)
        config = self.models[model]

        input_tokens = request.estimated_input_tokens
        output_tokens = request.estimated_output_tokens

        return CostEstimate(
            model=model,
            input_cost=input_tokens / 1000 * config.cost_per_1k_input,
            output_cost=output_tokens / 1000 * config.cost_per_1k_output,
            total_cost=(
                input_tokens / 1000 * config.cost_per_1k_input +
                output_tokens / 1000 * config.cost_per_1k_output
            ),
            estimated_latency=config.latency_p50,
        )
```

### 成本-质量-延迟权衡矩阵

| 策略 | 成本降低 | 延迟影响 | 质量影响 | 适用场景 |
|------|---------|---------|---------|---------|
| 使用小模型 | 60-80% | -30% | -10-20% | 简单分类、摘要 |
| 上下文压缩 | 30-50% | +10% | -5-10% | 长对话 |
| 缓存重复请求 | 40-70% | -80% | 0% | 高频相同问题 |
| 批处理合并 | 20-40% | +100% | 0% | 后台定时任务 |
| 延迟注入工具 | 15-25% | 0% | 0% | 工具数量多 |
| 提前终止 | 20-40% | -40% | +5% | 简单问答 |
| 结果缓存 | 50-90% | -90% | 0% | 幂等工具调用 |

### 缓存实现

```python
class TokenCache:
    def __init__(self, max_size: int = 1000, ttl: int = 3600):
        self.cache = {}
        self.max_size = max_size
        self.ttl = ttl
        self.hits = 0
        self.misses = 0

    def make_key(self, messages: list[dict]) -> str:
        import hashlib
        serialized = json.dumps(messages, sort_keys=True)
        return hashlib.sha256(serialized.encode()).hexdigest()

    async def get(self, request: CompletionRequest) -> Optional[str]:
        key = self.make_key(request.messages)
        entry = self.cache.get(key)

        if entry and time.time() - entry["time"] < self.ttl:
            self.hits += 1
            return entry["result"]

        self.misses += 1
        return None

    async def set(self, key: str, result: str):
        if len(self.cache) >= self.max_size:
            oldest = min(self.cache.keys(), key=lambda k: self.cache[k]["time"])
            del self.cache[oldest]

        self.cache[key] = {
            "result": result,
            "time": time.time(),
        }

    def hit_rate(self) -> float:
        total = self.hits + self.misses
        return self.hits / total if total > 0 else 0
```

### 第一层：成本优化的系统性方法

```
成本优化金字塔：

         +--------+
         | 不调用 |  最高优先级：完全不调用模型
         +--------+
             |
         +--------+
         | 缓存   |  相同的请求不要调两次
         +--------+
             |
         +--------+
         | 复用   |  相同的上下文不要重复传
         +--------+
             |
         +--------+
         | 压缩   |  减少传入的 Token 数量
         +--------+
             |
         +--------+
         | 精简   |  减少输出的 Token 数量
         +--------+
             |
         +--------+
         | 换模型 |  用更便宜的模型
         +--------+
```

```python
class LatencyProfiler:
    def __init__(self):
        self.breakdowns: list[StepBreakdown] = []

    async def profile_agent_execution(
        self,
        task: AgentTask,
    ) -> ProfileResult:
        step_breakdowns = []

        for step in range(task.max_steps):
            step_start = time.time()
            breakdown = {"step": step}

            t0 = time.time()
            context = await task.build_context()
            breakdown["build_context_ms"] = (time.time() - t0) * 1000

            t0 = time.time()
            thought = await task.call_llm(context)
            breakdown["llm_call_ms"] = (time.time() - t0) * 1000

            if thought.get("type") == "tool_call":
                t0 = time.time()
                result = await task.execute_tool(thought["tool_call"])
                breakdown["tool_call_ms"] = (time.time() - t0) * 1000
                breakdown["tool_name"] = thought["tool_call"]["name"]

            breakdown["total_ms"] = (time.time() - step_start) * 1000
            step_breakdowns.append(breakdown)

        return ProfileResult(step_breakdowns)

    async def analyze(self, task: AgentTask) -> OptimizationSuggestions:
        profile = await self.profile_agent_execution(task)
        suggestions = []

        for step in profile.steps:
            if step["llm_call_ms"] > 5000:
                suggestions.append(
                    Suggestion(
                        type="model_too_slow",
                        step=step["step"],
                        detail=f"LLM call took {step['llm_call_ms']}ms",
                        recommendation="Consider using faster model or enable streaming",
                    )
                )

            if step.get("tool_call_ms", 0) > 3000:
                suggestions.append(
                    Suggestion(
                        type="tool_too_slow",
                        step=step["step"],
                        tool=step.get("tool_name"),
                        detail=f"Tool call took {step['tool_call_ms']}ms",
                        recommendation="Consider caching tool results or parallel execution",
                    )
                )

        return OptimizationSuggestions(suggestions)
```

### 第三层：量化优化效果

```python
class CostAnalytics:
    def __init__(self):
        self.metrics = {
            "daily_cost": 0,
            "total_tokens_in": 0,
            "total_tokens_out": 0,
            "total_requests": 0,
            "cache_hits": 0,
            "cache_misses": 0,
        }

    def track_request(self, request: CompletionRequest, response: CompletionResponse):
        self.metrics["total_requests"] += 1
        self.metrics["total_tokens_in"] += request.tokens
        self.metrics["total_tokens_out"] += response.tokens
        self.metrics["daily_cost"] += self._calculate_cost(
            request.model,
            request.tokens,
            response.tokens,
        )

    def report(self) -> CostReport:
        avg_tokens_per_request = (
            self.metrics["total_tokens_in"] + self.metrics["total_tokens_out"]
        ) / max(self.metrics["total_requests"], 1)

        cost_per_request = self.metrics["daily_cost"] / max(
            self.metrics["total_requests"], 1
        )
        cache_hit_ratio = self.metrics["cache_hits"] / max(
            self.metrics["cache_hits"] + self.metrics["cache_misses"], 1
        )

        return CostReport(
            total_daily_cost=self.metrics["daily_cost"],
            total_requests=self.metrics["total_requests"],
            avg_tokens_per_request=avg_tokens_per_request,
            cost_per_request=cost_per_request,
            cache_hit_ratio=cache_hit_ratio,
            optimization_suggestions=self._generate_suggestions(),
        )

    def _generate_suggestions(self) -> list[str]:
        suggestions = []
        if self._cost_per_request > 0.01:
            suggestions.append("Cost per request > 1 cent, consider using cheaper model")
        if self.metrics["cache_hit_ratio"] < 0.3:
            suggestions.append("Cache hit ratio < 30%, review caching strategy")
        return suggestions
```

## 工程实践

```python
class ProductionCostOptimizer:
    def __init__(self, config: dict):
        self.token_optimizer = TokenOptimizer(config["token"])
        self.latency_optimizer = LatencyOptimizer(config["latency"])
        self.model_router = ModelRouter(config["models"])
        self.cache = TokenCache(
            max_size=config.get("cache_size", 5000),
            ttl=config.get("cache_ttl", 3600),
        )
        self.analytics = CostAnalytics()

    async def process(
        self,
        request: UserRequest,
    ) -> AgentResponse:
        optimized_context = await self.token_optimizer.optimize_context(
            request.context,
        )

        model = self.model_router.select(
            CompletionRequest(
                task=request.task,
                complexity=optimized_context.complexity,
            )
        )

        start = time.time()
        result = await self.latency_optimizer.execute_agent_loop(
            optimized_context,
            model=model,
        )
        latency_ms = (time.time() - start) * 1000

        self.analytics.track_request(
            request=request,
            model=model,
            tokens_in=optimized_context.total_tokens,
            tokens_out=result.tokens,
            latency_ms=latency_ms,
        )

        return result
```

## 常见误区

- **只看单次调用成本，不看总成本**：用便宜模型但需要多轮交互，总成本可能高于用贵模型一次解决
- **缓存不设 TTL**：旧的模型输出被返回给新用户，导致信息过时
- **过度压缩上下文**：压缩太狠导致模型频繁误解意图，反而增加重试次数和总成本
- **对所有请求用统一模型**：应该根据任务复杂度动态路由，简单问题用小模型
- **忽略输出 Token 成本**：输出 Token 通常比输入 Token 贵 3-5 倍，控制输出长度效果显著
