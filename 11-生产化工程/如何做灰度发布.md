# 如何做灰度发布？

## 面试问题

智能体系统的新功能、新提示词或新模型如何灰度发布？如何控制风险？

## 考察重点

- 是否理解智能体灰度发布和传统灰度发布的区别（语义影响难以量化）
- 是否能设计多维度的灰度策略（用户维度、流量维度、功能维度）
- 是否了解灰度期间的指标监控和自动回滚机制
- 是否考虑过提示词灰度发布和模型灰度发布的特殊要求

## 核心结论

智能体灰度发布的本质是"在真实流量中用可量化的指标验证变更效果"。因为模型输出不确定，灰度不仅是风险控制手段，更是效果验证的必要环节。

## 参考回答

### 灰度发布架构

```
+------------------------------------------------------+
|                    用户请求                            |
+------------------------------------------------------+
                        |
                +-------v-------+
                |  灰度路由器    |
                |  (根据策略分流) |
                +-------+-------+
                        |
          +-------------+-------------+
          |                           |
    +-----v-----+             +-------v------+
    | 基线版本   |             |  灰度版本    |
    | (稳定版)   |             | (新版本)     |
    | - 旧提示词 |             | - 新提示词   |
    | - 旧模型   |             | - 新模型     |
    | - 旧配置   |             | - 新配置     |
    +-----+-----+             +-------+------+
          |                           |
          +-------------+-------------+
                        |
                +-------v-------+
                |  结果收集器    |
                |  对比指标      |
                |  - 成功率      |
                |  - 延迟        |
                |  - 用户满意度  |
                |  - 任务完成率  |
                +-------+-------+
                        |
          +-------------+-------------+
          |                           |
    +-----v-----+             +-------v------+
    | 指标达标   |             |  指标不达标  |
    | -> 全量发布 |             | -> 自动回滚  |
    +-----------+             +--------------+
```

### 灰度策略实现

```python
import hashlib
import json
import time
from dataclasses import dataclass, field
from enum import Enum
from typing import Any, Optional


class GrayStrategy(Enum):
    USER_ID_HASH = "user_id_hash"       # 按用户 ID 哈希百分比
    RANDOM_SAMPLE = "random_sample"      # 随机采样
    BETA_USER = "beta_user"              # 仅白名单用户
    GEO_REGION = "geo_region"            # 按地理区域
    TIERED = "tiered"                    # 分层灰度（内测->公测->全量）
    FEATURE_FLAG = "feature_flag"        # 功能开关


@dataclass
class GrayConfig:
    name: str
    strategy: GrayStrategy
    traffic_percent: int = 0             # 0-100
    beta_users: list[str] = field(default_factory=list)
    regions: list[str] = field(default_factory=list)
    start_time: float = 0
    duration_minutes: int = 30
    auto_rollback_thresholds: dict = field(default_factory=dict)
    metrics_to_watch: list[str] = field(default_factory=list)


class GrayRouter:
    def __init__(self, config_store):
        self.config_store = config_store
        self.active_grays: dict[str, GrayConfig] = {}

    async def start_gray(self, config: GrayConfig):
        config.start_time = time.time()
        self.active_grays[config.name] = config
        await self.config_store.save_gray_config(config)

    async def stop_gray(self, name: str):
        self.active_grays.pop(name, None)
        await self.config_store.delete_gray_config(name)

    def should_use_gray(
        self,
        gray_name: str,
        user_id: str,
        request: UserRequest,
    ) -> bool:
        config = self.active_grays.get(gray_name)
        if not config:
            return False

        if config.strategy == GrayStrategy.USER_ID_HASH:
            return self._user_hash_match(user_id, config.traffic_percent)

        elif config.strategy == GrayStrategy.RANDOM_SAMPLE:
            return self._random_match(request.request_id, config.traffic_percent)

        elif config.strategy == GrayStrategy.BETA_USER:
            return user_id in config.beta_users

        elif config.strategy == GrayStrategy.GEO_REGION:
            return request.region in config.regions

        elif config.strategy == GrayStrategy.TIERED:
            return self._tiered_match(user_id, config)

        elif config.strategy == GrayStrategy.FEATURE_FLAG:
            return self._feature_flag_check(gray_name, user_id)

        return False

    def _user_hash_match(self, user_id: str, percent: int) -> bool:
        hash_val = int(hashlib.md5(user_id.encode()).hexdigest()[:8], 16)
        return (hash_val % 100) < percent

    def _random_match(self, request_id: str, percent: int) -> bool:
        hash_val = int(hashlib.md5(request_id.encode()).hexdigest()[:8], 16)
        return (hash_val % 100) < percent

    def _tiered_match(self, user_id: str, config: GrayConfig) -> bool:
        if user_id in config.beta_users:
            return True
        if self._user_hash_match(user_id, config.traffic_percent):
            return True
        return False

    def _feature_flag_check(self, flag_name: str, user_id: str) -> bool:
        return self.config_store.get_feature_flag(flag_name, user_id)
```

### 灰度发布管理器

```python
class GrayReleaseManager:
    def __init__(
        self,
        router: GrayRouter,
        metrics_collector,
        alerting,
    ):
        self.router = router
        self.metrics = metrics_collector
        self.alerting = alerting
        self.release_history: list[ReleaseRecord] = []

    async def create_release(
        self,
        name: str,
        release_type: str,
        changes: dict,
        strategy: GrayStrategy = GrayStrategy.USER_ID_HASH,
        initial_percent: int = 5,
        auto_rollback: bool = True,
    ) -> GrayConfig:
        config = GrayConfig(
            name=name,
            strategy=strategy,
            traffic_percent=initial_percent,
            auto_rollback_thresholds={
                "error_rate": 0.02,
                "p95_latency_ms": 10000,
                "task_completion_rate": 0.90,
            },
            metrics_to_watch=[
                "error_rate",
                "p95_latency",
                "task_completion_rate",
                "user_satisfaction",
            ],
        )

        await self.router.start_gray(config)
        return config

    async def gradual_rollout(
        self,
        config: GrayConfig,
        target_percent: int = 100,
        step_size: int = 10,
        step_interval_minutes: int = 10,
    ):
        current = config.traffic_percent
        while current < target_percent:
            current = min(current + step_size, target_percent)

            config.traffic_percent = current
            await self.router.config_store.save_gray_config(config)

            await asyncio.sleep(step_interval_minutes * 60)

            metrics = await self.metrics.get_gray_metrics(config.name)
            if self._should_rollback(metrics, config):
                await self._auto_rollback(config, metrics)
                return False

        return True

    async def monitor_and_decide(
        self,
        config: GrayConfig,
    ):
        elapsed = time.time() - config.start_time
        if elapsed < config.duration_minutes * 60:
            return

        metrics = await self.metrics.get_gray_metrics(config.name)
        baseline = await self.metrics.get_baseline_metrics()

        if self._should_promote(metrics, baseline, config):
            await self._promote(config)
            return "promoted"
        elif self._should_rollback(metrics, config):
            await self._auto_rollback(config, metrics)
            return "rolled_back"
        else:
            await self._extend_gray(config)
            return "extended"

    def _should_promote(
        self,
        metrics: dict,
        baseline: dict,
        config: GrayConfig,
    ) -> bool:
        for metric, threshold in config.auto_rollback_thresholds.items():
            gray_val = metrics.get(metric, 0)
            base_val = baseline.get(metric, 0)

            if metric in ("error_rate", "p95_latency_ms"):
                if gray_val > base_val * (1 + threshold):
                    return False
            elif metric in ("task_completion_rate", "user_satisfaction"):
                if gray_val < base_val * (1 - threshold):
                    return False

        return True

    def _should_rollback(self, metrics: dict, config: GrayConfig) -> bool:
        for metric, threshold in config.auto_rollback_thresholds.items():
            value = metrics.get(metric, 0)
            if metric == "error_rate" and value > threshold:
                return True
            if metric == "p95_latency_ms" and value > threshold:
                return True
            if metric == "task_completion_rate" and value < threshold:
                return True
        return False

    async def _auto_rollback(self, config: GrayConfig, metrics: dict):
        await self.router.stop_gray(config.name)

        record = ReleaseRecord(
            name=config.name,
            action="rollback",
            reason=f"Metrics breached: {json.dumps(metrics)}",
            timestamp=time.time(),
        )
        self.release_history.append(record)

        await self.alerting.send_alert(
            level="critical",
            title=f"Auto-rollback: {config.name}",
            message=f"Rolled back due to metrics: {json.dumps(metrics)}",
        )

    async def _promote(self, config: GrayConfig):
        await self.router.config_store.promote_to_production(config.name)

        record = ReleaseRecord(
            name=config.name,
            action="promote",
            timestamp=time.time(),
        )
        self.release_history.append(record)
```

### 灰度发布类型对比

| 类型 | 变更内容 | 灰度周期 | 验证方式 | 回滚难度 |
|------|---------|---------|---------|---------|
| 模型灰度 | 切换 LLM 模型 | 1-7 天 | Eval + 线上指标 | 低（换回旧模型） |
| 提示词灰度 | 修改 System Prompt | 1-3 天 | Eval + 用户反馈 | 中（需要恢复旧版本） |
| 功能灰度 | 新工具/新能力 | 3-14 天 | 功能完成率 | 中（需要关闭功能） |
| 配置灰度 | 修改参数 | 小时级 | 错误率/延迟 | 低（回滚配置） |
| 架构灰度 | 后端重构 | 1-4 周 | 全量指标 | 高（需要双跑） |

### 提示词灰度发布

```python
class PromptGrayRelease:
    """提示词灰度发布的专用实现"""

    def __init__(self, prompt_manager, eval_pipeline):
        self.prompt_manager = prompt_manager
        self.eval = eval_pipeline

    async def release_prompt(
        self,
        prompt_name: str,
        new_content: str,
        traffic_percent: int = 5,
    ):
        version = await self.prompt_manager.create_version(
            prompt_name=prompt_name,
            content=new_content,
            created_by="gray_release",
        )

        gray_config = GrayConfig(
            name=f"prompt:{prompt_name}:v{version.version_number}",
            strategy=GrayStrategy.USER_ID_HASH,
            traffic_percent=traffic_percent,
            auto_rollback_thresholds={
                "error_rate": 0.03,
                "task_completion_rate": 0.85,
                "tool_call_success_rate": 0.90,
                "avg_tokens_per_task": 5000,
            },
        )

        return version, gray_config

    async def validate_prompt_change(self, new_content: str) -> bool:
        eval_results = await self.eval.run_full_suite(
            prompts={"test": new_content}
        )
        return eval_results.overall_score >= 0.8
```

### 模型灰度发布

```python
class ModelGrayRelease:
    """模型切换灰度发布"""

    def __init__(self, model_router):
        self.model_router = model_router

    async def start_model_gray(
        self,
        new_model: str,
        percent: int = 1,
    ):
        gray_config = GrayConfig(
            name=f"model:{new_model}",
            strategy=GrayStrategy.USER_ID_HASH,
            traffic_percent=percent,
            auto_rollback_thresholds={
                "error_rate": 0.05,
                "p95_latency_ms": 15000,
                "task_completion_rate": 0.85,
                "output_format_valid_rate": 0.95,
            },
        )

        self.model_router.set_gray_model(new_model, gray_config)

    async def compare_model_outputs(
        self,
        old_model: str,
        new_model: str,
        sample_size: int = 100,
    ) -> ModelComparison:
        results = []
        for _ in range(sample_size):
            request = self._generate_test_request()

            old_result = await self.model_router.call(old_model, request)
            new_result = await self.model_router.call(new_model, request)

            comparison = await self._compare(
                old_result, new_result, request
            )
            results.append(comparison)

        avg_score = sum(r.similarity for r in results) / len(results)
        return ModelComparison(
            old_model=old_model,
            new_model=new_model,
            avg_similarity=avg_score,
            sample_results=results[:10],
        )
```

### 灰度监控指标

```python
class GrayMetricsCollector:
    def __init__(self, storage):
        self.storage = storage

    async def record_gray_request(
        self,
        gray_name: str,
        is_gray: bool,
        metrics: RequestMetrics,
    ):
        key = f"gray_metrics:{gray_name}:{time.strftime('%Y%m%d%H%M')}"
        entry = {
            "is_gray": is_gray,
            "timestamp": time.time(),
            "success": metrics.success,
            "latency_ms": metrics.latency_ms,
            "tokens_used": metrics.tokens_used,
            "task_completed": metrics.task_completed,
            "error_type": metrics.error_type,
        }
        await self.storage.append(key, entry)

    async def get_gray_metrics(
        self,
        gray_name: str,
        window_minutes: int = 5,
    ) -> dict:
        cutoff = time.time() - window_minutes * 60
        all_entries = await self._get_entries_since(gray_name, cutoff)

        gray_entries = [e for e in all_entries if e["is_gray"]]
        baseline_entries = [e for e in all_entries if not e["is_gray"]]

        return {
            "error_rate": self._calc_error_rate(gray_entries),
            "p95_latency": self._calc_p95(
                [e["latency_ms"] for e in gray_entries]
            ),
            "avg_tokens": sum(
                e["tokens_used"] for e in gray_entries
            ) / max(len(gray_entries), 1),
            "task_completion_rate": sum(
                1 for e in gray_entries if e["task_completed"]
            ) / max(len(gray_entries), 1),
            "sample_size": len(gray_entries),
            "baseline_error_rate": self._calc_error_rate(baseline_entries),
            "baseline_p95_latency": self._calc_p95(
                [e["latency_ms"] for e in baseline_entries]
            ),
        }
```

## 深度展开

### 第一层：灰度发布策略比较

| 策略 | 实施难度 | 统计显著性 | 影响范围控制 | 适用场景 |
|------|---------|-----------|------------|---------|
| User ID Hash | 低 | 好 | 精确 | 通用，推荐方案 |
| 随机采样 | 低 | 好 | 一般 | AB 测试 |
| Beta 用户 | 低 | 差（有偏） | 最小 | 早期验证 |
| 地理区域 | 中 | 中 | 中 | 区域合规 |
| 功能开关 | 低 | N/A | 精确 | 功能级控制 |
| 时间窗口 | 低 | N/A | 全局 | 定时发布 |

### 第二层：灰度决策矩阵

```
灰度指标健康度评估:

指标变化方向:
  绿色: 比基线好 (> +5%)
  黄色: 和基线持平 (within +/-5%)
  红色: 比基线差 (< -5%)

| 指标              | 绿色    | 黄色    | 红色    |
|-------------------|---------|---------|---------|
| 错误率            | < -20%  | +/-20%  | > +20%  |
| P95 延迟          | < -10%  | +/-10%  | > +10%  |
| 任务完成率        | > +5%   | +/-5%   | < -5%   |
| 用户满意度        | > +10%  | +/-10%  | < -10%  |
| Token 消耗        | < -10%  | +/-10%  | > +10%  |

决策:
- 全部绿色: 立即全量
- 有黄色: 继续灰度，增加样本量
- 有红色: 自动回滚
```

## 工程实践

```python
class GrayReleaseSystem:
    def __init__(self, config: dict):
        self.router = GrayRouter(config["store"])
        self.manager = GrayReleaseManager(
            router=self.router,
            metrics_collector=GrayMetricsCollector(config["storage"]),
            alerting=Alerting(config["alert"]),
        )
        self.prompt_gray = PromptGrayRelease(
            config["prompt_manager"],
            config["eval_pipeline"],
        )
        self.model_gray = ModelGrayRelease(config["model_router"])

    async def process_request(
        self,
        request: UserRequest,
    ) -> AgentResponse:
        for gray_name, gray_config in self.router.active_grays.items():
            if self.router.should_use_gray(gray_name, request.user_id, request):
                request.use_gray_version(gray_name)

        start = time.time()
        response = await self._execute_agent(request)
        latency = (time.time() - start) * 1000

        for gray_name in self.router.active_grays:
            metrics = RequestMetrics(
                success=response.success,
                latency_ms=latency,
                tokens_used=response.tokens_used,
                task_completed=response.task_completed,
                error_type=response.error_type if not response.success else None,
            )
            await self.metrics_collector.record_gray_request(
                gray_name,
                request.is_gray,
                metrics,
            )

        return response
```

## 常见误区

- **灰度就是让少量用户体验新功能**：灰度是严谨的统计实验，需要足够的样本量和时间才能得出结论
- **灰度指标只关注错误率**：智能体行为质量比错误率更重要，需要关注任务完成率、用户满意度
- **灰度时间太短**：5 分钟的灰度看不出效果，尤其对智能体这种多轮交互的系统
- **忽略基线漂移**：灰度期间基线也在变化（模型更新、促销活动），需要排除干扰
- **多次灰度交叉影响**：同时在灰度模型和灰度提示词，指标互相污染，无法归因
- **没有回滚预案**：灰度发现问题时才发现旧版本已经下线，无法回滚
