# 如何设计智能体服务架构？

## 面试问题

如果让你设计一个智能体服务架构，你会怎么设计？需要考虑哪些模块？

## 考察重点

- 是否理解智能体架构和传统微服务架构的核心差异
- 是否能清晰划分架构层次和各模块职责
- 是否关注架构的可扩展性、可观测性和可维护性
- 是否能说出架构设计中的关键决策和取舍

## 核心结论

智能体服务架构的核心是编排层与执行层的分离：编排层负责决策（怎么做），执行层负责资源（做什么），通过事件驱动的方式解耦各模块。

### 核心模块设计

#### 1. 意图识别模块

```python
class IntentRouter:
    def __init__(self):
        self.agents = {
            "customer_service": CustomerServiceAgent(),
            "code_assistant": CodeAssistantAgent(),
            "data_analyst": DataAnalysisAgent(),
            "general": GeneralAgent(),
        }
        self.classifier = self._load_classifier()

    async def route(self, request: UserRequest) -> AgentRoute:
        intent = await self.classifier.predict(request.message)
        confidence = intent.confidence

        if confidence < 0.4:
            return AgentRoute(
                agent="general",
                confidence=confidence,
                requires_clarification=True,
                suggested_clarifications=intent.alternatives,
            )

        agent = self.agents.get(intent.name, self.agents["general"])

        model_requirement = self._estimate_model_need(request, intent)
        if model_requirement == "high":
            return AgentRoute(
                agent=agent,
                model="claude-sonnet-4",
                confidence=confidence,
            )

        return AgentRoute(
            agent=agent,
            model="gpt-4o-mini",
            confidence=confidence,
        )
```

#### 2. 任务规划引擎

```python
class PlanningEngine:
    def __init__(self, config: PlanningConfig):
        self.max_iterations = config.max_iterations
        self.max_tool_calls = config.max_tool_calls
        self.strategy = config.strategy

    async def plan(self, task: Task, context: Context) -> Plan:
        if self.strategy == "reactive":
            return await self._reactive_plan(task, context)
        elif self.strategy == "structured":
            return await self._structured_plan(task, context)
        elif self.strategy == "hybrid":
            return await self._hybrid_plan(task, context)

    async def _reactive_plan(self, task: Task, context: Context) -> Plan:
        steps = []
        completed = []
        iteration = 0

        while not self._is_complete(task, completed) and iteration < self.max_iterations:
            next_action = await self._decide_next(task, context, completed)
            steps.append(next_action)

            if self._requires_tool(next_action):
                result = await self._execute_tool(next_action.tool_call)
                completed.append(ToolResult(next_action, result))
            else:
                completed.append(ReasoningStep(next_action))

            context = await self._update_context(context, completed[-1])
            iteration += 1

        return Plan(steps=steps, completed=completed, iterations=iteration)

    async def _structured_plan(self, task: Task, context: Context) -> Plan:
        initial_plan = await self._generate_initial_plan(task)

        validated = []
        for step in initial_plan.steps:
            valid, issues = self._validate_step(step)
            if not valid:
                step = await self._repair_step(step, issues)
            validated.append(step)

        return Plan(steps=validated, strategy="structured")

    async def _hybrid_plan(self, task: Task, context: Context) -> Plan:
        initial = await self._structured_plan(task, context)
        initial.strategy = "hybrid"

        if not self._needs_replanning(initial):
            return initial

        return await self._reactive_plan(task, context)
```

#### 3. 上下文管理器

```python
class ContextManager:
    def __init__(self, config: ContextConfig):
        self.short_term = ShortTermMemory(
            max_tokens=config.short_term_max_tokens
        )
        self.long_term = LongTermMemory(
            store=config.memory_store
        )
        self.retriever = Retriever(
            vector_store=config.vector_store,
            embedder=config.embedder,
        )

    async def build(self, request: UserRequest, session: Session) -> Context:
        system_prompt = await self._get_system_prompt(session.agent_type)
        user_intent = self._parse_intent(request.message)
        history = await self.short_term.get_recent(session.id, limit=20)
        long_term = await self.long_term.query(
            session.user_id,
            top_k=5,
        )
        retrieved = await self.retriever.search(
            query=request.message,
            top_k=3,
            filter={"session_id": session.id},
        )

        context = Context(
            system=system_prompt,
            user=user_intent,
            history=history,
            long_term=long_term,
            retrieved=retrieved,
        )

        return await self._compress_if_needed(context)
```

#### 4. 工具调用执行器

```python
class ToolExecutor:
    def __init__(self, config: ExecutorConfig):
        self.tools = self._register_tools()
        self.sandbox = Sandbox(config.sandbox)
        self.rate_limiter = RateLimiter(config.rate_limits)
        self.audit_logger = AuditLogger()

    async def execute(
        self,
        call: ToolCall,
        context: ExecutionContext,
    ) -> ToolResult:
        tool = self.tools.get(call.tool_name)
        if not tool:
            return ToolResult(
                success=False,
                error=f"Unknown tool: {call.tool_name}",
                error_type="NOT_FOUND",
            )

        permissions = await self._check_permissions(
            call.tool_name,
            context.user_role,
        )
        if not permissions.allowed:
            return ToolResult(
                success=False,
                error=f"Permission denied: {permissions.reason}",
                error_type="PERMISSION_DENIED",
            )

        validated_params = await self._validate_params(
            tool.schema,
            call.parameters,
        )

        await self.rate_limiter.check(call.tool_name)

        execution_plan = self._plan_execution(tool, validated_params)
        if execution_plan.requires_sandbox:
            result = await self.sandbox.run(
                tool.execute,
                validated_params,
                timeout=tool.timeout,
            )
        else:
            result = await tool.execute(**validated_params)

        await self.audit_logger.log(
            ToolCallLog(
                tool_name=call.tool_name,
                params=validated_params,
                result=result,
                user=context.user_id,
                timestamp=datetime.now(),
            )
        )

        return self._format_result(result, tool.output_schema)
```

## 深度展开

### 第一层：架构设计原则

| 原则 | 说明 | 违反后果 |
|------|------|---------|
| 编排与执行分离 | 决策逻辑和资源访问解耦 | 修改工具影响任务逻辑，难以单独测试 |
| 有状态服务无状态化 | session 状态外置到 Redis/DB | 扩容后请求路由到不同实例导致状态丢失 |
| 限流熔断隔离 | 每个依赖需要有独立保护 | 模型 API 超时拖垮整个系统 |
| 可观测优先 | 所有模块输出结构化日志和指标 | 线上问题定位困难 |
| 防御性设计 | 不信任模型输出、不信任工具结果 | 参数错误导致数据损坏 |

### 第二层：关键设计决策

**同步 vs 异步**：

```python
class ExecutionMode(Enum):
    SYNC = "sync"       # 用户等待结果，适用于简单查询
    ASYNC = "async"     # 任务提交后轮询，适用于复杂任务
    STREAM = "stream"   # SSE 流式推送，适用于逐步展示


class AgentOrchestrator:
    def execute(self, request: UserRequest, mode: ExecutionMode):
        if mode == ExecutionMode.SYNC:
            return self._execute_sync(request)
        elif mode == ExecutionMode.ASYNC:
            task_id = self._submit_task(request)
            return {"task_id": task_id, "status": "pending"}
        elif mode == ExecutionMode.STREAM:
            return self._execute_stream(request)
```

**单体 vs 微服务**：

| 维度 | 单体 | 微服务 |
|------|------|--------|
| 开发效率 | 高，本地即可调试 | 低，需要启动多个服务 |
| 部署复杂度 | 低，单容器部署 | 高，需要服务编排 |
| 弹性扩缩 | 全部一起扩缩 | 可按模块独立扩缩 |
| 故障隔离 | 一个模块崩溃整个系统挂 | 单个模块故障不影响其他 |
| 延迟 | 无网络开销 | 服务间调用有网络开销 |
| 适用阶段 | MVP / 小团队 | 团队 > 10 人 / 高并发 |

### 第三层：可扩展架构设计

```python
class PluginArchitecture:
    def __init__(self):
        self.plugins = {}
        self.hooks = {
            "pre_process": [],
            "post_process": [],
            "pre_tool_call": [],
            "post_tool_call": [],
            "on_error": [],
        }

    def register_plugin(self, name: str, plugin: AgentPlugin):
        self.plugins[name] = plugin
        for hook_point in plugin.hooks:
            self.hooks[hook_point].append(plugin)

    async def run_hooks(self, hook_point: str, context: dict):
        for plugin in self.hooks.get(hook_point, []):
            try:
                context = await plugin.execute(context)
            except Exception as e:
                logger.error(f"Plugin {plugin.name} failed: {e}")
        return context
```

## 工程实践

```python
class AgentServiceArchitecture:
    def __init__(self, config: ServiceConfig):
        self.gateway = Gateway(config.gateway)
        self.orchestrator = Orchestrator(config.orchestrator)
        self.executor = Executor(config.executor)
        self.data = DataLayer(config.data)
        self.observability = Observability(config.observability)

    async def start(self):
        await self.data.initialize()
        await self.orchestrator.start()
        await self.gateway.start()

        self.observability.register_health_checks([
            ("database", self.data.health_check),
            ("model_api", self.executor.model_health_check),
            ("orchestrator", self.orchestrator.health_check),
        ])

    async def process_request(self, request: UserRequest) -> AgentResponse:
        start_time = time.time()

        with self.observability.trace("full_request"):
            route = await self.orchestrator.intent_router.route(request)

            context = await self.orchestrator.context_manager.build(
                request,
                session=await self.data.get_session(request.session_id),
            )

            plan = await self.orchestrator.planning_engine.plan(
                route.task,
                context,
            )

            execution_result = await self.executor.execute_plan(plan)

            response = self._build_response(execution_result, context)

        self.observability.record_request(
            latency=time.time() - start_time,
            route=route,
            plan=plan,
            tokens_used=execution_result.tokens_used,
            success=execution_result.success,
        )

        return response
```

## 常见误区

- **过度设计**：一开始就上几十个微服务，结果开发效率远低于单体架构。应从单体开始，按需拆分
- **编排层与执行层耦合**：在执行层里写业务逻辑，导致改一个工具需要重新部署整个系统
- **忽略状态管理**：认为智能体可以像 REST API 一样完全无状态，结果会话恢复时丢失上下文
- **不做限流设计**：模型 API 的并发限制比传统 API 严格得多，不做限流会导致大量请求失败
- **观测体系缺失**：只在关键路径打日志，出了问题只能逐行看日志，无法快速定位
