# 如何做缓存？

## 面试问题

智能体系统中，哪些数据适合缓存？缓存策略如何设计？

## 考察重点

- 是否理解智能体系统中的缓存层次（模型输出、工具结果、检索结果、知识库）
- 是否知道语义缓存（semantic cache）和精确缓存的区别和适用场景
- 是否了解缓存的失效策略（TTL、LRU、写穿透、主动失效）
- 是否考虑过缓存对模型输出质量和新鲜度的影响

## 核心结论

智能体缓存的核心挑战是"语义等价"：传统缓存靠 key 精确匹配，智能体缓存需要判断两个语义相似的请求是否应该返回同一结果。

### 精确缓存实现

```python
import hashlib
import json
import time
from dataclasses import dataclass
from typing import Any, Optional


@dataclass
class CacheEntry:
    key: str
    value: Any
    created_at: float
    expires_at: float
    hit_count: int = 0
    size_bytes: int = 0


class ExactCache:
    """精确匹配缓存 - 适用于工具调用结果、固定查询"""

    def __init__(self, redis_client, default_ttl: int = 3600):
        self.redis = redis_client
        self.default_ttl = default_ttl
        self.local_cache: dict[str, CacheEntry] = {}
        self.local_max_size = 1000
        self.hits = 0
        self.misses = 0

    def make_key(self, *parts: str) -> str:
        raw = ":".join(str(p) for p in parts)
        return hashlib.sha256(raw.encode()).hexdigest()

    async def get(self, key: str) -> Optional[Any]:
        entry = self.local_cache.get(key)
        if entry and time.time() < entry.expires_at:
            entry.hit_count += 1
            self.hits += 1
            return entry.value

        redis_value = await self.redis.get(f"cache:{key}")
        if redis_value:
            value = json.loads(redis_value)
            self._update_local(key, value)
            self.hits += 1
            return value

        self.misses += 1
        return None

    async def set(
        self,
        key: str,
        value: Any,
        ttl: Optional[int] = None,
    ):
        ttl = ttl or self.default_ttl
        expires_at = time.time() + ttl

        entry = CacheEntry(
            key=key,
            value=value,
            created_at=time.time(),
            expires_at=expires_at,
        )

        self._update_local(key, entry)

        serialized = json.dumps(value)
        await self.redis.setex(
            f"cache:{key}",
            ttl,
            serialized,
        )

    async def invalidate(self, pattern: str):
        keys = await self.redis.keys(f"cache:{pattern}")
        if keys:
            await self.redis.delete(*keys)

        self.local_cache = {
            k: v for k, v in self.local_cache.items()
            if not k.startswith(pattern)
        }

    def _update_local(self, key: str, entry: CacheEntry):
        if len(self.local_cache) >= self.local_max_size:
            oldest = min(
                self.local_cache.keys(),
                key=lambda k: self.local_cache[k].hit_count,
            )
            del self.local_cache[oldest]

        self.local_cache[key] = entry

    def hit_rate(self) -> float:
        total = self.hits + self.misses
        return self.hits / total if total > 0 else 0
```

### 语义缓存实现

```python
import numpy as np


class SemanticCache:
    """语义缓存 - 适用于 LLM 输出缓存"""

    def __init__(
        self,
        embedder,
        similarity_threshold: float = 0.92,
        max_size: int = 5000,
        ttl: int = 3600,
    ):
        self.embedder = embedder
        self.similarity_threshold = similarity_threshold
        self.max_size = max_size
        self.ttl = ttl
        self.entries: list[SemanticEntry] = []

    async def get(self, query: str) -> Optional[str]:
        query_embedding = await self.embedder.embed(query)

        best_match = None
        best_score = 0

        for entry in self.entries:
            if time.time() > entry.expires_at:
                continue

            if entry.query_text == query:
                entry.hit_count += 1
                return entry.response

            similarity = self._cosine_similarity(
                query_embedding,
                entry.embedding,
            )

            if similarity > best_score and similarity >= self.similarity_threshold:
                best_score = similarity
                best_match = entry

        if best_match:
            best_match.hit_count += 1
            return best_match.response

        return None

    async def set(self, query: str, response: str):
        if len(self.entries) >= self.max_size:
            least_used = min(
                self.entries,
                key=lambda e: e.hit_count,
            )
            self.entries.remove(least_used)

        embedding = await self.embedder.embed(query)
        entry = SemanticEntry(
            query_text=query,
            response=response,
            embedding=embedding,
            created_at=time.time(),
            expires_at=time.time() + self.ttl,
        )
        self.entries.append(entry)

    def _cosine_similarity(
        self,
        a: np.ndarray,
        b: np.ndarray,
    ) -> float:
        dot = np.dot(a, b)
        norm_a = np.linalg.norm(a)
        norm_b = np.linalg.norm(b)
        return float(dot / (norm_a * norm_b + 1e-10))


@dataclass
class SemanticEntry:
    query_text: str
    response: str
    embedding: np.ndarray
    created_at: float
    expires_at: float
    hit_count: int = 0


class SemanticCacheMetrics:
    def __init__(self):
        self.total_queries = 0
        self.cache_hits = 0
        self.semantic_hits = 0
        self.recall_sum = 0.0

    def record_hit(self, is_semantic: bool):
        self.total_queries += 1
        self.cache_hits += 1
        if is_semantic:
            self.semantic_hits += 1

    def record_miss(self):
        self.total_queries += 1

    def hit_rate(self) -> float:
        return self.cache_hits / max(self.total_queries, 1)

    def semantic_hit_ratio(self) -> float:
        return self.semantic_hits / max(self.total_queries, 1)
```

### 智能体缓存策略

```python
class AgentCache:
    """智能体系统完整缓存策略"""

    def __init__(self, config: CacheConfig):
        self.exact_cache = ExactCache(config.redis, config.exact_ttl)
        self.semantic_cache = SemanticCache(
            embedder=config.embedder,
            similarity_threshold=config.semantic_threshold,
        )
        self.tool_result_cache = ExactCache(
            config.redis,
            default_ttl=config.tool_cache_ttl,
        )
        self.embedding_cache = ExactCache(
            config.redis,
            default_ttl=86400,
        )
        self.knowledge_cache = ExactCache(
            config.redis,
            default_ttl=config.knowledge_ttl,
        )

    async def get_llm_response(
        self,
        messages: list[dict],
        model: str,
    ) -> Optional[str]:
        query = self._messages_to_string(messages)

        result = await self.semantic_cache.get(query)
        if result:
            return result

        exact_key = self.exact_cache.make_key(
            model,
            hashlib.sha256(query.encode()).hexdigest(),
        )
        return await self.exact_cache.get(exact_key)

    async def set_llm_response(
        self,
        messages: list[dict],
        model: str,
        response: str,
    ):
        query = self._messages_to_string(messages)
        await self.semantic_cache.set(query, response)

        exact_key = self.exact_cache.make_key(
            model,
            hashlib.sha256(query.encode()).hexdigest(),
        )
        await self.exact_cache.set(exact_key, response, ttl=60)

    async def invalidate_llm_cache(self, pattern: str = ""):
        if pattern:
            await self.exact_cache.invalidate(pattern)
            self.semantic_cache.entries = [
                e for e in self.semantic_cache.entries
                if pattern not in e.query_text
            ]
        else:
            self.semantic_cache.entries = []

    async def get_tool_result(
        self,
        tool_name: str,
        parameters: dict,
    ) -> Optional[Any]:
        key = self.exact_cache.make_key(
            "tool",
            tool_name,
            json.dumps(parameters, sort_keys=True),
        )
        return await self.tool_result_cache.get(key)

    async def set_tool_result(
        self,
        tool_name: str,
        parameters: dict,
        result: Any,
        ttl: Optional[int] = None,
    ):
        key = self.exact_cache.make_key(
            "tool",
            tool_name,
            json.dumps(parameters, sort_keys=True),
        )
        await self.tool_result_cache.set(key, result, ttl)

    async def invalidate_tool_cache(self, tool_name: str):
        await self.tool_result_cache.invalidate(f"tool:{tool_name}")

    def _messages_to_string(self, messages: list[dict]) -> str:
        return "|||".join(
            f"{m['role']}: {m['content']}" for m in messages
        )

    def cache_stats(self) -> dict:
        return {
            "llm_cache_hit_rate": self.semantic_cache.hit_rate(),
            "tool_cache_hit_rate": self.tool_result_cache.hit_rate(),
            "embedding_cache_hit_rate": self.embedding_cache.hit_rate(),
            "knowledge_cache_hit_rate": self.knowledge_cache.hit_rate(),
        }
```

### 缓存失效策略

| 缓存类型 | 失效策略 | TTL | 主动失效时机 | 说明 |
|---------|---------|-----|------------|------|
| LLM 输出 | TTL + 语义相似度 | 1h | 提示词更新时 | 语义缓存比精确缓存更好 |
| 工具调用结果 | TTL + 显式失效 | 5m-1h | 数据变更时 | 读多写少的工具收益大 |
| Embedding | TTL | 24h | 无 | 变化极慢 |
| 知识库/文档 | 事件驱动失效 | 7d | 文档更新时 | CRUD 操作的 each 场景 |
| 用户 Session | LRU + TTL | 30m | 用户登出 | 内存友好 |

## 深度展开

### 第一层：缓存一致性

```python
class CacheConsistencyManager:
    def __init__(self, cache: AgentCache, db):
        self.cache = cache
        self.db = db

    async def on_data_update(
        self,
        entity_type: str,
        entity_id: str,
    ):
        invalidation_patterns = {
            "knowledge_doc": f"knowledge:{entity_id}",
            "user_profile": f"user:{entity_id}:*",
            "tool_data": f"tool:query:{entity_id}:*",
        }

        pattern = invalidation_patterns.get(entity_type)
        if pattern:
            await self.cache.invalidate_llm_cache(pattern)
            await self.cache.invalidate_tool_cache(pattern)

    async def execute_with_cache_and_invalidate(
        self,
        tool_name: str,
        params: dict,
        write_func: callable,
    ):
        result = await write_func(params)
        await self.cache.invalidate_tool_cache(tool_name)
        return result
```

### 第二层：缓存粒度控制

| 粒度 | 缓存内容 | 空间占用 | 命中率 | 适用场景 |
|-----|---------|---------|-------|---------|
| 粗粒度 | 完整 agent 输出 | 高 | 低 | 重复问题多 |
| 中粒度 | 单步 LLM 响应 | 中 | 中 | 标准 ReAct |
| 细粒度 | 工具调用结果 | 低 | 高 | 幂等工具 |
| 混合 | 按步骤混合 | 中 | 高 | 推荐方案 |

## 工程实践

```python
class ProductionCacheSystem:
    def __init__(self, config: dict):
        self.cache = AgentCache(CacheConfig(config))
        self.consistency = CacheConsistencyManager(
            self.cache,
            config["database"],
        )

    async def process_with_cache(self, request: UserRequest) -> AgentResponse:
        cached_response = await self.cache.get_llm_response(
            request.messages,
            request.model,
        )
        if cached_response:
            return AgentResponse(
                content=cached_response,
                from_cache=True,
                latency_ms=0,
            )

        start = time.time()
        result = await self._execute_agent(request)
        latency = (time.time() - start) * 1000

        if latency > 2000:
            await self.cache.set_llm_response(
                request.messages,
                request.model,
                result.content,
            )

        return AgentResponse(
            content=result.content,
            from_cache=False,
            latency_ms=latency,
        )
```

## 常见误区

- **忽略语义相似度阈值**：阈值设太高（如 0.99）几乎无命中，设太低（如 0.7）返回错误结果
- **缓存不设全局上限**：语义缓存向量存储在内存中，不限制大小会导致 OOM
- **工具结果无限缓存**：数据库查询结果缓存太久，用户看到过时数据
- **不考虑缓存的冷启动**：新系统上线时缓存为空，需要考虑预热策略
- **缓存穿透**：恶意构造不同的请求绕过缓存直接打到模型 API，需要布隆过滤器或空值缓存
