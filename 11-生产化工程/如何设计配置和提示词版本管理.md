# 如何设计配置和提示词版本管理？

## 面试问题

智能体系统有大量配置项和提示词（prompt），如何管理它们的版本、变更和发布？

## 考察重点

- 是否理解配置和提示词管理的核心挑战（可追溯、可回滚、可灰度）
- 是否能区分代码级配置和运行时配置的管理策略
- 是否了解提示词版本管理的特殊需求（语义版本、效果评测、AB 测试）
- 是否设计过配置中心和服务发现的工程方案

## 核心结论

配置和提示词版本管理的本质是"把非代码变更纳入工程化管理"：提示词改一行就相当于改代码逻辑，必须有版本控制、评审流程、灰度发布和回滚能力。

### 提示词版本管理

```python
import json
import time
import hashlib
from dataclasses import dataclass, field
from typing import Any, Optional
from enum import Enum


class PromptStatus(Enum):
    DRAFT = "draft"
    REVIEWING = "reviewing"
    STAGING = "staging"
    PRODUCTION = "production"
    ROLLED_BACK = "rolled_back"
    DEPRECATED = "deprecated"


class PromptVariableType(Enum):
    STRING = "string"
    INTEGER = "integer"
    BOOLEAN = "boolean"
    LIST = "list"
    DICT = "dict"


@dataclass
class PromptVariable:
    name: str
    type: PromptVariableType
    description: str
    default_value: Any = None
    required: bool = True


@dataclass
class PromptVersion:
    version_id: str
    prompt_name: str
    version_number: int
    content: str
    variables: list[PromptVariable] = field(default_factory=list)
    model: str = ""
    temperature: float = 0.7
    max_tokens: int = 2000
    status: PromptStatus = PromptStatus.DRAFT
    created_by: str = ""
    created_at: float = field(default_factory=time.time)
    commit_message: str = ""
    parent_version: Optional[str] = None
    eval_score: Optional[float] = None
    tags: list[str] = field(default_factory=list)

    def render(self, variables: dict[str, Any]) -> str:
        rendered = self.content
        for var in self.variables:
            value = variables.get(
                var.name,
                var.default_value,
            )
            if value is None and var.required:
                raise ValueError(
                    f"Required variable '{var.name}' not provided"
                )
            rendered = rendered.replace(
                f"{{{{{var.name}}}}}",
                str(value) if value is not None else "",
            )
        return rendered

    def hash(self) -> str:
        return hashlib.sha256(
            json.dumps({
                "content": self.content,
                "variables": [(v.name, v.type.value) for v in self.variables],
                "model": self.model,
                "temperature": self.temperature,
            }, sort_keys=True).encode()
        ).hexdigest()[:12]


class PromptManager:
    def __init__(self, store):
        self.store = store
        self.active_versions: dict[str, PromptVersion] = {}
        self._load_active_versions()

    async def create_version(
        self,
        prompt_name: str,
        content: str,
        variables: list[PromptVariable],
        model: str = "",
        temperature: float = 0.7,
        max_tokens: int = 2000,
        created_by: str = "",
        commit_message: str = "",
    ) -> PromptVersion:
        latest = await self.get_latest_version(prompt_name)
        version_number = (latest.version_number + 1) if latest else 1

        version = PromptVersion(
            version_id=f"{prompt_name}:v{version_number}",
            prompt_name=prompt_name,
            version_number=version_number,
            content=content,
            variables=variables,
            model=model,
            temperature=temperature,
            max_tokens=max_tokens,
            status=PromptStatus.DRAFT,
            created_by=created_by,
            commit_message=commit_message,
            parent_version=latest.version_id if latest else None,
        )

        await self.store.save_version(version)
        return version

    async def promote(
        self,
        version_id: str,
        target_status: PromptStatus,
    ):
        version = await self.store.get_version(version_id)
        if not version:
            raise ValueError(f"Version {version_id} not found")

        transitions = {
            PromptStatus.DRAFT: [PromptStatus.REVIEWING],
            PromptStatus.REVIEWING: [PromptStatus.STAGING, PromptStatus.DRAFT],
            PromptStatus.STAGING: [PromptStatus.PRODUCTION, PromptStatus.DRAFT],
            PromptStatus.PRODUCTION: [PromptStatus.DEPRECATED],
        }

        allowed = transitions.get(version.status, [])
        if target_status not in allowed:
            raise ValueError(
                f"Cannot transition from {version.status.value} "
                f"to {target_status.value}"
            )

        version.status = target_status
        await self.store.save_version(version)

        if target_status == PromptStatus.PRODUCTION:
            await self._activate_version(version)

    async def rollback(
        self,
        prompt_name: str,
        target_version_number: int,
    ):
        current = await self.get_active_version(prompt_name)
        target = await self.get_version_by_number(
            prompt_name,
            target_version_number,
        )

        if not target:
            raise ValueError(f"Version {target_version_number} not found")

        current.status = PromptStatus.ROLLED_BACK
        await self.store.save_version(current)

        target.status = PromptStatus.PRODUCTION
        await self.store.save_version(target)
        await self._activate_version(target)

    async def get_active_version(
        self,
        prompt_name: str,
    ) -> Optional[PromptVersion]:
        return self.active_versions.get(prompt_name)

    async def get_latest_version(
        self,
        prompt_name: str,
    ) -> Optional[PromptVersion]:
        return await self.store.get_latest_version(prompt_name)

    async def get_version_by_number(
        self,
        prompt_name: str,
        version_number: int,
    ) -> Optional[PromptVersion]:
        return await self.store.get_version_by_number(
            prompt_name,
            version_number,
        )

    async def list_versions(
        self,
        prompt_name: str,
    ) -> list[PromptVersion]:
        return await self.store.list_versions(prompt_name)

    async def diff(
        self,
        version_id_a: str,
        version_id_b: str,
    ) -> str:
        v_a = await self.store.get_version(version_id_a)
        v_b = await self.store.get_version(version_id_b)

        if not v_a or not v_b:
            raise ValueError("Version not found")

        return self._compute_diff(v_a.content, v_b.content)

    def _compute_diff(self, a: str, b: str) -> str:
        import difflib
        diff = difflib.unified_diff(
            a.splitlines(keepends=True),
            b.splitlines(keepends=True),
            lineterm="",
        )
        return "".join(diff)

    async def _activate_version(self, version: PromptVersion):
        self.active_versions[version.prompt_name] = version
        await self.store.save_active_version(version)

    def _load_active_versions(self):
        versions = self.store.get_all_active_versions()
        self.active_versions = {v.prompt_name: v for v in versions}


class PromptRenderer:
    def __init__(self, prompt_manager: PromptManager):
        self.prompt_manager = prompt_manager

    async def render_prompt(
        self,
        prompt_name: str,
        variables: dict[str, Any],
        use_staging: bool = False,
    ) -> str:
        if use_staging:
            version = await self._get_staging_version(prompt_name)
        else:
            version = await self.prompt_manager.get_active_version(prompt_name)

        if not version:
            raise ValueError(f"No active version for prompt '{prompt_name}'")

        return version.render(variables)

    async def _get_staging_version(
        self,
        prompt_name: str,
    ) -> Optional[PromptVersion]:
        versions = await self.prompt_manager.list_versions(prompt_name)
        for v in versions:
            if v.status == PromptStatus.STAGING:
                return v
        return None
```

### 配置文件版本管理

```python
import yaml


class ConfigVersionManager:
    def __init__(self, config_path: str, git_repo=None):
        self.config_path = config_path
        self.git = git_repo
        self.configs: dict[str, ConfigEntry] = {}

    async def load_config(self, name: str) -> dict:
        if name in self.configs:
            return self.configs[name].data

        filepath = f"{self.config_path}/{name}.yaml"
        with open(filepath) as f:
            data = yaml.safe_load(f)

        self.configs[name] = ConfigEntry(
            name=name,
            data=data,
            loaded_at=time.time(),
        )
        return data

    async def update_config(
        self,
        name: str,
        new_data: dict,
        commit_message: str,
        author: str,
    ):
        current = await self.load_config(name)

        diff = self._compute_config_diff(current, new_data)

        approval = await self._request_approval(
            name=name,
            changes=diff,
            proposed_by=author,
        )
        if not approval.approved:
            raise ConfigChangeRejected(approval.reason)

        filepath = f"{self.config_path}/{name}.yaml"
        with open(filepath, "w") as f:
            yaml.dump(new_data, f, default_flow_style=False)

        if self.git:
            self.git.add(filepath)
            self.git.commit(f"{commit_message}\n\nAuthor: {author}")
            self.git.push()

        self.configs[name] = ConfigEntry(
            name=name,
            data=new_data,
            loaded_at=time.time(),
        )
```

### 配置热更新

```python
class HotReloadConfig:
    def __init__(self, redis_client, config_manager):
        self.redis = redis_client
        self.config_manager = config_manager
        self.watch_keys = set()

    async def watch_config(self, config_key: str):
        self.watch_keys.add(config_key)
        pubsub = self.redis.pubsub()
        await pubsub.subscribe(f"config:change:{config_key}")

        async def listener():
            async for message in pubsub.listen():
                if message["type"] == "message":
                    await self._reload_config(config_key)

        asyncio.create_task(listener())

    async def update_and_notify(self, config_key: str, new_value: Any):
        await self.redis.set(f"config:{config_key}", json.dumps(new_value))
        await self.redis.publish(
            f"config:change:{config_key}",
            json.dumps({"updated_at": time.time()}),
        )

    async def _reload_config(self, config_key: str):
        value = await self.redis.get(f"config:{config_key}")
        if value:
            self.config_manager.configs[config_key] = ConfigEntry(
                name=config_key,
                data=json.loads(value),
                loaded_at=time.time(),
            )
```

## 深度展开

### 第一层：提示词版本管理的特殊性

| 维度 | 代码版本管理 | 提示词版本管理 |
|------|------------|--------------|
| 变更验证 | 单元测试 + CI | 需要语义评测（Eval） |
| 回滚风险 | 低（语法错误可快速发现） | 高（可能影响所有用户） |
| 灰度方式 | 金丝雀发布 | 按用户/按比例灰度 |
| 效果衡量 | 明确的性能指标 | 需要人工评分或自动评估 |
| 分支策略 | Git Flow / Trunk | 通常线性版本 |
| 冲突解决 | Merge 工具 | 需要人工评审 |

### 第二层：提示词灰度发布

```python
class PromptCanaryRelease:
    def __init__(self, prompt_manager: PromptManager):
        self.prompt_manager = prompt_manager
        self.canary_configs: dict[str, CanaryConfig] = {}

    async def start_canary(
        self,
        prompt_name: str,
        canary_version: PromptVersion,
        traffic_percent: int = 10,
        duration_minutes: int = 30,
        eval_threshold: float = 0.85,
    ):
        config = CanaryConfig(
            prompt_name=prompt_name,
            canary_version=canary_version,
            traffic_percent=traffic_percent,
            duration_minutes=duration_minutes,
            eval_threshold=eval_threshold,
            start_time=time.time(),
        )
        self.canary_configs[prompt_name] = config

    def should_use_canary(self, prompt_name: str, user_id: str) -> bool:
        config = self.canary_configs.get(prompt_name)
        if not config:
            return False

        if time.time() - config.start_time > config.duration_minutes * 60:
            return False

        user_hash = hashlib.md5(user_id.encode()).hexdigest()
        user_percent = int(user_hash[:8], 16) % 100

        return user_percent < config.traffic_percent

    async def evaluate_and_promote(
        self,
        prompt_name: str,
    ) -> Optional[CanaryResult]:
        config = self.canary_configs.get(prompt_name)
        if not config:
            return None

        metrics = await self._collect_canary_metrics(prompt_name)
        control_metrics = await self._collect_control_metrics(prompt_name)

        improvements = {}
        for metric in ["success_rate", "latency", "user_satisfaction"]:
            imp = metrics.get(metric, 0) - control_metrics.get(metric, 0)
            improvements[metric] = imp

        avg_improvement = sum(improvements.values()) / len(improvements)

        if avg_improvement >= config.eval_threshold:
            await self.prompt_manager.promote(
                config.canary_version.version_id,
                PromptStatus.PRODUCTION,
            )
            return CanaryResult(
                success=True,
                improvements=improvements,
                action="promoted",
            )
        else:
            return CanaryResult(
                success=False,
                improvements=improvements,
                action="rolled_back",
            )
```

## 工程实践

```python
class ProductionConfigSystem:
    def __init__(self, config: dict):
        self.prompt_manager = PromptManager(config["store"])
        self.config_manager = ConfigVersionManager(config["config_path"])
        self.hot_reload = HotReloadConfig(config["redis"], self.config_manager)
        self.canary = PromptCanaryRelease(self.prompt_manager)

    async def get_prompt(
        self,
        prompt_name: str,
        variables: dict,
        user_id: str = "",
    ) -> str:
        if self.canary.should_use_canary(prompt_name, user_id):
            version = await self.prompt_manager.get_active_version(prompt_name)
            canary_config = self.canary.canary_configs[prompt_name]
            return canary_config.canary_version.render(variables)
        else:
            version = await self.prompt_manager.get_active_version(prompt_name)
            return version.render(variables)

    async def get_config(self, config_name: str) -> dict:
        return await self.config_manager.load_config(config_name)

    async def update_prompt(
        self,
        prompt_name: str,
        content: str,
        variables: list,
        author: str,
    ):
        version = await self.prompt_manager.create_version(
            prompt_name=prompt_name,
            content=content,
            variables=variables,
            created_by=author,
            commit_message=f"Update {prompt_name}",
        )
        return version
```

## 常见误区

- **提示词直接写死在代码里**：改提示词要重新部署，无法热更新，也无法回滚
- **没有提示词版本号**：线上出了问题不知道当前用的是哪个版本的提示词
- **配置和代码用同一个 Git 仓库**：配置变更需要走完整代码发布流程，影响效率
- **不做灰度直接全量**：提示词改一个字可能导致所有智能体行为异常
- **忽略环境差异**：开发环境、测试环境、生产环境的提示词混用，导致不可复现的问题
- **没有提示词评审流程**：任何开发都能改提示词，改完直接上线，没有 Review
