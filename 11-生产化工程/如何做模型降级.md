# 如何做模型降级？

## 面试问题

智能体依赖的 LLM API 出现故障或超时，你的系统如何做模型降级？

## 考察重点

- 是否理解降级不是简单换模型，而是系统性应对模型不可用的策略
- 是否能设计多级降级方案（模型级、功能级、体验级）
- 是否了解熔断、限流、优雅降级的工程实现
- 是否考虑过降级对用户体验的影响和恢复机制

## 核心结论

模型降级的本质是在服务质量和服务可用性之间做取舍。好的降级策略让系统在模型不可用时"优雅地变差"，而不是"直接崩溃"。

## 参考回答

### 降级架构总览

```
正常状态:
+--------+     +--------+     +--------+
| 请求   | --> | 主模型  | --> | 完整   |
|        |     | Sonnet  |     | 能力   |
+--------+     +--------+     +--------+

一级降级 (主模型超时/限流):
+--------+     +--------+     +--------+
| 请求   | --> | 备模型  | --> | 完整   |
|        |     | GPT-4o  |     | 能力   |
+--------+     +--------+     +--------+

二级降级 (所有高配模型不可用):
+--------+     +--------+     +--------+
| 请求   | --> | 低配模型 | --> | 有限   |
|        |     | Mini    |     | 能力   |
+--------+     +--------+     +--------+

三级降级 (所有远程模型不可用):
+--------+     +--------+     +--------+
| 请求   | --> | 本地模型 | --> | 最小   |
|        |     | 7B      |     | 能力   |
+--------+     +--------+     +--------+

四级降级 (所有模型不可用):
+--------+     +--------+
| 请求   | --> | 降级响应 | --> 直接返回缓存/预设回复
+--------+     +--------+
```

### 熔断器实现

```python
import asyncio
import time
from enum import Enum


class CircuitState(Enum):
    CLOSED = "closed"          # 正常
    OPEN = "open"              # 熔断开启
    HALF_OPEN = "half_open"    # 半开状态（探测是否恢复）


class CircuitBreaker:
    def __init__(
        self,
        failure_threshold: int = 5,
        recovery_timeout: float = 30.0,
        half_open_max_requests: int = 3,
    ):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.half_open_max_requests = half_open_max_requests

        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.last_failure_time = 0.0
        self.half_open_requests = 0

    async def call(self, func: callable, *args, **kwargs) -> Any:
        if self.state == CircuitState.OPEN:
            if time.time() - self.last_failure_time > self.recovery_timeout:
                self.state = CircuitState.HALF_OPEN
                self.half_open_requests = 0
            else:
                raise CircuitBreakerOpen("Circuit breaker is open")

        if self.state == CircuitState.HALF_OPEN:
            if self.half_open_requests >= self.half_open_max_requests:
                raise CircuitBreakerOpen(
                    "Half-open max requests reached"
                )
            self.half_open_requests += 1

        try:
            result = await func(*args, **kwargs)
            self._on_success()
            return result
        except Exception as e:
            self._on_failure()
            raise

    def _on_success(self):
        self.failure_count = 0
        if self.state == CircuitState.HALF_OPEN:
            self.state = CircuitState.CLOSED
            self.half_open_requests = 0

    def _on_failure(self):
        self.failure_count += 1
        self.last_failure_time = time.time()
        if self.failure_count >= self.failure_threshold:
            self.state = CircuitState.OPEN


class CircuitBreakerOpen(Exception):
    pass
```

### 模型降级管理器

```python
class ModelDegradationManager:
    def __init__(self, config: dict):
        self.primary_provider = OpenAIProvider(config["openai"])
        self.fallback_providers = [
            AnthropicProvider(config["anthropic"]),
            LocalProvider(config["local"]),
        ]
        self.circuit_breakers = {
            "primary": CircuitBreaker(
                failure_threshold=3,
                recovery_timeout=30,
            ),
            "fallback_1": CircuitBreaker(
                failure_threshold=5,
                recovery_timeout=60,
            ),
            "fallback_2": CircuitBreaker(
                failure_threshold=10,
                recovery_timeout=120,
            ),
        }
        self.current_level = 0
        self.stats = DegradationStats()

    async def complete(
        self,
        request: CompletionRequest,
    ) -> CompletionResponse:
        try:
            return await self._try_primary(request)
        except (CircuitBreakerOpen, Exception) as e:
            self.stats.record_failure("primary", str(e))
            return await self._fallback_chain(request)

    async def _try_primary(
        self,
        request: CompletionRequest,
    ) -> CompletionResponse:
        breaker = self.circuit_breakers["primary"]
        return await breaker.call(
            self.primary_provider.complete,
            request,
        )

    async def _fallback_chain(
        self,
        request: CompletionRequest,
    ) -> CompletionResponse:
        errors = []

        for i, provider in enumerate(self.fallback_providers):
            provider_name = f"fallback_{i + 1}"
            breaker = self.circuit_breakers.get(provider_name)

            try:
                self.current_level = i + 1
                context = self._adapt_context_for_model(request, provider)

                if breaker:
                    result = await breaker.call(
                        provider.complete,
                        context,
                    )
                else:
                    result = await provider.complete(context)

                self.stats.record_success(provider_name)
                return result

            except CircuitBreakerOpen:
                errors.append(f"{provider_name}: circuit open")
                continue
            except Exception as e:
                errors.append(f"{provider_name}: {str(e)}")
                self.stats.record_failure(provider_name, str(e))
                continue

        return await self._last_resort_response(request, errors)

    def _adapt_context_for_model(
        self,
        request: CompletionRequest,
        provider: BaseProvider,
    ) -> CompletionRequest:
        if provider.model_type == "fast_cheap":
            return CompletionRequest(
                messages=self._compress_messages_for_small_model(
                    request.messages,
                ),
                max_tokens=min(request.max_tokens, 500),
                temperature=0.3,
            )
        return request

    def _compress_messages_for_small_model(
        self,
        messages: list[dict],
    ) -> list[dict]:
        if len(messages) <= 3:
            return messages

        system = messages[0] if messages[0]["role"] == "system" else None
        recent = messages[-2:]
        summary = {
            "role": "user",
            "content": "[Previous conversation summarized: "
                       f"{len(messages) - len(recent) - (1 if system else 0)} messages omitted]"
        }

        result = []
        if system:
            result.append(system)
        result.append(summary)
        result.extend(recent)
        return result

    async def _last_resort_response(
        self,
        request: CompletionRequest,
        errors: list[str],
    ) -> CompletionResponse:
        last_question = request.messages[-1]["content"] if request.messages else ""

        return CompletionResponse(
            content=(
                f"I'm currently experiencing connectivity issues with my AI models. "
                f"I can still help with basic queries. "
                f"Please try again later for more complex tasks.\n\n"
                f"Your input was: {last_question[:200]}..."
            ),
            model="fallback_response",
            from_cache=True,
            degradation_level=4,
        )


class DegradationStats:
    def __init__(self):
        self.failures: dict[str, list[dict]] = {}
        self.successes: dict[str, int] = {}

    def record_failure(self, provider: str, error: str):
        if provider not in self.failures:
            self.failures[provider] = []
        self.failures[provider].append({
            "time": time.time(),
            "error": error,
        })
        if len(self.failures[provider]) > 100:
            self.failures[provider].pop(0)

    def record_success(self, provider: str):
        self.successes[provider] = self.successes.get(provider, 0) + 1

    def get_error_rate(self, provider: str, window: int = 300) -> float:
        now = time.time()
        failures = [
            f for f in self.failures.get(provider, [])
            if now - f["time"] < window
        ]
        successes = self.successes.get(provider, 0)
        total = len(failures) + successes
        return len(failures) / total if total > 0 else 0
```

### 功能级降级

```python
class FunctionalDegradation:
    """按功能维度降级，而非粗暴切换模型"""

    CAPABILITY_LEVELS = {
        0: {  # 全部可用
            "reasoning": True,
            "tool_calls": True,
            "long_context": True,
            "streaming": True,
            "multi_modal": True,
        },
        1: {  # 主模型故障
            "reasoning": True,
            "tool_calls": True,
            "long_context": False,
            "streaming": True,
            "multi_modal": False,
        },
        2: {  # 高配模型都不可用
            "reasoning": False,
            "tool_calls": True,
            "long_context": False,
            "streaming": False,
            "multi_modal": False,
        },
        3: {  # 仅基本问答
            "reasoning": False,
            "tool_calls": False,
            "long_context": False,
            "streaming": False,
            "multi_modal": False,
        },
    }

    def __init__(self, level: int = 0):
        self.level = level
        self.capabilities = self.CAPABILITY_LEVELS[level]

    def has_capability(self, capability: str) -> bool:
        return self.capabilities.get(capability, False)

    async def execute_agent_task(
        self,
        task: AgentTask,
    ) -> AgentResult:
        if task.requires_reasoning and not self.has_capability("reasoning"):
            task = self._simplify_task(task)

        if task.requires_tools and not self.has_capability("tool_calls"):
            return AgentResult(
                success=False,
                answer="Tool execution is currently unavailable",
                degradation_level=self.level,
            )

        if task.requires_long_context and not self.has_capability("long_context"):
            task.context = self._truncate_context(task.context)

        return await self._execute_with_current_model(task)

    def _simplify_task(self, task: AgentTask) -> AgentTask:
        task.config.max_steps = min(task.config.max_steps, 3)
        task.config.max_tool_calls = min(task.config.max_tool_calls, 2)
        task.config.temperature = 0.2  # 降低创造性，提高确定性

        return task

    def _truncate_context(self, context: AgentContext) -> AgentContext:
        max_tokens = 8000
        while context.total_tokens > max_tokens:
            context.history = context.history[1:]
            context.recalculate_tokens()
        return context
```

### 降级策略对比

| 降级级别 | 触发条件 | 模型 | 功能限制 | 用户体验影响 | 恢复方式 |
|---------|---------|------|---------|------------|---------|
| 0-正常 | 无 | 主模型 (Sonnet) | 全部可用 | 无 | - |
| 1-轻降 | 主模型超时/限流 | 备模型 (GPT-4o) | 无 | 几乎无感 | 自动探测恢复 |
| 2-中降 | 高配模型都不可用 | 小模型 (Mini) | 无工具调用、无长上下文 | 明显变笨，不能操作 | 手动/自动 |
| 3-重降 | 所有远程模型不可用 | 本地模型 (7B) | 仅基本问答 | 回答质量显著下降 | 手动恢复 |
| 4-停机 | 本地模型也不可用 | 无 | 返回预设消息 | 不能回答问题 | 页面展示维护 |

### 第一层：降级的触发条件

class DegradationTrigger:
    @staticmethod
    def should_degrade(request_stats: RequestStats) -> int:
        triggers = []

        if request_stats.p95_latency > 10000:
            triggers.append("latency")
        if request_stats.error_rate > 0.05:
            triggers.append("error_rate")
        if request_stats.token_quota_remaining < 100000:
            triggers.append("quota")
        if request_stats.circuit_breaker_state == "open":
            triggers.append("circuit_breaker")

        if len(triggers) >= 3:
            return 2
        if len(triggers) >= 1:
            return 1
        return 0

    @staticmethod
    async def check_and_degrade(
        degradation_manager: ModelDegradationManager,
        stats_collector: StatsCollector,
    ):
        stats = await stats_collector.get_recent_stats(window=60)
        level = DegradationTrigger.should_degrade(stats)

            degradation_manager.current_level = level

## 工程实践

```python
class ProductionDegradationSystem:
    def __init__(self, config: dict):
        self.manager = ModelDegradationManager(config)
        self.functional = FunctionalDegradation()
        self.trigger = DegradationTrigger()
        self.stats = StatsCollector(config["stats"])

    async def process_request(
        self,
        request: UserRequest,
    ) -> AgentResponse:
        level = await self.trigger.check_and_degrade(self.manager, self.stats)

        if level >= 4:
            return AgentResponse(
                success=False,
                content="System is currently undergoing maintenance. "
                        "Please try again later.",
                degradation_level=4,
            )

        try:
            degraded_request = self.functional._simplify_task(request)
            result = await self.manager.complete(degraded_request)
            return AgentResponse(
                success=True,
                content=result.content,
                degradation_level=self.manager.current_level,
                model_used=result.model,
            )
        except Exception as e:
            return AgentResponse(
                success=False,
                content=f"An error occurred: {str(e)}",
                degradation_level=4,
            )

    async def recovery_check(self):
        """定期检查是否可以恢复"""
        stats = await self.stats.get_recent_stats(window=60)
        if stats.error_rate < 0.01 and stats.p95_latency < 3000:
            self.manager.current_level = max(0, self.manager.current_level - 1)
            await alerting.send(
                f"Auto-recovery: degradation level reduced to "
                f"{self.manager.current_level}"
            )
```

## 常见误区

- **降级就是换模型**：降级应该系统性地降低能力边界，而不是简单换个 API。换模型后输出格式可能变更
- **不做降级通知**：降级后用户可能不理解为什么智能体变笨了，需要适当提示"当前服务降级中"
- **降级后不尝试恢复**：降到低配模型后永远不检查主模型是否恢复，需要定期探测
- **降级粒度过粗**：不应该全量切换，可以按用户 ID Hash 百分比降级，让部分用户体验降级
- **忽略降级成本**：降级到低配模型可能因为需要更多轮交互反而增加总成本
