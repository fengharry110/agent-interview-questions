# 如何处理中途失败和回滚？

## 面试问题

如何处理中途失败和回滚？

## 考察重点

- 是否理解失败恢复的重要性
- 是否能设计有效的回滚机制
- 是否了解补偿事务模式
- 是否有处理部分失败的经验

## 核心结论

中途失败和回滚是智能体可靠性的关键。核心策略是**最小化回滚范围** + **补偿事务** + **幂等性设计**。不是所有操作都能回滚，更重要的是设计可恢复的执行计划。

---

## 失败的类型

### 类型一：可恢复失败

```
网络超时：等待后重试
服务暂时不可用：稍后重试
资源暂时不足：等待后重试
```

### 类型二：不可恢复失败

```
参数错误：无法执行
权限不足：无法执行
目标不可达：需要重新规划
```

### 类型三：部分失败

```
某些步骤成功，某些步骤失败
结果部分满足要求
```

## 回滚策略

### 策略一：检查点回滚

```python
class CheckpointRollback:
    def __init__(self):
        self.checkpoints: dict[str, Checkpoint] = {}

    def create_checkpoint(self, task_id: str, state: TaskState):
        self.checkpoints[task_id] = Checkpoint(
            task_id=task_id,
            state=copy.deepcopy(state),
            timestamp=datetime.now()
        )

    def rollback(self, task_id: str) -> TaskState:
        checkpoint = self.checkpoints.get(task_id)
        if not checkpoint:
            raise NoCheckpointError(f"No checkpoint for task {task_id}")

        return copy.deepcopy(checkpoint.state)

    def cleanup(self, task_id: str):
        if task_id in self.checkpoints:
            del self.checkpoints[task_id]
```

### 策略二：补偿事务

```python
class CompensationManager:
    def __init__(self):
        self.compensation_actions: dict[str, CompensationAction] = {}

    def register_compensation(
        self,
        action_id: str,
        action: Action,
        compensation: Callable
    ):
        self.compensation_actions[action_id] = CompensationAction(
            action=action,
            compensation=compensation,
            status="pending"
        )

    async def execute_compensations(
        self,
        failed_action_id: str
    ) -> CompensationResult:
        # 找到需要补偿的所有操作
        to_compensate = []
        for action_id, comp in self.compensation_actions.items():
            if comp.status == "pending" and action_id != failed_action_id:
                to_compensate.append(comp)

        # 按相反顺序执行补偿
        results = []
        for comp in reversed(to_compensate):
            try:
                await comp.compensation()
                comp.status = "completed"
                results.append(CompensationResult(action_id=comp.action.id, success=True))
            except Exception as e:
                comp.status = "failed"
                results.append(CompensationResult(
                    action_id=comp.action.id,
                    success=False,
                    error=str(e)
                ))

        return CompensationResult(success=all(r.success for r in results), results=results)
```

### 策略三：幂等性设计

```python
class IdempotentExecutor:
    def __init__(self):
        self.executed_operations: set[str] = set()

    async def execute_idempotent(
        self,
        operation: Operation,
        idempotency_key: str
    ) -> OperationResult:
        # 检查是否已经执行过
        if idempotency_key in self.executed_operations:
            return OperationResult(
                success=True,
                already_executed=True,
                result=self.get_cached_result(idempotency_key)
            )

        # 执行操作
        result = await operation.execute()

        # 缓存结果
        self.cache_result(idempotency_key, result)
        self.executed_operations.add(idempotency_key)

        return result
```

## 执行计划恢复

### 恢复流程

```python
class ExecutionRecovery:
    def __init__(self):
        self.checkpoint_manager = CheckpointManager()
        self.compensation_manager = CompensationManager()

    async def recover(
        self,
        task_id: str,
        execution: Execution
    ) -> RecoveryResult:
        # 1. 分析失败原因
        failure = self.analyze_failure(execution.last_error)

        # 2. 确定恢复策略
        if failure.type == "retryable":
            return await self.retry(execution)

        elif failure.type == "recoverable":
            return await self.recover_with_compensation(execution)

        elif failure.type == "replan":
            return await self.replan(execution)

        else:
            return await self.rollback(execution)

    async def recover_with_compensation(
        self,
        execution: Execution
    ) -> RecoveryResult:
        # 执行补偿
        comp_result = await self.compensation_manager.execute_compensations(
            execution.failed_action_id
        )

        if comp_result.success:
            # 重新执行失败的操作
            return await self.retry_from_checkpoint(execution)
        else:
            # 补偿失败，需要更激进的恢复
            return await self.full_rollback(execution)
```

### 部分恢复

```python
class PartialRecovery:
    def analyze_recoverable_subset(
        self,
        execution: Execution
    ) -> list[SubTask]:
        """分析哪些子任务可以恢复"""
        completed = []
        failed = []
        blocked = []

        for subtask in execution.subtasks:
            if subtask.status == "completed":
                completed.append(subtask)
            elif subtask.status == "failed":
                if subtask.error.type in ["retryable", "compensable"]:
                    blocked.append(subtask)
                else:
                    blocked.append(subtask)
            else:
                blocked.append(subtask)

        # 检查依赖关系
        recoverable = []
        for subtask in blocked:
            deps = subtask.dependencies
            if all(dep.status == "completed" for dep in deps):
                recoverable.append(subtask)

        return recoverable
```

## 重试策略

### 指数退避

```python
class ExponentialBackoffRetry:
    def __init__(
        self,
        base_delay: float = 1.0,
        max_delay: float = 60.0,
        max_attempts: int = 5,
        backoff_factor: float = 2.0
    ):
        self.base_delay = base_delay
        self.max_delay = max_delay
        self.max_attempts = max_attempts
        self.backoff_factor = backoff_factor

    def should_retry(self, attempt: int, error: Exception) -> bool:
        if attempt >= self.max_attempts:
            return False

        if isinstance(error, NonRetryableError):
            return False

        return True

    def get_delay(self, attempt: int) -> float:
        delay = self.base_delay * (self.backoff_factor ** attempt)
        return min(delay, self.max_delay)

    async def retry(self, func: callable, *args, **kwargs) -> Any:
        last_error = None

        for attempt in range(self.max_attempts):
            try:
                return await func(*args, **kwargs)
            except Exception as e:
                last_error = e

                if not self.should_retry(attempt, e):
                    raise last_error

                delay = self.get_delay(attempt)
                await asyncio.sleep(delay)

        raise last_error
```

### 重试预算

```python
class RetryBudget:
    def __init__(self, max_retries_per_task: int = 10):
        self.max_retries = max_retries_per_task
        self.retry_counts: dict[str, int] = {}

    def can_retry(self, task_id: str) -> bool:
        return self.retry_counts.get(task_id, 0) < self.max_retries

    def record_retry(self, task_id: str):
        self.retry_counts[task_id] = self.retry_counts.get(task_id, 0) + 1

    def reset(self, task_id: str):
        self.retry_counts[task_id] = 0
```

## 失败处理最佳实践

### 1. 失败分类要准确

```python
class ErrorClassifier:
    def classify(self, error: Exception) -> ErrorType:
        if isinstance(error, TimeoutError):
            return ErrorType.RETRYABLE
        if isinstance(error, ValidationError):
            return ErrorType.NON_RETRYABLE
        if isinstance(error, ResourceExhaustedError):
            return ErrorType.CIRCUIT_BREAKER
        if isinstance(error, CompensatableError):
            return ErrorType.COMPENSATABLE
        return ErrorType.UNKNOWN
```

### 2. 降级处理

```python
class DegradationHandler:
    def __init__(self):
        self.degradation_strategies = {
            "high": self.full_functionality,
            "medium": self.reduced_functionality,
            "low": self.basic_functionality,
            "critical": self.minimal_functionality
        }

    def get_strategy(self, health: float) -> str:
        if health > 0.8:
            return "high"
        elif health > 0.5:
            return "medium"
        elif health > 0.2:
            return "low"
        else:
            return "critical"
```
