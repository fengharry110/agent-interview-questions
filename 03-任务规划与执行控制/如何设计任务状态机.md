# 如何设计任务状态机？

## 面试问题

如何设计任务状态机？

## 考察重点

- 是否理解状态机在任务管理中的作用
- 是否能设计清晰的状态转换规则
- 是否了解状态持久化和恢复机制
- 是否有处理复杂任务状态的经验

## 核心结论

任务状态机是智能体执行控制的核心，它定义了任务从创建到完成的完整生命周期。设计良好的状态机能清晰追踪任务进度、支持中断恢复、便于问题排查。

## 任务状态定义

### 基本状态

```python
from enum import Enum

class TaskState(Enum):
    CREATED = "created"           # 任务已创建
    PENDING = "pending"           # 等待执行
    RUNNING = "running"           # 执行中
    PAUSED = "paused"            # 暂停（可恢复）
    WAITING = "waiting"          # 等待外部条件
    COMPLETED = "completed"       # 已完成
    FAILED = "failed"            # 失败
    CANCELLED = "cancelled"      # 已取消
```

## 状态机实现

### 核心结构

```python
class TaskStateMachine:
    def __init__(self, task_id: str):
        self.task_id = task_id
        self.state = TaskState.CREATED
        self.history: list[StateTransition] = []
        self.context: dict = {}

    def transition(self, new_state: TaskState, reason: str = None) -> bool:
        """执行状态转换"""
        # 验证转换是否合法
        if not self.can_transition(new_state):
            raise InvalidTransitionError(
                f"Cannot transition from {self.state} to {new_state}"
            )

        # 记录转换历史
        self.history.append(StateTransition(
            from_state=self.state,
            to_state=new_state,
            reason=reason,
            timestamp=datetime.now()
        ))

        # 执行转换
        old_state = self.state
        self.state = new_state

        # 执行转换后的钩子
        self.on_exit(old_state)
        self.on_enter(new_state)

        return True

    def can_transition(self, new_state: TaskState) -> bool:
        """检查状态转换是否合法"""
        valid_transitions = {
            TaskState.CREATED: [TaskState.PENDING, TaskState.CANCELLED],
            TaskState.PENDING: [TaskState.RUNNING, TaskState.CANCELLED],
            TaskState.RUNNING: [
                TaskState.COMPLETED,
                TaskState.FAILED,
                TaskState.PAUSED,
                TaskState.WAITING,
                TaskState.CANCELLED
            ],
            TaskState.WAITING: [TaskState.RUNNING, TaskState.CANCELLED],
            TaskState.PAUSED: [TaskState.RUNNING, TaskState.CANCELLED],
            TaskState.COMPLETED: [],  # 终态
            TaskState.FAILED: [TaskState.PENDING],  # 可以重试
            TaskState.CANCELLED: [],  # 终态
        }

        return new_state in valid_transitions.get(self.state, [])
```

### 状态转换钩子

```python
class TaskStateMachine:
    def on_enter(self, state: TaskState):
        """进入状态时的处理"""
        if state == TaskState.RUNNING:
            self.context["start_time"] = datetime.now()

        elif state == TaskState.WAITING:
            # 等待时保存当前进度
            self.save_checkpoint()

        elif state == TaskState.COMPLETED:
            self.context["end_time"] = datetime.now()
            self.context["duration"] = (
                self.context["end_time"] - self.context["start_time"]
            )
            self.cleanup()

    def on_exit(self, state: TaskState):
        """离开状态时的处理"""
        if state == TaskState.RUNNING:
            # 保存进度
            self.save_checkpoint()
```

## 子任务状态机

### 层级结构

```python
class SubTaskStateMachine:
    def __init__(self, parent_task_id: str, sub_task_id: str):
        self.parent_task_id = parent_task_id
        self.sub_task_id = sub_task_id
        self.state = SubTaskState.PENDING
        self.dependencies: list[str] = []

    def is_runnable(self) -> bool:
        """检查依赖是否满足"""
        for dep_id in self.dependencies:
            dep_task = get_task(dep_id)
            if dep_task.state != SubTaskState.COMPLETED:
                return False
        return self.state == SubTaskState.PENDING
```

### 并行执行控制

```python
class ParallelTaskExecutor:
    def __init__(self, max_parallel: int = 5):
        self.max_parallel = max_parallel
        self.running_tasks: set[str] = set()

    def can_start(self, task: SubTaskStateMachine) -> bool:
        """检查是否可以启动任务"""
        if not task.is_runnable():
            return False

        if len(self.running_tasks) >= self.max_parallel:
            return False

        return True

    def on_task_complete(self, task_id: str):
        """任务完成时的处理"""
        self.running_tasks.discard(task_id)

        # 触发依赖此任务的其他任务
        for waiting_task in self.get_waiting_tasks():
            if task_id in waiting_task.dependencies:
                if waiting_task.is_runnable():
                    self.start_task(waiting_task)
```

## 状态持久化

### 检查点机制

```python
class CheckpointManager:
    def save_checkpoint(self, task: TaskStateMachine):
        """保存任务检查点"""
        checkpoint = Checkpoint(
            task_id=task.task_id,
            state=task.state,
            context=task.context,
            history=task.history,
            timestamp=datetime.now()
        )

        # 持久化到存储
        storage.save(f"checkpoint/{task.task_id}", checkpoint)

        # 更新任务索引
        index.add_or_update(task.task_id, checkpoint.timestamp)

    async def restore(self, task_id: str) -> TaskStateMachine:
        """恢复任务"""
        checkpoint = storage.load(f"checkpoint/{task_id}")

        task = TaskStateMachine(task_id)
        task.state = checkpoint.state
        task.context = checkpoint.context
        task.history = checkpoint.history

        return task
```

### 定时检查点

```python
class AutoCheckpoint:
    def __init__(self, interval_seconds: int = 60):
        self.interval = interval_seconds
        self.last_checkpoint_time = {}

    def maybe_checkpoint(self, task: TaskStateMachine):
        """条件检查点"""
        now = datetime.now()

        if task.task_id not in self.last_checkpoint_time:
            self.last_checkpoint_time[task.task_id] = now
            return

        elapsed = (now - self.last_checkpoint_time[task.task_id]).seconds

        # 定期检查点
        if elapsed >= self.interval:
            self.save_checkpoint(task)
            self.last_checkpoint_time[task.task_id] = now

        # 关键步骤后立即检查点
        elif self.is_critical_step(task):
            self.save_checkpoint(task)
            self.last_checkpoint_time[task.task_id] = now
```

## 状态机在 Claude Code 中的体现

Claude Code 的任务状态管理：

```typescript
interface TaskState {
  id: string;
  status: 'pending' | 'running' | 'completed' | 'failed' | 'cancelled';
  currentStep: number;
  maxSteps: number;
  checkpoint: Checkpoint | null;
  consecutiveFailures: number;
}

class TaskExecutor {
  private state: TaskState;

  async transition(newStatus: TaskState['status'], reason?: string) {
    const validTransitions = this.getValidTransitions(this.state.status);

    if (!validTransitions.includes(newStatus)) {
      throw new Error(`Invalid transition: ${this.state.status} -> ${newStatus}`);
    }

    // 保存检查点
    if (this.shouldSaveCheckpoint(newStatus)) {
      await this.saveCheckpoint();
    }

    this.state.status = newStatus;

    // 状态特定处理
    if (newStatus === 'running') {
      this.state.currentStep++;
    }
  }

  private shouldSaveCheckpoint(newStatus: string): boolean {
    return ['completed', 'failed', 'waiting'].includes(newStatus);
  }
}
```

## 状态机设计最佳实践

### 1. 状态要正交

```
❌ 不好：RUNNING + WAITING_FOR_USER_INPUT
✅ 好：RUNNING 状态，加上 context.waiting_for = "user_input"
```

### 2. 终态要明确

```python
TERMINAL_STATES = {TaskState.COMPLETED, TaskState.CANCELLED}

def is_terminal(state: TaskState) -> bool:
    return state in TERMINAL_STATES
```

### 3. 错误要有具体分类

```python
class FailureReason(Enum):
    TIMEOUT = "timeout"
    RESOURCE_NOT_FOUND = "resource_not_found"
    PERMISSION_DENIED = "permission_denied"
    MODEL_ERROR = "model_error"
    USER_CANCELLED = "user_cancelled"
    UNKNOWN = "unknown"
```
