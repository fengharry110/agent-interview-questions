# 如何防止智能体无限循环？

## 面试问题

如何防止智能体无限循环？

## 考察重点

- 是否理解无限循环的本质原因
- 是否能设计有效的检测机制
- 是否了解防止循环的策略
- 是否有处理边缘情况的经验

## 核心结论

无限循环的本质是**反馈闭环失效**：智能体不断执行相同的操作序列但没有进展。防止无限循环需要**检测机制**（识别重复模式）+**中断机制**（强制停止或重规划）。

## 无限循环的类型

### 类型一：重复执行

```
Agent: 执行操作A
Result: 没有进展
Agent: 执行操作A（再次）
Result: 仍然没有进展
Agent: 执行操作A（第三次）
...
```

### 类型二：振荡

```
Agent: 执行操作A
Result: 状态变成X
Agent: 执行操作B
Result: 状态变成Y
Agent: 执行操作A
Result: 状态变成X
Agent: 执行操作B
Result: 状态变成Y
...
在两个状态之间振荡
```

### 类型三：螺旋

```
Agent: 执行操作A
Result: 小幅进展
Agent: 执行操作A（改进版）
Result: 小幅进展（但方向错了）
Agent: 执行操作A（再改进版）
Result: 越来越偏离目标
...
```

## 循环检测机制

### 重复检测

```python
class LoopDetector:
    def __init__(self, window_size: int = 10):
        self.window_size = window_size
        self.recent_actions: deque = deque(maxlen=window_size)
        self.recent_results: deque = deque(maxlen=window_size)

    def record(self, action: Action, result: ActionResult):
        self.recent_actions.append(action)
        self.recent_results.append(result)

    def detect_loop(self) -> LoopDetectionResult:
        # 检测完全重复
        if self.is_exact_repeat():
            return LoopDetectionResult(
                type="exact_repeat",
                confidence=1.0,
                repeated_actions=self.get_repeated_sequence()
            )

        # 检测振荡
        if self.is_oscillation():
            return LoopDetectionResult(
                type="oscillation",
                confidence=0.9,
                oscillation_states=self.get_oscillation_states()
            )

        # 检测进展缓慢
        if self.is_spiral():
            return LoopDetectionResult(
                type="spiral",
                confidence=0.7,
                direction=self.get_spiral_direction()
            )

        return None

    def is_exact_repeat(self) -> bool:
        """检测完全重复"""
        if len(self.recent_actions) < 3:
            return False

        # 检查最后几个动作是否相同
        recent = list(self.recent_actions)[-3:]
        return len(set(recent)) == 1

    def is_oscillation(self) -> bool:
        """检测振荡模式 A-B-A-B"""
        if len(self.recent_actions) < 4:
            return False

        actions = list(self.recent_actions)[-4:]
        return actions[0] == actions[2] and actions[1] == actions[3]
```

### 状态距离检测

```python
class StateDistanceDetector:
    def __init__(self, embedding_model):
        self.embedding_model = embedding_model

    def detect_spiral(self, state_history: list[State]) -> bool:
        """检测螺旋式偏离"""
        if len(state_history) < 5:
            return False

        # 计算相邻状态之间的距离
        distances = []
        for i in range(len(state_history) - 1):
            dist = self.state_distance(state_history[i], state_history[i + 1])
            distances.append(dist)

        # 检查距离是否递增
        increasing_count = sum(
            1 for i in range(len(distances) - 1)
            if distances[i + 1] > distances[i]
        )

        # 如果大部分距离都在增加，说明在偏离
        return increasing_count / len(distances) > 0.7

    def state_distance(self, state1: State, state2: State) -> float:
        """计算两个状态的距离"""
        emb1 = self.embedding_model.embed(state1.to_vector())
        emb2 = self.embedding_model.embed(state2.to_vector())
        return cosine_distance(emb1, emb2)
```

## 防止循环的策略

### 策略一：强制停止

```python
class LoopProtection:
    def __init__(self, max_repeat_actions: int = 3):
        self.max_repeat_actions = max_repeat_actions

    def should_intervene(self, detection: LoopDetectionResult) -> Intervention:
        if detection.type == "exact_repeat":
            if detection.confidence > 0.9:
                return Intervention(
                    action="stop",
                    reason="检测到重复操作序列",
                    alternative="建议重新规划或转人工"
                )

        elif detection.type == "oscillation":
            return Intervention(
                action="redirect",
                reason="检测到操作振荡",
                alternative="尝试不同的操作顺序"
            )

        elif detection.type == "spiral":
            return Intervention(
                action="replan",
                reason="检测到目标偏离",
                alternative="重新分析目标，制定新计划"
            )

        return None
```

### 策略二：重规划

```python
class Replanner:
    def __init__(self):
        self.max_replans = 3

    def replan(self, task: Task, context: ExecutionContext) -> Plan:
        # 分析失败原因
        failure_analysis = self.analyze_loop_failure(context)

        # 生成新的计划
        new_plan = self.generate_alternative_plan(
            task=task,
            constraints=failure_analysis.constraints,
            attempts=context.replan_count + 1
        )

        return new_plan

    def analyze_loop_failure(self, context: ExecutionContext) -> FailureAnalysis:
        # 分析失败模式
        loop_type = self.classify_loop_type(context)

        if loop_type == "wrong_approach":
            return FailureAnalysis(
                type="wrong_approach",
                message="当前方法可能不对，需要换思路",
                constraints=["不要重复相同操作"]
            )

        elif loop_type == "insufficient_info":
            return FailureAnalysis(
                type="insufficient_info",
                message="信息不足，无法正确规划",
                constraints=["先收集更多信息"]
            )

        return FailureAnalysis(
            type="unknown",
            message="未知原因导致循环",
            constraints=[]
        )
```

class RandomizedExecutor:
    def execute_with_randomization(self, action: Action) -> ActionResult:
        # 给某些操作添加随机因素
        if action.type == "search" and self.should_randomize(action):
            action.params["offset"] = random.randint(0, 10)
            action.params["order_by"] = random.choice(["relevance", "time", "random"])

        if action.type == "retry":
            # 重试时等待随机时间
            time.sleep(random.uniform(0.1, 1.0))

        return self.base_executor.execute(action)

    def should_randomize(self, action: Action) -> bool:
        # 连续失败时增加随机化
        if self.context.consecutive_failures > 2:
            return True

        # 某些操作默认随机化
        return action.type in ["search", "query", "fetch"]

## Circuit Breaker 模式

### 实现

```python
class CircuitBreaker:
    def __init__(
        self,
        failure_threshold: int = 5,
        recovery_timeout: int = 60,
        half_open_attempts: int = 3
    ):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.half_open_attempts = half_open_attempts

        self.failure_count = 0
        self.last_failure_time = None
        self.state = "closed"  # closed, open, half_open

    def call(self, func: callable) -> Any:
        if self.state == "open":
            if self.should_attempt_reset():
                self.state = "half_open"
            else:
                raise CircuitBreakerOpenError("Circuit breaker is open")

        try:
            result = func()
            self.on_success()
            return result
        except Exception as e:
            self.on_failure()
            raise e

    def on_success(self):
        self.failure_count = 0
        self.state = "closed"

    def on_failure(self):
        self.failure_count += 1
        self.last_failure_time = datetime.now()

        if self.failure_count >= self.failure_threshold:
            self.state = "open"

    def should_attempt_reset(self) -> bool:
        elapsed = (datetime.now() - self.last_failure_time).seconds
        return elapsed >= self.recovery_timeout
```

## Claude Code 的循环保护

Claude Code 的循环检测机制：

```typescript
interface LoopProtection {
  consecutiveSameActions: number;
  oscillationPattern: string[] | null;
  progressRate: number;
}

function detectLoop(state: AgentState): LoopProtection | null {
  const recentActions = state.recentActions.slice(-5);

  // 检测重复操作
  if (allSame(recentActions)) {
    return {
      consecutiveSameActions: recentActions.length,
      oscillationPattern: null,
      progressRate: calculateProgressRate(state)
    };
  }

  // 检测振荡模式
  if (isOscillating(recentActions)) {
    return {
      consecutiveSameActions: 0,
      oscillationPattern: extractOscillationPattern(recentActions),
      progressRate: 0
    };
  }

  return null;
}

function handleLoop(detection: LoopProtection): Intervention {
  if (detection.consecutiveSameActions >= 3) {
    return {
      type: 'replan',
      reason: 'repeated_same_action',
      message: '检测到重复操作，尝试不同方法'
    };
  }

  if (detection.oscillationPattern) {
    return {
      type: 'redirect',
      reason: 'oscillation_detected',
      message: '检测到振荡模式，重新分析目标'
    };
  }

  if (detection.progressRate < 0.01) {
    return {
      type: 'stop',
      reason: 'no_progress',
      message: '长时间无进展，停止执行'
    };
  }
}
```

## 监控和告警

```python
class LoopMonitor:
    def __init__(self):
        self.metrics = MetricsClient()

    def record_execution(
        self,
        task_id: str,
        action: Action,
        loop_detected: bool
    ):
        self.metrics.increment(
            "agent.loop.detected",
            tags={"task_id": task_id, "loop_type": loop_detected.type}
        )

        if loop_detected:
            self.alert(f"任务 {task_id} 检测到循环: {loop_detected.type}")

    def get_loop_statistics(self, time_window: timedelta) -> dict:
        """获取循环统计"""
        return {
            "loop_detection_rate": self.metrics.get_rate(
                "agent.loop.detected",
                time_window
            ),
            "most_common_loop_types": self.metrics.get_top(
                "agent.loop.detected",
                k=5
            ),
            "avg_loops_per_task": self.metrics.get_average(
                "agent.loop.count",
                time_window
            )
        }
```
