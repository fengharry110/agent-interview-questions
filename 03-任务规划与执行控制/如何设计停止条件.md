# 如何设计停止条件？

## 面试问题

如何设计停止条件？

## 考察重点

- 是否理解停止条件的重要性（防止无限循环、浪费资源）
- 是否能设计多维度的停止条件
- 是否了解显式停止和隐式停止的区别
- 是否有处理"部分完成"场景的经验

## 核心结论

停止条件是智能体的"安全阀"，需要从多个维度设计：**资源限制**（轮次、Token、时间）、**结果评估**（任务完成、置信度）、**用户控制**（主动停止）。好的停止条件能防止智能体无限运行，同时允许合理的"模糊完成"。

## 停止条件的类型

### 类型一：资源限制

```
硬限制（必须停止）：
- 最大轮次
- 最大 Token 数
- 最大执行时间
- 最大内存使用

软限制（可以继续但要谨慎）：
- 接近资源上限
- 性能下降
```

### 类型二：结果评估

```
任务完成：
- 模型返回完成信号
- 所有子任务完成
- 用户确认结果

任务失败：
- 连续失败超过阈值
- 错误不可恢复
- 目标无法达成
```

### 类型三：用户控制

```
主动停止：
- 用户发送停止指令
- 用户关闭窗口
- 用户开始新任务

被动停止：
- 用户长时间无响应
- 用户会话超时
```

## 停止条件实现

### 多维度检查

```python
class StopConditionChecker:
    def __init__(self, config: StopConditionConfig):
        self.max_turns = config.max_turns
        self.max_tokens = config.max_tokens
        self.max_time_seconds = config.max_time_seconds
        self.max_consecutive_failures = config.max_consecutive_failures

    def should_stop(self, context: ExecutionContext) -> StopReason:
        # 1. 检查轮次限制
        if context.turn_count >= self.max_turns:
            return StopReason(
                type="resource",
                reason="max_turns_exceeded",
                message=f"已达到最大轮次 {self.max_turns}"
            )

        # 2. 检查 Token 限制
        if context.total_tokens >= self.max_tokens:
            return StopReason(
                type="resource",
                reason="max_tokens_exceeded",
                message=f"已使用 {context.total_tokens} tokens"
            )

        # 3. 检查时间限制
        elapsed = (datetime.now() - context.start_time).seconds
        if elapsed >= self.max_time_seconds:
            return StopReason(
                type="resource",
                reason="max_time_exceeded",
                message=f"执行时间 {elapsed}s 超过限制"
            )

        # 4. 检查连续失败
        if context.consecutive_failures >= self.max_consecutive_failures:
            return StopReason(
                type="failure",
                reason="consecutive_failures",
                message=f"连续失败 {self.max_consecutive_failures} 次"
            )

        # 5. 检查任务完成
        if context.task.is_completed():
            return StopReason(
                type="completion",
                reason="task_completed",
                message="任务已完成"
            )

        # 6. 检查用户停止
        if context.user_stop_requested:
            return StopReason(
                type="user",
                reason="user_requested",
                message="用户请求停止"
            )

        return None  # 不停止
```

### Claude Code 的停止条件

Claude Code 的停止检查逻辑：

```typescript
interface StopCondition {
  maxTurns: number;
  maxTokens: number;
  maxTime: number;
  circuitBreakerThreshold: number;
}

function shouldStop(state: AgentState): StopReason | null {
  // 1. 用户主动停止
  if (state.stopSignalReceived) {
    return { type: 'user', reason: 'stop_signal' };
  }

  // 2. 达到最大轮次
  if (state.turnCount >= state.maxTurns) {
    return { type: 'resource', reason: 'max_turns' };
  }

  // 3. Token 预算耗尽
  if (state.tokenBudget <= 0) {
    return { type: 'resource', reason: 'token_budget_exhausted' };
  }

  // 4. Circuit breaker 触发
  if (state.hasAttemptedReactiveCompact && state.consecutiveFailures > 0) {
    return { type: 'failure', reason: 'circuit_breaker' };
  }

  // 5. 任务完成
  if (state.lastModelOutput.type === 'final_response') {
    return { type: 'completion', reason: 'final_response' };
  }

  // 6. 目标不可达
  if (state.goalUnreachable) {
    return { type: 'failure', reason: 'goal_unreachable' };
  }

  return null;
}
```

## 停止条件的阈值设计

### 保守策略

```python
class ConservativeStopConfig:
    """保守策略：早点停止，减少资源浪费"""
    max_turns = 10
    max_tokens = 50000
    max_time_seconds = 300  # 5分钟
    max_consecutive_failures = 2

    # 置信度阈值
    min_confidence_to_complete = 0.8
    min_progress_rate = 0.1  # 每轮至少10%进度
```

### 激进策略

```python
class AggressiveStopConfig:
    """激进策略：给更多轮次，追求完成率"""
    max_turns = 50
    max_tokens = 200000
    max_time_seconds = 1800  # 30分钟
    max_consecutive_failures = 5

    min_confidence_to_complete = 0.6
    min_progress_rate = 0.05
```

### 动态调整

```python
class AdaptiveStopCondition:
    def __init__(self, base_config: StopConditionConfig):
        self.config = base_config
        self.task_complexity_estimator = ComplexityEstimator()

    def get_effective_limits(self, task: Task) -> dict:
        """根据任务复杂度动态调整限制"""
        complexity = self.task_complexity_estimator.estimate(task)

        multiplier = {
            "simple": 0.5,
            "medium": 1.0,
            "complex": 2.0,
            "very_complex": 4.0
        }.get(complexity, 1.0)

        return {
            "max_turns": int(self.config.max_turns * multiplier),
            "max_tokens": int(self.config.max_tokens * multiplier),
            "max_time_seconds": int(self.config.max_time_seconds * multiplier)
        }
```

## 部分完成处理

### 检测部分完成

```python
class PartialCompletionDetector:
    def __init__(self):
        self.completion_evaluator = CompletionEvaluator()

    def evaluate_completion(
        self,
        task: Task,
        result: ExecutionResult
    ) -> CompletionStatus:
        evaluation = self.completion_evaluator.evaluate(task, result)

        if evaluation.completion_rate >= 1.0:
            return CompletionStatus.FULL

        elif evaluation.completion_rate >= 0.5:
            return CompletionStatus.PARTIAL

        elif evaluation.completion_rate >= 0.1:
            return CompletionStatus.MINIMAL

        else:
            return CompletionStatus.NONE
```

### 处理策略

```python
def handle_partial_completion(
    task: Task,
    result: ExecutionResult,
    status: CompletionStatus
) -> str:
    if status == CompletionStatus.FULL:
        return result.summary

    elif status == CompletionStatus.PARTIAL:
        return f"""任务部分完成。

已完成：
{format_completed_items(result.completed_items)}

未完成：
{format_incomplete_items(result.incomplete_items)}

建议：
- 继续执行剩余步骤
- 或者先提交当前结果，稍后继续"""

    elif status == CompletionStatus.MINIMAL:
        return f"""任务完成度较低。

已完成：{format_completed_items(result.completed_items)}

遇到了以下问题：
{format_issues(result.issues)}

建议重新规划或转人工处理"""

    return "任务未能完成"
```

## 停止条件的测试

```python
def test_stop_conditions():
    """测试停止条件的各种场景"""

    # 场景1：正常完成
    context = create_context(turn_count=5, task_completed=True)
    checker = StopConditionChecker(default_config)
    assert checker.should_stop(context).type == "completion"

    # 场景2：达到最大轮次
    context = create_context(turn_count=10, max_turns=10)
    assert checker.should_stop(context).reason == "max_turns_exceeded"

    # 场景3：连续失败
    context = create_context(consecutive_failures=3, max_failures=3)
    assert checker.should_stop(context).reason == "consecutive_failures"

    # 场景4：用户停止
    context = create_context(user_stop_requested=True)
    assert checker.should_stop(context).reason == "user_requested"
```
